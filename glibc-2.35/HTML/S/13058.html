  19 #include <sysdep.h>
  20 #include "asm-syntax.h"
  21 
  22 #define PARMS   4       /* no space for saved regs */
  23 #define STR1    PARMS
  24 #define STR2    STR1+4
  25 
  26         .text
  27 ENTRY (strcmp)
  28 
  29     movl    STR1(%esp), %ecx
  30     movl    STR2(%esp), %edx
  31 
  32 L(oop): movb    (%ecx), %al
  33     cmpb    (%edx), %al
  34     jne L(neq)
  35     incl    %ecx
  36     incl    %edx
  37     testb   %al, %al
  38     jnz L(oop)
  39 
  40     xorl    %eax, %eax
  41     /* when strings are equal, pointers rest one beyond
  42        the end of the NUL terminators.  */
  43     ret
  44 
  45 L(neq): movl    $1, %eax
  46     movl    $-1, %ecx
  47     cmovbl  %ecx, %eax
  48 
  49     ret
  50 END (strcmp)
  51 libc_hidden_builtin_def (strcmp)
