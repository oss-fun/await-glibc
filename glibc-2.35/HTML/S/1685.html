  84 static int
  85 do_test (void)
  86 {
  87   int retval = 0;
  88   int i;
  89 
  90   __nss_configure_lookup ("passwd", "test1 test2");
  91 
  92   setpwent ();
  93 
  94   i = 0;
  95   for (struct passwd *p = getpwent (); p != NULL; ++i, p = getpwent ())
  96     {
  97       retval += compare_passwds (i, & pwd_expected[i], p);
  98 
  99       if (p->pw_uid != pwd_expected[i].pw_uid || strcmp (p->pw_name, pwd_expected[i].pw_name) != 0)
 100       {
 101     printf ("FAIL: getpwent for %u.%s returned %u.%s\n",
 102         pwd_expected[i].pw_uid, pwd_expected[i].pw_name,
 103         p->pw_uid, p->pw_name);
 104     retval = 1;
 105     break;
 106       }
 107     }
 108 
 109   endpwent ();
 110 
 111   for (i=0; tests[i].name; i++)
 112     {
 113       struct passwd *p = getpwnam (tests[i].name);
 114       if (strcmp (p->pw_name, tests[i].name) != 0
 115       || p->pw_uid != tests[i].uid)
 116     {
 117       printf("FAIL: getpwnam for %u.%s returned %u.%s\n",
 118          tests[i].uid, tests[i].name,
 119          p->pw_uid, p->pw_name);
 120       retval = 1;
 121     }
 122 
 123       p = getpwuid (tests[i].uid);
 124       if (strcmp (p->pw_name, tests[i].name) != 0
 125       || p->pw_uid != tests[i].uid)
 126     {
 127       printf("FAIL: getpwuid for %u.%s returned %u.%s\n",
 128          tests[i].uid, tests[i].name,
 129          p->pw_uid, p->pw_name);
 130       retval = 1;
 131     }
 132     }
 133 
 134   return retval;
 135 }
