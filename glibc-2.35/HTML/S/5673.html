  28 /* This file provides definitions of _IO_2_1_stdin_, _IO_2_1_stdout_,
  29    and _IO_2_1_stderr_, the default values of stdin, stdout, stderr.
  30    See oldstdfiles.c for glibc 2.0 legacy definitions without wide
  31    character support.  */
  32 
  33 #include "libioP.h"
  34 
  35 #ifdef _IO_MTSAFE_IO
  36 # define DEF_STDFILE(NAME, FD, CHAIN, FLAGS) \
  37   static _IO_lock_t _IO_stdfile_##FD##_lock = _IO_lock_initializer; \
  38   static struct _IO_wide_data _IO_wide_data_##FD \
  39     = { ._wide_vtable = &_IO_wfile_jumps }; \
  40   struct _IO_FILE_plus NAME \
  41     = {FILEBUF_LITERAL(CHAIN, FLAGS, FD, &_IO_wide_data_##FD), \
  42        &_IO_file_jumps};
  43 #else
  44 # define DEF_STDFILE(NAME, FD, CHAIN, FLAGS) \
  45   static struct _IO_wide_data _IO_wide_data_##FD \
  46     = { ._wide_vtable = &_IO_wfile_jumps }; \
  47   struct _IO_FILE_plus NAME \
  48     = {FILEBUF_LITERAL(CHAIN, FLAGS, FD, &_IO_wide_data_##FD), \
  49        &_IO_file_jumps};
  50 #endif
  51 
  52 DEF_STDFILE(_IO_2_1_stdin_, 0, 0, _IO_NO_WRITES);
  53 DEF_STDFILE(_IO_2_1_stdout_, 1, &_IO_2_1_stdin_, _IO_NO_READS);
  54 DEF_STDFILE(_IO_2_1_stderr_, 2, &_IO_2_1_stdout_, _IO_NO_READS+_IO_UNBUFFERED);
  55
