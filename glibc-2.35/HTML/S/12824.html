  26 int
  27 __pthread_kill (pthread_t thread, int sig)
  28 {
  29   struct __pthread *pthread;
  30   struct hurd_signal_detail detail;
  31   struct hurd_sigstate *ss;
  32 
  33   /* Lookup the thread structure for THREAD.  */
  34   pthread = __pthread_getid (thread);
  35   if (pthread == NULL)
  36     return ESRCH;
  37 
  38   if (pthread->kernel_thread == MACH_PORT_DEAD)
  39     /* The pthread ID is still valid but we cannot send a signal any more.  */
  40     return 0;
  41 
  42   ss = _hurd_thread_sigstate (pthread->kernel_thread);
  43   assert (ss);
  44 
  45   if (sig == 0)
  46     return 0;
  47 
  48   detail.exc = 0;
  49   detail.code = sig;
  50   detail.error = 0;
  51 
  52   __spin_lock (&ss->lock);
  53   return _hurd_raise_signal (ss, sig, &detail);
  54 }
