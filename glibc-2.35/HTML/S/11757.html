  35 #include <sysdep.h>
  36 
  37 /* GCC will emit calls to this routine under -mtp=soft.  Linux has an
  38    equivalent helper function (which clobbers fewer registers than
  39    a normal function call) in a high page of memory; tail call to the
  40    helper.  */
  41 
  42     .hidden __aeabi_read_tp
  43 ENTRY (__aeabi_read_tp)
  44 #ifdef ARCH_HAS_HARD_TP
  45     mrc p15, 0, r0, c13, c0, 3
  46     bx  lr
  47 #elif defined(__thumb2__)
  48     movw    r0, #0x0fe0
  49     movt    r0, #0xffff
  50     bx  r0
  51 #else
  52     mov r0, #0xffff0fff
  53     sub pc, r0, #31
  54 #endif
  55 END (__aeabi_read_tp)
