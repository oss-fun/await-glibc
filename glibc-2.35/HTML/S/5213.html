  64 static int
  65 do_test (void)
  66 {
  67   char *path;
  68   int fd = create_temp_file ("tst-fstat.", &path);
  69   TEST_VERIFY_EXIT (fd >= 0);
  70   support_write_file_string (path, "abc");
  71 
  72   bool check_ns = support_stat_nanoseconds (path);
  73   if (!check_ns)
  74     printf ("warning: timestamp with nanoseconds not supported\n");
  75 
  76   struct statx stx;
  77   TEST_COMPARE (statx (fd, path, 0, STATX_BASIC_STATS, &stx), 0);
  78 
  79   test_t tests[] = { stat_check, lstat_check, fstat_check, fstatat_check };
  80 
  81   for (int i = 0; i < array_length (tests); i++)
  82     {
  83       struct stat st;
  84       tests[i](fd, path, &st);
  85 
  86       TEST_COMPARE (stx.stx_dev_major, major (st.st_dev));
  87       TEST_COMPARE (stx.stx_dev_minor, minor (st.st_dev));
  88       TEST_COMPARE (stx.stx_ino, st.st_ino);
  89       TEST_COMPARE (stx.stx_mode, st.st_mode);
  90       TEST_COMPARE (stx.stx_nlink, st.st_nlink);
  91       TEST_COMPARE (stx.stx_uid, st.st_uid);
  92       TEST_COMPARE (stx.stx_gid, st.st_gid);
  93       TEST_COMPARE (stx.stx_rdev_major, major (st.st_rdev));
  94       TEST_COMPARE (stx.stx_rdev_minor, minor (st.st_rdev));
  95       TEST_COMPARE (stx.stx_blksize, st.st_blksize);
  96       TEST_COMPARE (stx.stx_blocks, st.st_blocks);
  97 
  98       TEST_COMPARE (stx.stx_ctime.tv_sec, st.st_ctim.tv_sec);
  99       TEST_COMPARE (stx.stx_mtime.tv_sec, st.st_mtim.tv_sec);
 100       if (check_ns)
 101     {
 102       TEST_COMPARE (stx.stx_ctime.tv_nsec, st.st_ctim.tv_nsec);
 103       TEST_COMPARE (stx.stx_mtime.tv_nsec, st.st_mtim.tv_nsec);
 104     }
 105     }
 106 
 107   return 0;
 108 }
