  18 #ifndef _SYS_ACCT_H
  19 
  20 #define _SYS_ACCT_H 1
  21 #include <features.h>
  22 
  23 #include <bits/types/time_t.h>
  24 
  25 __BEGIN_DECLS
  26 
  27 #define ACCT_COMM 16
  28 
  29 struct acct
  30   {
  31     char ac_comm[ACCT_COMM];        /* Accounting command name.  */
  32     time_t ac_utime;            /* Accounting user time.  */
  33     time_t ac_stime;            /* Accounting system time.  */
  34     time_t ac_etime;            /* Accounting elapsed time.  */
  35     time_t ac_btime;            /* Beginning time.  */
  36     unsigned int ac_uid;        /* Accounting user ID.  */
  37     unsigned int ac_gid;        /* Accounting group ID.  */
  38     unsigned int ac_tty;        /* Controlling tty.  */
  39     /* Please note that the value of the `ac_tty' field, a device number,
  40        is encoded differently in the kernel and for the libc dev_t type.  */
  41     char ac_flag;           /* Accounting flag.  */
  42     long int ac_minflt;         /* Accounting minor pagefaults.  */
  43     long int ac_majflt;         /* Accounting major pagefaults.  */
  44     long int ac_exitcode;       /* Accounting process exitcode.  */
  45   };
  46 
  47 enum
  48   {
  49     AFORK = 0001,       /* Has executed fork, but no exec.  */
  50     ASU = 0002,         /* Used super-user privileges.  */
  51     ACORE = 0004,       /* Dumped core.  */
  52     AXSIG = 0010        /* Killed by a signal.  */
  53   };
  54 
  55 #define AHZ     100
  56 
  57 
  58 /* Switch process accounting on and off.  */
  59 extern int acct (const char *__filename) __THROW;
  60 
  61 __END_DECLS
  62 
  63 #endif  /* sys/acct.h */
