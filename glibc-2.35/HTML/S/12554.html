  47 // API
  48 //==============================================================
  49 // float cbrtf(float)
  50 //
  51 // Overview of operation
  52 //==============================================================
  53 // Background
  54 //
  55 // Implementation
  56 //
  57 // Let y= frcpa(a), where a is the argument
  58 //
  59 // cbrt(a)= cbrt(a*y)/cbrt(y) = cbrt(1 - (1-a*y)) * (1/cbrt(y))
  60 //
  61 // For all values of y, the 3 possible significands of 1/cbrt(y)
  62 // are stored in a table (T0) to 64 bits of accuracy. (There are
  63 // 3 possible significands because the exponent of y modulo 3
  64 // can be 0, 1, or 2.)
  65 //
  66 //
  67 // * cbrt(1 - (1-a*y)) is approximated by a degree-2 polynomial
  68 //
  69 // 1 - (1/3)*r - (1/9)*r^2
  70 //
  71 // in r = 1-a*y.
  72 //
  73 // The table values are stored for three exponent values and are
  74 // then multiplied by 2^(e/3) where e is the exponent of the input number.
  75 // This computation is carried out in parallel with the polynomial
  76 // evaluation:
  77 //
  78 // T= 2^(e/3) * T0
  79 
  80 
  81 
  82 
  83 
  84 //===============
  85 // input= x
  86 // C= frcpa(x)
  87 // r= 1 - C * x
  88 //

 553 {.mfi
 554        getf.sig GR_SIGNIF = f8
 555        // will continue only for normal/denormal numbers
 556        fclass.nm.unc p12, p7 = f8, 0x1b
 557        // GR_GP = pointer to C_1, C_2 followed by T_table
 558        nop.i 0
 559 }
 560 {.mfi
 561        addl GR_GP = @ltoff(poly_coeffs), gp
 562        // normalize a
 563        fma.s1 FR_ARG = f8, f1, f0
 564        // GR_CT3 = bias-((2^8-1)/3) -63 = 0xffff-0x55-0x3f = 0xff6b
 565        mov GR_CT3 = 0xff6b ;;
 566 }
 567 
 568 {.mmi
 569        // get exponent
 570        getf.exp GR_ARGEXP = f8
 571        // load start address for C_1, C_2 followed by T_table
 572        ld8 GR_ADDR = [ GR_GP ]
 573        nop.i 0 ;;
 574 }
 575 
 576 {.mlx
 577        // check if input significand is 0
 578  (p7)  cmp.eq p12, p7 = GR_SIGNIF, r0
 579        // GR_2P63 = 2^63
 580        movl GR_2P63 = 0x8000000000000000 ;;
 581 }
 582 
 583 {.mfi
 584        nop.m 0
 585        // y = frcpa(a)
 586        // p7 = 1 for normal and denormal (but non-zero) arguments
 587  (p7)  frcpa.s0 FR_Y, p0 = f1, f8
 588        // p9 = 1 if denormal input
 589        cmp.gtu p9, p0 = GR_2P63, GR_SIGNIF
 590 }
 591 {.mfb
 592        // load C_1
 593        ldfe FR_COEFF1 = [ GR_ADDR ], 16
 594        // if argument is 0, +/-Infinity, or NaN, return
 595  (p12) fma.s.s0 f8 = f8, f1, f0
 596  (p12) br.ret.spnt b0 ;;
 597 }
 598 
 599 {.mmi
 600        // get normalized significand (for denormal inputs only)
 601  (p9)  getf.sig GR_SIGNIF = FR_ARG
 602        // load C_2
 603        ldfe FR_COEFF2 = [ GR_ADDR ], 16
 604        // GR_CT2 = bias-(2^8-1)
 605        mov GR_CT2 = 0xff00
 606 }
 607 
 608 {.mii
 609        // get exponent (for denormal inputs only)
 610  (p9)  getf.exp GR_ARGEXP = FR_ARG
 611        nop.i 0
 612        mov GR_CONST = 0x20000 ;;
 613 }
 614 
 615 
 616 {.mii
 617        // get GR_SIGN = sign
 618        and GR_SIGN = GR_ARGEXP, GR_CONST
 619        // eliminate leading 1 from GR_I1 = 1st table index
 620        shl GR_I1 = GR_SIGNIF, 1
 621        // eliminate sign from exponent
 622        andcm GR_EBIAS = GR_ARGEXP, GR_CONST ;;
 623 }
 624 
 625 
 626 {.mfi
 627        // subtract bias from GR_EXP = exponent
 628        sub GR_EXP = GR_EBIAS, GR_CT2
 629        // r = 1-a*y
 630        fnma.s1 FR_R = FR_Y, FR_ARG, f1
 631        // GR_IT1 = 1st table index (y_index8 bits)
 632        shr.u GR_IT1 = GR_I1, 56 ;;
 633 }
 634 
 635 
 636 {.mii
 637        // 1: exponent* = 5; // (2^{16}-1)/3 = 0x5555
 638        shladd GR_E5 = GR_EXP, 2, GR_EXP
 639        // GR_IT1_3 = 3*y_index
 640        shladd GR_IT1_3 = GR_IT1, 1, GR_IT1
 641        nop.i 0 ;;
 642 }
 643 
 644 
 645 {.mmi
 646        // GR_TMP5 = (5*expon)*16+5*expon = (0x55)*expon
 647        shladd GR_TMP5 = GR_E5, 4, GR_E5
 648        // adjust T_table pointer by 1st index
 649        shladd GR_TP1 = GR_IT1_3, 3, GR_ADDR
 650        nop.i 0 ;;
 651 }
 652 
 653 
 654 {.mmi
 655        // FR_T0 = T [ 0 ] [ y ]
 656        ldf8 FR_T0 = [ GR_TP1 ], 8
 657        // get 2^{-63}
 658        mov GR_TMP63 = 0xffff + 63
 659        // GR_TMP = (0x5500)*expon
 660        shl GR_TMP = GR_TMP5, 8 ;;
 661 }
 662 
 663 
 664 {.mfi
 665        // FR_T1 = T [ 1 ] [ y ]
 666        ldf8 FR_T1 = [ GR_TP1 ], 8
 667        // P_1 = C_1+C_2*r
 668        fma.s1 FR_COEFF1 = FR_COEFF2, FR_R, FR_COEFF1
 669        // GR_TMP2 = (0x5555)*expon
 670        add GR_TMP2 = GR_TMP, GR_TMP5 ;;
 671 }
 672 
 673 
 674 {.mmi
 675        // GR_TMP3 = (0x5556)*expon // 0x5556 = (2^{16}+2)/3
 676        add GR_TMP3 = GR_TMP2, GR_EXP ;;
 677        // FR_T2 = T [ 2 ] [ y ]
 678        ldf8 FR_T2 = [ GR_TP1 ]
 679        // GR_EXP3 = floor(expon/3)
 680        shr GR_EXP3 = GR_TMP3, 16 ;;
 681 }
 682 
 683 
 684 {.mmi
 685        setf.exp FR_2M63 = GR_TMP63
 686        // GR_TMP4 = 3*exponent
 687        shladd GR_TMP4 = GR_EXP3, 1, GR_EXP3
 688        // bias exponent
 689        add GR_EBIAS3 = GR_CT3, GR_EXP3 ;;
 690 }
 691 
 692 
 693 {.mmf
 694        // get remainder of exponent/3
 695        sub GR_REM = GR_EXP, GR_TMP4
 696        // add sign to exponent
 697        or GR_SEXP = GR_EBIAS3, GR_SIGN
 698        // P_2 = -r*P_1
 699        fnma.s1 FR_R = FR_COEFF1, FR_R, f0 ;;
 700 }

 704 {.mmi
 705        // FR_ARG = sign*2^{exponent/3}
 706        setf.exp FR_ARG = GR_SEXP
 707        nop.m 0
 708        // remainder = 0 ?
 709        // p7=1 if input exponent is 3*j (remainder is 0)
 710        cmp.eq.unc p7, p8 = r0, GR_REM ;;
 711 }

 715        // remainder = 1 ?
 716        // p8=1 if input exponent is 3*j+1 (remainder is 1)
 717        // p12=1 if input exponent is 3*j+2 (remainder is 2)
 718   (p8) cmp.eq.unc p8, p12 = 1, GR_REM
 719        // p7=1 -> remainder = 0 -> use T = FR_T0
 720   (p7)  fma.s1 f8 = FR_T0, FR_R, FR_T0
 721        // argument is of the form 2^(3*k) ?
 722        // ( GR_I1 holds significand bits, without the leading 1)
 723        or GR_I1 = GR_I1, GR_REM ;;
 724 }
 725

 727 .pred.rel "mutex", p12, p8
 728 {.mfi
 729        nop.m 0
 730        // p8=1 -> remainder = 1 -> use FR_T1
 731   (p8) fma.s1 f8 = FR_T1, FR_R, FR_T1
 732        // argument is of the form 2^(3*k) ?
 733        cmp.eq p14, p7 = GR_I1, r0
 734 }
 735

 738        nop.m 0
 739        // p12=1 -> remainder=2 -> result = T+T*P_2
 740  (p12) fma.s1 f8 = FR_T2, FR_R, FR_T2
 741        nop.i 0 ;;
 742 }

 750        nop.i 0
 751 }

 753        nop.m 0
 754        // T* = sgn*2^{expon/3}
 755   (p7) fma.s.s0 f8 = f8, FR_ARG, f0
 756        br.ret.sptk b0 ;;
