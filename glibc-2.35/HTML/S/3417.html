  42 float
  43 __ynf (int n, float x)
  44 {
  45   if (__builtin_expect (islessequal (x, 0.0f)
  46             || isgreater (x, AS_FLOAT_CONSTANT (X_TLOSS)), 0)
  47       && _LIB_VERSION != _IEEE_)
  48     {
  49       if (x < 0.0f)
  50     {
  51       /* d = zero/(x-x) */
  52       feraiseexcept (FE_INVALID);
  53       return __kernel_standard_f (n, x, 113);
  54     }
  55       else if (x == 0.0)
  56     {
  57       /* d = -one/(x-x) */
  58       feraiseexcept (FE_DIVBYZERO);
  59       return __kernel_standard_f (n, x, 112);
  60     }
  61       else if (_LIB_VERSION != _POSIX_)
  62     /* yn(n,x>X_TLOSS) */
  63     return __kernel_standard_f (n, x, 139);
  64     }
  65
