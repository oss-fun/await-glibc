  19 /*
  20  * ALGORITHM DESCRIPTION:
  21  *
  22  *      HIGH LEVEL OVERVIEW
  23  *
  24  *      Calculate z = (x*x+y*y)
  25  *      Calculate reciplicle sqrt (z)
  26  *      Calculate error = z*(rsqrt(z)*rsqrt(z)) - 1
  27  *      Calculate fixing part p with polynom
  28  *      Fix answer with sqrt(z) = z * rsqrt(z) + error * p * z
  29  *
  30  *      ALGORITHM DETAILS
  31  *
  32  *    Multiprecision branch for _HA_ only
  33  *      Remove sigm from both arguments
  34  *      Find maximum (_x) and minimum (_y) (by abs value) between arguments
  35  *      Split _x int _a and _b for multiprecision
  36  *      If _x >> _y we will we will not split _y for multiprecision
  37  *      all _y will be put into lower part (_d) and higher part (_c = 0)
  38  *      Fixing _hilo_mask for the case _x >> _y
  39  *      Split _y into _c and _d for multiprecision with fixed mask
  40  *
  41  *      compute Hi and Lo parts of _z = _x*_x + _y*_y
  42  *
  43  *      _zHi = _a*_a + _c*_c
  44  *      _zLo = (_x + _a)*_b + _d*_y + _d*_c
  45  *      _z = _zHi + _zLo
  46  *
  47  *    No multiprecision branch for _LA_ and _EP_
  48  *      _z = _VARG1 * _VARG1 + _VARG2 * _VARG2
  49  *
  50  *    Check _z exponent to be withing borders [3BC ; 441] else goto Callout
  51  *
  52  *    _s  ~ 1.0/sqrt(_z)
  53  *    _s2 ~ 1.0/(sqrt(_z)*sqrt(_z)) ~ 1.0/_z = (1.0/_z + O)
  54  *    _e[rror]  =  (1.0/_z + O) * _z - 1.0
  55  *    calculate fixing part _p
  56  *    _p = (((_POLY_C5*_e + _POLY_C4)*_e +_POLY_C3)*_e +_POLY_C2)*_e + _POLY_C1
  57  *    some parts of polynom are skipped for lower flav
  58  *
  59  *    result = _z * (1.0/sqrt(_z) + O) + _p * _e[rror] * _z
  60  *
  61  *
  62  */
  63 
  64 /* Offsets for data table __svml_dhypot_data_internal
  65  */
  66 #define _dHiLoMask                      0
  67 #define _dAbsMask                       16
  68 #define _dOne                           32
  69 #define _POLY_C5                        48
  70 #define _POLY_C4                        64
  71 #define _POLY_C3                        80
  72 #define _POLY_C2                        96
  73 #define _POLY_C1                        112
  74 #define _LowBoundary                    128
  75 #define _HighBoundary                   144

  77 #include <sysdep.h>

  79         .text
  80     .section .text.sse4,"ax",@progbits
  81 ENTRY(_ZGVbN2vv_hypot_sse4)
  82         subq      $88, %rsp
  83         cfi_def_cfa_offset(96)
  84 
  85 /*
  86  *  Defines
  87  *  Implementation
  88  * Multiprecision branch for _HA_ only
  89  * _z = _VARG1 * _VARG1 + _VARG2 * _VARG2
  90  */
  91         movaps    %xmm0, %xmm10
  92         movaps    %xmm1, %xmm2
  93         mulpd     %xmm0, %xmm10
  94         mulpd     %xmm1, %xmm2
  95         addpd     %xmm2, %xmm10
  96 
  97 /*
  98  * _s  ~ 1.0/sqrt(_z)
  99  * _s2 ~ 1.0/(sqrt(_z)*sqrt(_z)) ~ 1.0/_z
 100  */
 101         cvtpd2ps  %xmm10, %xmm7
 102         movlhps   %xmm7, %xmm7
 103         rsqrtps   %xmm7, %xmm8
 104         cvtps2pd  %xmm8, %xmm11
 105         movaps    %xmm11, %xmm2
 106         mulpd     %xmm11, %xmm2
 107 
 108 /* _e[rror]  ~  (1.0/_z + O) * _z - 1.0 */
 109         mulpd     %xmm10, %xmm2
 110         subpd     _dOne+__svml_dhypot_data_internal(%rip), %xmm2
 111 
 112 /*
 113  * calculate fixing part _p
 114  * _p = (((_POLY_C5*_e + _POLY_C4)*_e +_POLY_C3)*_e +_POLY_C2)*_e + _POLY_C1
 115  * some parts of polynom are skipped for lower flav
 116  */
 117         movups    _POLY_C4+__svml_dhypot_data_internal(%rip), %xmm9
 118         mulpd     %xmm2, %xmm9
 119         addpd     _POLY_C3+__svml_dhypot_data_internal(%rip), %xmm9
 120         mulpd     %xmm2, %xmm9
 121         addpd     _POLY_C2+__svml_dhypot_data_internal(%rip), %xmm9
 122         mulpd     %xmm2, %xmm9
 123         addpd     _POLY_C1+__svml_dhypot_data_internal(%rip), %xmm9
 124 
 125 /* result = _z * (1.0/sqrt(_z) + O) + _p * _e[rror] * _z */
 126         mulpd     %xmm9, %xmm2
 127         mulpd     %xmm11, %xmm2
 128         mulpd     %xmm10, %xmm11
 129         mulpd     %xmm10, %xmm2
 130 
 131 /* Check _z exponent to be withing borders [3BC ; 441] else goto Callout */
 132         movq      _LowBoundary+__svml_dhypot_data_internal(%rip), %xmm5
 133         movq      _HighBoundary+__svml_dhypot_data_internal(%rip), %xmm3
 134         pshufd    $221, %xmm10, %xmm4
 135         pcmpgtd   %xmm4, %xmm5
 136         pcmpgtd   %xmm3, %xmm4
 137         por       %xmm4, %xmm5
 138         pshufd    $80, %xmm5, %xmm6
 139         movmskpd  %xmm6, %edx
 140         addpd     %xmm11, %xmm2
 141 
 142 /*  The end of implementation  */
 143         testl     %edx, %edx
 144 
 145 /* Go to special inputs processing branch */
 146         jne       L(SPECIAL_VALUES_BRANCH)
 147                                 # LOE rbx rbp r12 r13 r14 r15 edx xmm0 xmm1 xmm2
 148 
 149 /* Restore registers
 150  * and exit the function
 151  */
 152 
 153 L(EXIT):
 154         movaps    %xmm2, %xmm0
 155         addq      $88, %rsp
 156         cfi_def_cfa_offset(8)
 157         ret
 158         cfi_def_cfa_offset(96)
 159 
 160 /* Branch to process
 161  * special inputs
 162  */
 163 
 164 L(SPECIAL_VALUES_BRANCH):
 165         movups    %xmm0, 32(%rsp)
 166         movups    %xmm1, 48(%rsp)
 167         movups    %xmm2, 64(%rsp)
 168                                 # LOE rbx rbp r12 r13 r14 r15 edx
 169 
 170         xorl      %eax, %eax
 171         movq      %r12, 16(%rsp)
 172         cfi_offset(12, -80)
 173         movl      %eax, %r12d
 174         movq      %r13, 8(%rsp)
 175         cfi_offset(13, -88)
 176         movl      %edx, %r13d
 177         movq      %r14, (%rsp)
 178         cfi_offset(14, -96)
 179                                 # LOE rbx rbp r15 r12d r13d
 180 
 181 /* Range mask
 182  * bits check
 183  */
 184 
 185 L(RANGEMASK_CHECK):
 186         btl       %r12d, %r13d
 187 
 188 /* Call scalar math function */
 189         jc        L(SCALAR_MATH_CALL)
 190                                 # LOE rbx rbp r15 r12d r13d
 191 
 192 /* Special inputs
 193  * processing loop
 194  */
 195 
 196 L(SPECIAL_VALUES_LOOP):
 197         incl      %r12d
 198         cmpl      $2, %r12d
 199 
 200 /* Check bits in range mask */
 201         jl        L(RANGEMASK_CHECK)
 202                                 # LOE rbx rbp r15 r12d r13d
 203 
 204         movq      16(%rsp), %r12
 205         cfi_restore(12)
 206         movq      8(%rsp), %r13
 207         cfi_restore(13)
 208         movq      (%rsp), %r14
 209         cfi_restore(14)
 210         movups    64(%rsp), %xmm2
 211 
 212 /* Go to exit */
 213         jmp       L(EXIT)
 214         cfi_offset(12, -80)
 215         cfi_offset(13, -88)
 216         cfi_offset(14, -96)
 217                                 # LOE rbx rbp r12 r13 r14 r15 xmm2
 218 
 219 /* Scalar math fucntion call
 220  * to process special input
 221  */
 222 
 223 L(SCALAR_MATH_CALL):
 224         movl      %r12d, %r14d
 225         movsd     32(%rsp,%r14,8), %xmm0
 226         movsd     48(%rsp,%r14,8), %xmm1
 227         call      hypot@PLT
 228                                 # LOE rbx rbp r14 r15 r12d r13d xmm0
 229 
 230         movsd     %xmm0, 64(%rsp,%r14,8)
 231 
 232 /* Process special inputs in loop */
 233         jmp       L(SPECIAL_VALUES_LOOP)
 234                                 # LOE rbx rbp r15 r12d r13d
 235 END(_ZGVbN2vv_hypot_sse4)
