  23 /* Type used to represent a TLS descriptor in the GOT.  */

  25 {
  26   union
  27     {
  28       void *pointer;
  29       long value;
  30     } argument;
  31   ptrdiff_t (*entry)(struct tlsdesc *);
  32 };
