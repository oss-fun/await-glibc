 104 static inline int
 105 issignalingf_inline (float x)
 106 {
 107   uint32_t ix = asuint (x);
 108   if (HIGH_ORDER_BIT_IS_SET_FOR_SNAN)
 109     return (ix & 0x7fc00000) == 0x7fc00000;
 110   return 2 * (ix ^ 0x00400000) > 2 * 0x7fc00000UL;
 111 }
