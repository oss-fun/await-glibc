  19 /*
  20  * ALGORITHM DESCRIPTION:
  21  *
  22  *   exp2(x)  = 2^n * T[j] * (1 + P(y))
  23  *   where
  24  *        x = m*(1/K) + y,    y in [-1/K..1/K]
  25  *        m = n*K + j,           m,n,j - signed integer, j in [-K/2..K/2]
  26  *
  27  *        values of 2^j/K are tabulated
  28  *
  29  *        P(y) is a minimax polynomial approximation of exp2(x)-1
  30  *        on small interval [-1/K..1/K]
  31  *
  32  *  Special cases:
  33  *
  34  *   exp2(NaN)  = NaN
  35  *   exp2(+INF) = +INF
  36  *   exp2(-INF) = 0
  37  *   exp2(x)    = 1 for subnormals
  38  *   For IEEE float
  39  *     if x >= 128.0 then exp2f(x) overflow
  40  *     if x < -151.0 then exp2f(x) underflow
  41  *
  42  */
  43 
  44 /* Offsets for data table __svml_sexp2_data_internal
  45  */
  46 #define _sShifter                       0
  47 #define _sPC0                           16
  48 #define _sPC1                           32
  49 #define _sPC2                           48
  50 #define _sPC3                           64
  51 #define _sPC4                           80
  52 #define _sPC5                           96
  53 #define _sPC6                           112
  54 #define _iAbsMask                       128
  55 #define _iDomainRange                   144
  56 
  57 #include <sysdep.h>
  58 
  59         .text
  60     .section .text.sse4,"ax",@progbits
  61 ENTRY(_ZGVbN4v_exp2f_sse4)
  62         subq      $72, %rsp
  63         cfi_def_cfa_offset(80)
  64 
  65 /* Check for overflow\underflow  */
  66         movups    __svml_sexp2_data_internal(%rip), %xmm1
  67 
  68 /*  Implementation  */
  69         movaps    %xmm1, %xmm5
  70 
  71 /*  Polynomial  */
  72         movups    _sPC6+__svml_sexp2_data_internal(%rip), %xmm4
  73         addps     %xmm0, %xmm5
  74         movaps    %xmm5, %xmm3
  75 
  76 /*  2^N  */
  77         pslld     $23, %xmm5
  78 
  79 /* Check for overflow\underflow  */
  80         movdqu    _iAbsMask+__svml_sexp2_data_internal(%rip), %xmm2
  81         subps     %xmm1, %xmm3
  82 
  83 /*  R  */
  84         movaps    %xmm0, %xmm1
  85         pand      %xmm0, %xmm2
  86         pcmpgtd   _iDomainRange+__svml_sexp2_data_internal(%rip), %xmm2
  87         subps     %xmm3, %xmm1
  88         movmskps  %xmm2, %edx
  89         mulps     %xmm1, %xmm4
  90         addps     _sPC5+__svml_sexp2_data_internal(%rip), %xmm4
  91         mulps     %xmm1, %xmm4
  92         addps     _sPC4+__svml_sexp2_data_internal(%rip), %xmm4
  93         mulps     %xmm1, %xmm4
  94         addps     _sPC3+__svml_sexp2_data_internal(%rip), %xmm4
  95         mulps     %xmm1, %xmm4
  96         addps     _sPC2+__svml_sexp2_data_internal(%rip), %xmm4
  97         mulps     %xmm1, %xmm4
  98         addps     _sPC1+__svml_sexp2_data_internal(%rip), %xmm4
  99         mulps     %xmm4, %xmm1
 100         addps     _sPC0+__svml_sexp2_data_internal(%rip), %xmm1
 101 
 102 /*  Reconstruction  */
 103         paddd     %xmm5, %xmm1
 104         testl     %edx, %edx
 105 
 106 /* Go to special inputs processing branch */
 107         jne       L(SPECIAL_VALUES_BRANCH)
 108                                 # LOE rbx rbp r12 r13 r14 r15 edx xmm0 xmm1
 109 
 110 /* Restore registers
 111  * and exit the function
 112  */
 113 
 114 L(EXIT):
 115         movaps    %xmm1, %xmm0
 116         addq      $72, %rsp
 117         cfi_def_cfa_offset(8)
 118         ret
 119         cfi_def_cfa_offset(80)
 120 
 121 /* Branch to process
 122  * special inputs
 123  */
 124 
 125 L(SPECIAL_VALUES_BRANCH):
 126         movups    %xmm0, 32(%rsp)
 127         movups    %xmm1, 48(%rsp)
 128                                 # LOE rbx rbp r12 r13 r14 r15 edx
 129 
 130         xorl      %eax, %eax
 131         movq      %r12, 16(%rsp)
 132         cfi_offset(12, -64)
 133         movl      %eax, %r12d
 134         movq      %r13, 8(%rsp)
 135         cfi_offset(13, -72)
 136         movl      %edx, %r13d
 137         movq      %r14, (%rsp)
 138         cfi_offset(14, -80)
 139                                 # LOE rbx rbp r15 r12d r13d
 140 
 141 /* Range mask
 142  * bits check
 143  */
 144 
 145 L(RANGEMASK_CHECK):
 146         btl       %r12d, %r13d
 147 
 148 /* Call scalar math function */
 149         jc        L(SCALAR_MATH_CALL)
 150                                 # LOE rbx rbp r15 r12d r13d
 151 
 152 /* Special inputs
 153  * processing loop
 154  */
 155 
 156 L(SPECIAL_VALUES_LOOP):
 157         incl      %r12d
 158         cmpl      $4, %r12d
 159 
 160 /* Check bits in range mask */
 161         jl        L(RANGEMASK_CHECK)
 162                                 # LOE rbx rbp r15 r12d r13d
 163 
 164         movq      16(%rsp), %r12
 165         cfi_restore(12)
 166         movq      8(%rsp), %r13
 167         cfi_restore(13)
 168         movq      (%rsp), %r14
 169         cfi_restore(14)
 170         movups    48(%rsp), %xmm1
 171 
 172 /* Go to exit */
 173         jmp       L(EXIT)
 174         cfi_offset(12, -64)
 175         cfi_offset(13, -72)
 176         cfi_offset(14, -80)
 177                                 # LOE rbx rbp r12 r13 r14 r15 xmm1
 178 
 179 /* Scalar math fucntion call
 180  * to process special input
 181  */
 182 
 183 L(SCALAR_MATH_CALL):
 184         movl      %r12d, %r14d
 185         movss     32(%rsp,%r14,4), %xmm0
 186         call      exp2f@PLT
 187                                 # LOE rbx rbp r14 r15 r12d r13d xmm0
 188 
 189         movss     %xmm0, 48(%rsp,%r14,4)
 190 
 191 /* Process special inputs in loop */
 192         jmp       L(SPECIAL_VALUES_LOOP)
 193                                 # LOE rbx rbp r15 r12d r13d
 194 END(_ZGVbN4v_exp2f_sse4)
