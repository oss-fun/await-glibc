  26 struct icmphdr
  27 {
  28   uint8_t type;     /* message type */
  29   uint8_t code;     /* type sub-code */
  30   uint16_t checksum;
  31   union
  32   {
  33     struct
  34     {
  35       uint16_t  id;
  36       uint16_t  sequence;
  37     } echo;         /* echo datagram */
  38     uint32_t    gateway;    /* gateway address */
  39     struct
  40     {
  41       uint16_t  __glibc_reserved;
  42       uint16_t  mtu;
  43     } frag;         /* path mtu discovery */
  44   } un;
  45 };
  46

 155 struct icmp
 156 {
 157   uint8_t  icmp_type;   /* type of message, see below */
 158   uint8_t  icmp_code;   /* type sub code */
 159   uint16_t icmp_cksum;  /* ones complement checksum of struct */
 160   union
 161   {
 162     unsigned char ih_pptr;  /* ICMP_PARAMPROB */
 163     struct in_addr ih_gwaddr;   /* gateway address */
 164     struct ih_idseq     /* echo datagram */
 165     {
 166       uint16_t icd_id;
 167       uint16_t icd_seq;
 168     } ih_idseq;
 169     uint32_t ih_void;
 170 
 171     /* ICMP_UNREACH_NEEDFRAG -- Path MTU Discovery (RFC1191) */
 172     struct ih_pmtu
 173     {
 174       uint16_t ipm_void;
 175       uint16_t ipm_nextmtu;
 176     } ih_pmtu;
 177 
 178     struct ih_rtradv
 179     {
 180       uint8_t irt_num_addrs;
 181       uint8_t irt_wpa;
 182       uint16_t irt_lifetime;
 183     } ih_rtradv;
 184   } icmp_hun;
 185 #define icmp_pptr   icmp_hun.ih_pptr
 186 #define icmp_gwaddr icmp_hun.ih_gwaddr
 187 #define icmp_id     icmp_hun.ih_idseq.icd_id
 188 #define icmp_seq    icmp_hun.ih_idseq.icd_seq
 189 #define icmp_void   icmp_hun.ih_void
 190 #define icmp_pmvoid icmp_hun.ih_pmtu.ipm_void
 191 #define icmp_nextmtu    icmp_hun.ih_pmtu.ipm_nextmtu
 192 #define icmp_num_addrs  icmp_hun.ih_rtradv.irt_num_addrs
 193 #define icmp_wpa    icmp_hun.ih_rtradv.irt_wpa
 194 #define icmp_lifetime   icmp_hun.ih_rtradv.irt_lifetime
 195   union
 196   {
 197     struct
 198     {
 199       uint32_t its_otime;
 200       uint32_t its_rtime;
 201       uint32_t its_ttime;
 202     } id_ts;
 203     struct
 204     {
 205       struct ip idi_ip;
 206       /* options and then 64 bits of data */
 207     } id_ip;
 208     struct icmp_ra_addr id_radv;
 209     uint32_t   id_mask;
 210     uint8_t    id_data[1];
 211   } icmp_dun;
 212 #define icmp_otime  icmp_dun.id_ts.its_otime
 213 #define icmp_rtime  icmp_dun.id_ts.its_rtime
 214 #define icmp_ttime  icmp_dun.id_ts.its_ttime
 215 #define icmp_ip     icmp_dun.id_ip.idi_ip
 216 #define icmp_radv   icmp_dun.id_radv
 217 #define icmp_mask   icmp_dun.id_mask
 218 #define icmp_data   icmp_dun.id_data
 219 };

 221 /*
 222  * Lower bounds on packet lengths for various types.
 223  * For the error advice packets must first insure that the
 224  * packet is large enough to contain the returned ip header.
 225  * Only then can we do the check to see if 64 bits of packet
 226  * data have been returned, since we need to check the returned
 227  * ip header length.
 228  */
 229 #define ICMP_MINLEN 8               /* abs minimum */
 230 #define ICMP_TSLEN  (8 + 3 * sizeof (n_time))   /* timestamp */
 231 #define ICMP_MASKLEN    12              /* address mask */
 232 #define ICMP_ADVLENMIN  (8 + sizeof (struct ip) + 8)    /* min */
 233 #ifndef _IP_VHL
 234 #define ICMP_ADVLEN(p)  (8 + ((p)->icmp_ip.ip_hl << 2) + 8)
 235     /* N.B.: must separately check that ip_hl >= 5 */
 236 #else
 237 #define ICMP_ADVLEN(p)  (8 + (IP_VHL_HL((p)->icmp_ip.ip_vhl) << 2) + 8)
 238     /* N.B.: must separately check that header length >= 5 */
 239 #endif
