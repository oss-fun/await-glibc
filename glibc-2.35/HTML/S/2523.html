  42 static inline void __attribute__ ((always_inline))

  67         if (r2 == NULL)
  68           r2 = r;
  69         end2 = r;
  70       }
  71     else
  72 # endif
  73       elf_machine_lazy_rel (map, scope, l_addr, r, skip_ifunc);
  74 
  75 # ifdef ELF_MACHINE_IRELATIVE
  76       if (r2 != NULL)
  77     for (; r2 <= end2; ++r2)
  78       if (ELFW(R_TYPE) (r2->r_info) == ELF_MACHINE_IRELATIVE)
