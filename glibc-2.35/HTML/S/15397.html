  19 /*
  20  * ALGORITHM DESCRIPTION:
  21  *   After computing exp(x) in high-low parts, an accurate computation is performed to obtain exp(x)-1
  22  *   Typical exp() implementation, except that:
  23  *    - tables are small (32 elements), allowing for fast gathers
  24  *    - all arguments processed in the main path
  25  *        - final VSCALEF assists branch-free design (correct overflow/underflow and special case responses)
  26  *        - a VAND is used to ensure the reduced argument |R|<2, even for large inputs
  27  *        - RZ mode used to avoid oveflow to +/-Inf for x*log2(e); helps with special case handling
  28  *
  29  *
  30  */
  31

  33  */

  38 #define Threshold                       384

  40 #define L2H                             512
  41 #define L2L                             576

  49         .text
  50     .section .text.exex512,"ax",@progbits
  51 ENTRY(_ZGVeN16v_expm1f_skx)
  52         pushq     %rbp
  53         cfi_def_cfa_offset(16)
  54         movq      %rsp, %rbp
  55         cfi_def_cfa(6, 16)
  56         cfi_offset(6, -16)
  57         andq      $-64, %rsp
  58         subq      $192, %rsp
  59         vmovups   L2E+__svml_sexpm1_data_internal_avx512(%rip), %zmm5
  60         vmovups   Shifter+__svml_sexpm1_data_internal_avx512(%rip), %zmm3
  61         vmovups   L2H+__svml_sexpm1_data_internal_avx512(%rip), %zmm8
  62         vmovups   L2L+__svml_sexpm1_data_internal_avx512(%rip), %zmm4
  63         vmovups   __svml_sexpm1_data_internal_avx512(%rip), %zmm6
  64 
  65 /* polynomial */
  66         vmovups   poly_coeff3+__svml_sexpm1_data_internal_avx512(%rip), %zmm9
  67         vmovups   poly_coeff2+__svml_sexpm1_data_internal_avx512(%rip), %zmm12
  68         vmovups   Exp_tbl_L+__svml_sexpm1_data_internal_avx512(%rip), %zmm11
  69         vmovups   Threshold+__svml_sexpm1_data_internal_avx512(%rip), %zmm2
  70 
  71 /* Th - 1 */
  72         vmovups   One+__svml_sexpm1_data_internal_avx512(%rip), %zmm14
  73         vmovaps   %zmm0, %zmm1
  74 
  75 /* 2^(52-5)*1.5 + x * log2(e) */
  76         vfmadd213ps {rn-sae}, %zmm3, %zmm1, %zmm5
  77         vcmpps    $29, {sae}, %zmm2, %zmm1, %k0
  78 
  79 /* Z0 ~ x*log2(e), rounded to 5 fractional bits */
  80         vsubps    {rn-sae}, %zmm3, %zmm5, %zmm7
  81         vpermt2ps Exp_tbl_H+64+__svml_sexpm1_data_internal_avx512(%rip), %zmm5, %zmm6
  82         vpermt2ps Exp_tbl_L+64+__svml_sexpm1_data_internal_avx512(%rip), %zmm5, %zmm11
  83         vandps    SgnMask+__svml_sexpm1_data_internal_avx512(%rip), %zmm1, %zmm0
  84 
  85 /* R = x - Z0*log(2) */
  86         vfnmadd213ps {rn-sae}, %zmm1, %zmm7, %zmm8
  87 
  88 /* scale Th */
  89         vscalefps {rn-sae}, %zmm7, %zmm6, %zmm2
  90         vfnmadd231ps {rn-sae}, %zmm7, %zmm4, %zmm8
  91         kmovw     %k0, %edx
  92 
  93 /* ensure |R|<2 even for special cases */
  94         vandps    EMask+__svml_sexpm1_data_internal_avx512(%rip), %zmm8, %zmm13
  95         vsubps    {rn-sae}, %zmm14, %zmm2, %zmm8
  96         vmulps    {rn-sae}, %zmm13, %zmm13, %zmm10
  97         vfmadd231ps {rn-sae}, %zmm13, %zmm9, %zmm12
  98 
  99 /* Tlr + R+ R2*Poly */
 100         vfmadd213ps {rn-sae}, %zmm11, %zmm10, %zmm12
 101         vaddps    {rn-sae}, %zmm13, %zmm12, %zmm15
 102 
 103 /* (Th-1)+Th*(Tlr + R+ R*Poly) */
 104         vfmadd213ps {rn-sae}, %zmm8, %zmm15, %zmm2
 105         vorps     %zmm0, %zmm2, %zmm0
 106         testl     %edx, %edx
 107 
 108 /* Go to special inputs processing branch */
 109         jne       L(SPECIAL_VALUES_BRANCH)
 110                                 # LOE rbx r12 r13 r14 r15 edx zmm0 zmm1
 111 
 112 /* Restore registers
 113  * and exit the function
 114  */
 115 
 116 L(EXIT):
 117         movq      %rbp, %rsp
 118         popq      %rbp
 119         cfi_def_cfa(7, 8)
 120         cfi_restore(6)
 121         ret
 122         cfi_def_cfa(6, 16)
 123         cfi_offset(6, -16)
 124 
 125 /* Branch to process
 126  * special inputs
 127  */
 128 
 129 L(SPECIAL_VALUES_BRANCH):
 130         vmovups   %zmm1, 64(%rsp)
 131         vmovups   %zmm0, 128(%rsp)
 132                                 # LOE rbx r12 r13 r14 r15 edx zmm0
 133

 135                                 # LOE rbx r12 r13 r14 r15 eax edx

 139         /*  DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -176; DW_OP_plus)  */

 141         movl      %eax, %r12d
 142         movq      %r13, 8(%rsp)
 143         /*  DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -184; DW_OP_plus)  */
 144         .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x48, 0xff, 0xff, 0xff, 0x22
 145         movl      %edx, %r13d
 146         movq      %r14, (%rsp)
 147         /*  DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -192; DW_OP_plus)  */
 148         .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x40, 0xff, 0xff, 0xff, 0x22
 149                                 # LOE rbx r15 r12d r13d

 152  * bits check

 154 
 155 L(RANGEMASK_CHECK):

 157 
 158 /* Call scalar math function */
 159         jc        L(SCALAR_MATH_CALL)
 160                                 # LOE rbx r15 r12d r13d

 188         /*  DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -192; DW_OP_plus)  */
