  25 /* Open FILE with access OFLAG.  Interpret relative paths relative to
  26    the directory associated with FD.  If O_CREAT or O_TMPFILE is in OFLAG, a
  27    third argument is the file protection.  */
  28 int
  29 __openat64 (int fd, const char *file, int oflag, ...)
  30 {
  31   int mode;
  32 
  33   if (file == NULL)
  34     {
  35       __set_errno (EINVAL);
  36       return -1;
  37     }
  38 
  39   if (fd != AT_FDCWD && file[0] != '/')
  40     {
  41       /* Check FD is associated with a directory.  */
  42       struct stat64 st;
  43       if (__fstat64 (fd, &st) != 0)
  44     return -1;
  45 
  46       if (!S_ISDIR (st.st_mode))
  47     {
  48       __set_errno (ENOTDIR);
  49       return -1;
  50     }
  51     }
  52 
  53   if (__OPEN_NEEDS_MODE (oflag))
  54     {
  55       va_list arg;
  56       va_start (arg, oflag);
  57       mode = va_arg (arg, int);
  58       va_end (arg);
  59 
  60       ignore_value (mode);
  61     }
  62 
  63   __set_errno (ENOSYS);
  64   return -1;
  65 }
