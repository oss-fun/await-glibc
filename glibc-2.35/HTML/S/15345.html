  19 /*
  20  * ALGORITHM DESCRIPTION:
  21  *
  22  *      HIGH LEVEL OVERVIEW
  23  *
  24  *      Calculate z = (x*x+y*y)
  25  *      Calculate reciplicle sqrt (z)
  26  *      Calculate make two NR iterations
  27  *
  28  *      ALGORITHM DETAILS
  29  *
  30  *    Multiprecision branch for _HA_ only
  31  *      Remove sigm from both arguments
  32  *      Find maximum (_x) and minimum (_y) (by abs value) between arguments
  33  *      Split _x int _a and _b for multiprecision
  34  *      If _x >> _y we will we will not split _y for multiprecision
  35  *      all _y will be put into lower part (_d) and higher part (_c = 0)
  36  *      Fixing _hilo_mask for the case _x >> _y
  37  *      Split _y into _c and _d for multiprecision with fixed mask
  38  *
  39  *      compute Hi and Lo parts of _z = _x*_x + _y*_y
  40  *
  41  *      _zHi = _a*_a + _c*_c
  42  *      _zLo = (_x + _a)*_b + _d*_y + _d*_c
  43  *      _z = _zHi + _zLo
  44  *
  45  *    No multiprecision branch for _LA_ and _EP_
  46  *      _z = _VARG1 * _VARG1 + _VARG2 * _VARG2
  47  *
  48  *    Check _z exponent to be withing borders [1E3 ; 60A] else goto Callout
  49  *
  50  *    Compute resciplicle sqrt s0 ~ 1.0/sqrt(_z),
  51  *      that multiplied by _z, is final result for _EP_ version.
  52  *
  53  *    First iteration (or zero iteration):
  54  *       s =  z * s0
  55  *       h = .5 * s0
  56  *       d =  s *  h - .5
  57  *
  58  *    Second iteration:
  59  *       h = d * h + h
  60  *       s = s * d + s
  61  *       d = s * s - z (in multiprecision for _HA_)
  62  *
  63  *    result = s - h * d
  64  *
  65  *    EP version of the function can be implemented as y[i]=sqrt(a[i]^2+b[i]^2)
  66  *    with all intermediate operations done in target precision for i=1,..,n.
  67  *    It can return result y[i]=0 in case a[i]^2 and b[i]^2 underflow in target
  68  *    precision (for some i). It can return result y[i]=NAN in case
  69  *    a[i]^2+b[i]^2 overflow in target precision, for some i. It can return
  70  *    result y[i]=NAN in case a[i] or b[i] is infinite, for some i.
  71  *
  72  *
  73  */
  74

  85         .text
  86     .section .text.avx2,"ax",@progbits
  87 ENTRY(_ZGVdN8vv_hypotf_avx2)
  88         pushq     %rbp
  89         cfi_def_cfa_offset(16)
  90         movq      %rsp, %rbp
  91         cfi_def_cfa(6, 16)
  92         cfi_offset(6, -16)
  93         andq      $-32, %rsp
  94         subq      $128, %rsp
  95 
  96 /*
  97  *  Implementation
  98  * Multiprecision branch for _HA_ only
  99  * No multiprecision branch for _LA_
 100  * _z = _VARG1 * _VARG1 + _VARG2 * _VARG2
 101  */
 102         vmulps    %ymm0, %ymm0, %ymm8
 103 
 104 /*
 105  *  Variables
 106  *  Defines
 107  *  Constants loading
 108  */
 109         vmovups   _sHalf+__svml_shypot_data_internal(%rip), %ymm7
 110 
 111 /* Check _z exponent to be withing borders [1E3 ; 60A] else goto Callout */
 112         vmovups   _LowBoundary+__svml_shypot_data_internal(%rip), %ymm2
 113         vfmadd231ps %ymm1, %ymm1, %ymm8
 114 
 115 /* _s0  ~ 1.0/sqrt(_z) */
 116         vrsqrtps  %ymm8, %ymm6
 117         vpcmpgtd  %ymm8, %ymm2, %ymm3
 118 
 119 /* First iteration */
 120         vmulps    %ymm8, %ymm6, %ymm9
 121         vmulps    %ymm7, %ymm6, %ymm2
 122         vfnmadd231ps %ymm9, %ymm2, %ymm7
 123         vfmadd213ps %ymm9, %ymm7, %ymm9
 124 
 125 /* Second iteration */
 126         vfmadd132ps %ymm7, %ymm2, %ymm2
 127         vpcmpgtd  _HighBoundary+__svml_shypot_data_internal(%rip), %ymm8, %ymm4
 128         vpor      %ymm4, %ymm3, %ymm5
 129 
 130 /* Finish second iteration in native precision for _LA_ */
 131         vfmsub231ps %ymm9, %ymm9, %ymm8
 132         vmovmskps %ymm5, %edx
 133         vfnmadd213ps %ymm9, %ymm8, %ymm2
 134 
 135 /*  The end of implementation  */
 136         testl     %edx, %edx
 137 
 138 /* Go to special inputs processing branch */
 139         jne       L(SPECIAL_VALUES_BRANCH)
 140                                 # LOE rbx r12 r13 r14 r15 edx ymm0 ymm1 ymm2
 141 
 142 /* Restore registers
 143  * and exit the function
 144  */
 145 
 146 L(EXIT):
 147         vmovaps   %ymm2, %ymm0
 148         movq      %rbp, %rsp
 149         popq      %rbp
 150         cfi_def_cfa(7, 8)
 151         cfi_restore(6)
 152         ret
 153         cfi_def_cfa(6, 16)
 154         cfi_offset(6, -16)
 155 
 156 /* Branch to process
 157  * special inputs
 158  */
 159 
 160 L(SPECIAL_VALUES_BRANCH):
 161         vmovups   %ymm0, 32(%rsp)
 162         vmovups   %ymm1, 64(%rsp)
 163         vmovups   %ymm2, 96(%rsp)
 164                                 # LOE rbx r12 r13 r14 r15 edx ymm2
 165 
 166         xorl      %eax, %eax
 167                                 # LOE rbx r12 r13 r14 r15 eax edx
 168 
 169         vzeroupper
 170         movq      %r12, 16(%rsp)
 171         /*  DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -112; DW_OP_plus)  */
 172         .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x90, 0xff, 0xff, 0xff, 0x22
 173         movl      %eax, %r12d
 174         movq      %r13, 8(%rsp)
 175         /*  DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -120; DW_OP_plus)  */
 176         .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x88, 0xff, 0xff, 0xff, 0x22
 177         movl      %edx, %r13d
 178         movq      %r14, (%rsp)
 179         /*  DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -128; DW_OP_plus)  */
 180         .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x80, 0xff, 0xff, 0xff, 0x22
 181                                 # LOE rbx r15 r12d r13d
 182 
 183 /* Range mask
 184  * bits check
 185  */
 186 
 187 L(RANGEMASK_CHECK):
 188         btl       %r12d, %r13d
 189 
 190 /* Call scalar math function */
 191         jc        L(SCALAR_MATH_CALL)
 192                                 # LOE rbx r15 r12d r13d
 193 
 194 /* Special inputs
 195  * processing loop
 196  */
 197 
 198 L(SPECIAL_VALUES_LOOP):
 199         incl      %r12d
 200         cmpl      $8, %r12d
 201 
 202 /* Check bits in range mask */
 203         jl        L(RANGEMASK_CHECK)
 204                                 # LOE rbx r15 r12d r13d
 205 
 206         movq      16(%rsp), %r12
 207         cfi_restore(12)
 208         movq      8(%rsp), %r13
 209         cfi_restore(13)
 210         movq      (%rsp), %r14
 211         cfi_restore(14)
 212         vmovups   96(%rsp), %ymm2
 213 
 214 /* Go to exit */
 215         jmp       L(EXIT)
 216         /*  DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -112; DW_OP_plus)  */
 217         .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x90, 0xff, 0xff, 0xff, 0x22
 218         /*  DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -120; DW_OP_plus)  */
 219         .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x88, 0xff, 0xff, 0xff, 0x22
 220         /*  DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -128; DW_OP_plus)  */
 221         .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x80, 0xff, 0xff, 0xff, 0x22
 222                                 # LOE rbx r12 r13 r14 r15 ymm2
 223 
 224 /* Scalar math fucntion call
 225  * to process special input
 226  */
 227 
 228 L(SCALAR_MATH_CALL):
 229         movl      %r12d, %r14d
 230         movss     32(%rsp,%r14,4), %xmm0
 231         movss     64(%rsp,%r14,4), %xmm1
 232         call      hypotf@PLT
 233                                 # LOE rbx r14 r15 r12d r13d xmm0
 234 
 235         movss     %xmm0, 96(%rsp,%r14,4)
 236 
 237 /* Process special inputs in loop */
 238         jmp       L(SPECIAL_VALUES_LOOP)
 239                                 # LOE rbx r15 r12d r13d
