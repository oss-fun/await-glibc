  24 void
  25 nis_ping (const_nis_name dirname, unsigned int utime,
  26       const nis_object *dirobj)
  27 {
  28   nis_result *res = NULL;
  29   nis_object *obj;
  30   ping_args args;
  31   unsigned int i;
  32 
  33   if (dirname == NULL && dirobj == NULL)
  34     abort ();
  35 
  36   if (dirobj == NULL)
  37     {
  38       res = nis_lookup (dirname, MASTER_ONLY);
  39       if (res == NULL || NIS_RES_STATUS (res) != NIS_SUCCESS)
  40     {
  41       nis_freeresult (res);
  42       return;
  43     }
  44       obj = res->objects.objects_val;
  45     }
  46   else
  47     obj = (nis_object *) dirobj;
  48 
  49   /* Check if obj is really a diryectory object */
  50   if (__type_of (obj) != NIS_DIRECTORY_OBJ)
  51     {
  52       nis_freeresult (res);
  53       return;
  54     }
  55 
  56   if (dirname == NULL)
  57     args.dir = obj->DI_data.do_name;
  58   else
  59     args.dir = (char *) dirname;
  60   args.stamp = utime;
  61 
  62   /* Send the ping only to replicas */
  63   for (i = 1; i < obj->DI_data.do_servers.do_servers_len; ++i)
  64     __do_niscall2 (&obj->DI_data.do_servers.do_servers_val[i], 1,
  65            NIS_PING, (xdrproc_t) _xdr_ping_args,
  66            (caddr_t) &args, (xdrproc_t) xdr_void,
  67            (caddr_t) NULL, 0, NULL);
  68   nis_freeresult (res);
  69 }
