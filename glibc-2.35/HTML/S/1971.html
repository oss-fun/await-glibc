  52 size_t
  53 SIMPLE_STRCSPN (const CHAR *s, const CHAR *rej)
  54 {
  55   const CHAR *r, *str = s;
  56   CHAR c;
  57 
  58   while ((c = *s++) != '\0')
  59     for (r = rej; *r != '\0'; ++r)
  60       if (*r == c)
  61     return s - str - 1;
  62   return s - str - 1;
  63 }

  65 size_t
  66 STUPID_STRCSPN (const CHAR *s, const CHAR *rej)
  67 {
  68   size_t ns = STRLEN (s), nrej = STRLEN (rej);
  69   size_t i, j;
  70 
  71   for (i = 0; i < ns; ++i)
  72     for (j = 0; j < nrej; ++j)
  73       if (s[i] == rej[j])
  74     return i;
  75   return i;
  76 }
