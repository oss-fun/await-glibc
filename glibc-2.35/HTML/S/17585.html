  33 long int
  34 __lrint (double x)
  35 {
  36   long int y;
  37   /* The z196 zarch "convert to fixed" (cgdbra) instruction is rounding
  38      according to current rounding mode (M3-field: 0).
  39      First convert x with suppressed inexact exception and check if the
  40      resulting value is beyond the target limits (indicated by cc=3;
  41      Note: a nan is also indicated by cc=3).
  42      If the resulting value is within the target limits, redo
  43      without suppressing the inexact exception.  */
  44   __asm__ (INSN " %0,0,%1,4 \n\t"
  45        "jo 1f \n\t"
  46        INSN " %0,0,%1,0 \n\t"
  47        "1:"
  48        : "=&d" (y) : "f" (x) : "cc");
  49   return y;
  50 }
