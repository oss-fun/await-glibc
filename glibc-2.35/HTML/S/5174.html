  59 #include <stdio.h>
  60 #include <mach_error.h>
  61 #include <mach/boolean.h>
  62 
  63 extern char * mach_error_string_int(mach_error_t, boolean_t *);
  64 
  65 void
  66 mach_error(char *str, mach_error_t err)
  67 {
  68     char * err_str;
  69     char buf[1024];
  70     boolean_t diag;
  71 
  72     err_str=mach_error_string_int(err, &diag);
  73 
  74     if ( diag ) {
  75         sprintf( buf, "%s %s (%x)", mach_error_type(err), err_str, err );
  76         err_str = buf;
  77     }
  78 
  79     fprintf(stderr, "%s %s\n", str, err_str);
  80 }
