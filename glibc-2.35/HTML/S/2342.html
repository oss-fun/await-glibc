  78           return NULL;
  79         }

  85         {
  86           ++i;

 153 void
 154 nis_freeservlist (nis_server **serv)
 155 {
 156   int i;
 157 
 158   if (serv == NULL)
 159     return;
 160 
 161   i = 0;
 162   while (serv[i] != NULL)
 163     {
