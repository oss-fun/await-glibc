  25 /* Add an action to FILE-ACTIONS which tells the implementation to call
  26    `open' for the given file during the `spawn' call.  */
  27 int
  28 __posix_spawn_file_actions_addopen (posix_spawn_file_actions_t *file_actions,
  29                     int fd, const char *path, int oflag,
  30                     mode_t mode)
  31 {
  32   struct __spawn_action *rec;
  33 
  34   if (!__spawn_valid_fd (fd))
  35     return EBADF;
  36 
  37   char *path_copy = __strdup (path);
  38   if (path_copy == NULL)
  39     return ENOMEM;
  40 
  41   /* Allocate more memory if needed.  */
  42   if (file_actions->__used == file_actions->__allocated
  43       && __posix_spawn_file_actions_realloc (file_actions) != 0)
  44     {
  45       /* This can only mean we ran out of memory.  */
  46       free (path_copy);
  47       return ENOMEM;
  48     }
  49 
  50   /* Add the new value.  */
  51   rec = &file_actions->__actions[file_actions->__used];
  52   rec->tag = spawn_do_open;
  53   rec->action.open_action.fd = fd;
  54   rec->action.open_action.path = path_copy;
  55   rec->action.open_action.oflag = oflag;
  56   rec->action.open_action.mode = mode;
  57 
  58   /* Account for the new entry.  */
  59   ++file_actions->__used;
  60 
  61   return 0;
  62 }
