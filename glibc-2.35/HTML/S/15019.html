  22  *    Get short reciprocal approximation Rcp ~ 1/mantissa(x)

  53         vgetexpps {sae}, %zmm0, %zmm5

  57         vsubps    {rn-sae}, %zmm1, %zmm3, %zmm11

  59         vpermps   coeff3+__svml_slog10_data_internal_avx512(%rip), %zmm7, %zmm2

  61         vpermps   coeff2+__svml_slog10_data_internal_avx512(%rip), %zmm7, %zmm4
  62         vpermps   coeff1+__svml_slog10_data_internal_avx512(%rip), %zmm7, %zmm8
  63 
  64 /* x<=0? */
  65         vfpclassps $94, %zmm0, %k0
  66         vfmadd213ps {rn-sae}, %zmm2, %zmm11, %zmm1
  67         vmulps    {rn-sae}, %zmm10, %zmm9, %zmm12
  68         vfmadd213ps {rn-sae}, %zmm4, %zmm11, %zmm1
  69         kmovw     %k0, %edx
  70         vfmadd213ps {rn-sae}, %zmm8, %zmm11, %zmm1
  71         vfmadd213ps {rn-sae}, %zmm12, %zmm11, %zmm1
  72         testl     %edx, %edx
  73 
  74 /* Go to special inputs processing branch */
  75         jne       L(SPECIAL_VALUES_BRANCH)
  76                                 # LOE rbx r12 r13 r14 r15 edx zmm0 zmm1
  77 
  78 /* Restore registers
  79  * and exit the function
  80  */
  81 
  82 L(EXIT):
  83         vmovaps   %zmm1, %zmm0
  84         movq      %rbp, %rsp
