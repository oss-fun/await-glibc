  19 /*
  20  * ALGORITHM DESCRIPTION:
  21  *
  22  *      HIGH LEVEL OVERVIEW
  23  *
  24  *      Calculate z = (x*x+y*y)
  25  *      Calculate reciplicle sqrt (z)
  26  *      Calculate error = z*(rsqrt(z)*rsqrt(z)) - 1
  27  *      Calculate fixing part p with polynom
  28  *      Fix answer with sqrt(z) = z * rsqrt(z) + error * p * z
  29  *
  30  *      ALGORITHM DETAILS
  31  *
  32  *    Multiprecision branch for _HA_ only
  33  *      Remove sigm from both arguments
  34  *      Find maximum (_x) and minimum (_y) (by abs value) between arguments
  35  *      Split _x int _a and _b for multiprecision
  36  *      If _x >> _y we will we will not split _y for multiprecision
  37  *      all _y will be put into lower part (_d) and higher part (_c = 0)
  38  *      Fixing _hilo_mask for the case _x >> _y
  39  *      Split _y into _c and _d for multiprecision with fixed mask
  40  *
  41  *      compute Hi and Lo parts of _z = _x*_x + _y*_y
  42  *
  43  *      _zHi = _a*_a + _c*_c
  44  *      _zLo = (_x + _a)*_b + _d*_y + _d*_c
  45  *      _z = _zHi + _zLo
  46  *
  47  *    No multiprecision branch for _LA_ and _EP_
  48  *      _z = _VARG1 * _VARG1 + _VARG2 * _VARG2
  49  *
  50  *    Check _z exponent to be withing borders [3BC ; 441] else goto Callout
  51  *
  52  *    _s  ~ 1.0/sqrt(_z)
  53  *    _s2 ~ 1.0/(sqrt(_z)*sqrt(_z)) ~ 1.0/_z = (1.0/_z + O)
  54  *    _e[rror]  =  (1.0/_z + O) * _z - 1.0
  55  *    calculate fixing part _p
  56  *    _p = (((_POLY_C5*_e + _POLY_C4)*_e +_POLY_C3)*_e +_POLY_C2)*_e + _POLY_C1
  57  *    some parts of polynom are skipped for lower flav
  58  *
  59  *    result = _z * (1.0/sqrt(_z) + O) + _p * _e[rror] * _z
  60  *
  61  *
  62  */
  63

  75 ENTRY(_ZGVeN8vv_hypot_skx)
  76         pushq     %rbp
  77         cfi_def_cfa_offset(16)
  78         movq      %rsp, %rbp
  79         cfi_def_cfa(6, 16)
  80         cfi_offset(6, -16)
  81         andq      $-64, %rsp
  82         subq      $256, %rsp
  83         vgetexppd {sae}, %zmm0, %zmm2
  84         vgetexppd {sae}, %zmm1, %zmm3
  85         vmovups   _dHalf+__svml_dhypot_data_internal(%rip), %zmm9
  86         vmaxpd    {sae}, %zmm3, %zmm2, %zmm4
  87         vmulpd    {rn-sae}, %zmm0, %zmm0, %zmm2
  88         vandpd    _dAbsMask+__svml_dhypot_data_internal(%rip), %zmm4, %zmm5
  89         vfmadd231pd {rn-sae}, %zmm1, %zmm1, %zmm2
  90 
  91 /* Select exponent bound so that no scaling is needed */
  92         vpcmpq    $5, _lExpBound_uisa+__svml_dhypot_data_internal(%rip), %zmm5, %k0
  93         vrsqrt14pd %zmm2, %zmm6
  94         kmovw     %k0, %edx
  95         vmulpd    {rn-sae}, %zmm6, %zmm2, %zmm7
  96         vmulpd    {rn-sae}, %zmm6, %zmm9, %zmm8
  97         vfnmadd231pd {rn-sae}, %zmm7, %zmm8, %zmm9
  98         vfmadd231pd {rn-sae}, %zmm9, %zmm8, %zmm8
  99         vfmadd213pd {rn-sae}, %zmm7, %zmm7, %zmm9
 100         vfnmadd231pd {rn-sae}, %zmm9, %zmm9, %zmm2
 101         vfmadd213pd {rn-sae}, %zmm9, %zmm8, %zmm2
 102 
 103 /*  The end of implementation  */
 104         testl     %edx, %edx
 105 
 106 /* Go to special inputs processing branch */
 107         jne       L(SPECIAL_VALUES_BRANCH)
 108                                 # LOE rbx r12 r13 r14 r15 edx zmm0 zmm1 zmm2
 109 
 110 /* Restore registers
 111  * and exit the function
 112  */
 113 
 114 L(EXIT):
 115         vmovaps   %zmm2, %zmm0
 116         movq      %rbp, %rsp
 117         popq      %rbp
 118         cfi_def_cfa(7, 8)
 119         cfi_restore(6)
 120         ret
 121         cfi_def_cfa(6, 16)
 122         cfi_offset(6, -16)

 131         vmovups   %zmm2, 192(%rsp)
 132                                 # LOE rbx r12 r13 r14 r15 edx zmm2

 148         .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x00, 0xff, 0xff, 0xff, 0x22
 149                                 # LOE rbx r15 r12d r13d

 151 /* Range mask
 152  * bits check
 153  */
 154

 188         /*  DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -256; DW_OP_plus)  */
 189         .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x00, 0xff, 0xff, 0xff, 0x22
 190                                 # LOE rbx r12 r13 r14 r15 zmm2
