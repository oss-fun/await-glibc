  19 /*
  20  * ALGORITHM DESCRIPTION:
  21  *
  22  *      HIGH LEVEL OVERVIEW
  23  *
  24  *      Calculate z = (x*x+y*y)
  25  *      Calculate reciplicle sqrt (z)
  26  *      Calculate error = z*(rsqrt(z)*rsqrt(z)) - 1
  27  *      Calculate fixing part p with polynom
  28  *      Fix answer with sqrt(z) = z * rsqrt(z) + error * p * z
  29  *
  30  *      ALGORITHM DETAILS
  31  *
  32  *    Multiprecision branch for _HA_ only
  33  *      Remove sigm from both arguments
  34  *      Find maximum (_x) and minimum (_y) (by abs value) between arguments
  35  *      Split _x int _a and _b for multiprecision
  36  *      If _x >> _y we will we will not split _y for multiprecision
  37  *      all _y will be put into lower part (_d) and higher part (_c = 0)
  38  *      Fixing _hilo_mask for the case _x >> _y
  39  *      Split _y into _c and _d for multiprecision with fixed mask
  40  *
  41  *      compute Hi and Lo parts of _z = _x*_x + _y*_y
  42  *
  43  *      _zHi = _a*_a + _c*_c
  44  *      _zLo = (_x + _a)*_b + _d*_y + _d*_c
  45  *      _z = _zHi + _zLo
  46  *
  47  *    No multiprecision branch for _LA_ and _EP_
  48  *      _z = _VARG1 * _VARG1 + _VARG2 * _VARG2
  49  *
  50  *    Check _z exponent to be withing borders [3BC ; 441] else goto Callout
  51  *
  52  *    _s  ~ 1.0/sqrt(_z)
  53  *    _s2 ~ 1.0/(sqrt(_z)*sqrt(_z)) ~ 1.0/_z = (1.0/_z + O)
  54  *    _e[rror]  =  (1.0/_z + O) * _z - 1.0
  55  *    calculate fixing part _p
  56  *    _p = (((_POLY_C5*_e + _POLY_C4)*_e +_POLY_C3)*_e +_POLY_C2)*_e + _POLY_C1
  57  *    some parts of polynom are skipped for lower flav
  58  *
  59  *    result = _z * (1.0/sqrt(_z) + O) + _p * _e[rror] * _z
  60  *
  61  *
  62  */
  63 
  64 /* Offsets for data table __svml_dhypot_data_internal
  65  */

  67 #define _dAbsMask                       32
  68 #define _dOne                           64
  69 #define _POLY_C5                        96
  70 #define _POLY_C4                        128
  71 #define _POLY_C3                        160
  72 #define _POLY_C2                        192
  73 #define _POLY_C1                        224
  74 #define _LowBoundary                    256
  75 #define _HighBoundary                   288

  79         .text
  80     .section .text.avx2,"ax",@progbits
  81 ENTRY(_ZGVdN4vv_hypot_avx2)
  82         pushq     %rbp
  83         cfi_def_cfa_offset(16)
  84         movq      %rsp, %rbp
  85         cfi_def_cfa(6, 16)
  86         cfi_offset(6, -16)
  87         andq      $-32, %rsp
  88         subq      $128, %rsp
  89         vmovapd   %ymm1, %ymm2
  90         vmovapd   %ymm0, %ymm1
  91 
  92 /*
  93  *  Defines
  94  *  Implementation
  95  * Multiprecision branch for _HA_ only
  96  * _z = _VARG1 * _VARG1 + _VARG2 * _VARG2
  97  */
  98         vmulpd    %ymm1, %ymm1, %ymm0
  99 
 100 /*
 101  * calculate fixing part _p
 102  * _p = (((_POLY_C5*_e + _POLY_C4)*_e +_POLY_C3)*_e +_POLY_C2)*_e + _POLY_C1
 103  * some parts of polynom are skipped for lower flav
 104  */
 105         vmovupd   _POLY_C4+__svml_dhypot_data_internal(%rip), %ymm15
 106         vmovups   _LowBoundary+__svml_dhypot_data_internal(%rip), %xmm4
 107         vfmadd231pd %ymm2, %ymm2, %ymm0
 108 
 109 /*
 110  * _s  ~ 1.0/sqrt(_z)
 111  * _s2 ~ 1.0/(sqrt(_z)*sqrt(_z)) ~ 1.0/_z
 112  */
 113         vcvtpd2ps %ymm0, %xmm12
 114 
 115 /* Check _z exponent to be withing borders [3BC ; 441] else goto Callout */
 116         vextractf128 $1, %ymm0, %xmm3
 117         vrsqrtps  %xmm12, %xmm13
 118         vshufps   $221, %xmm3, %xmm0, %xmm5
 119         vcvtps2pd %xmm13, %ymm3
 120         vpcmpgtd  %xmm5, %xmm4, %xmm6
 121         vpcmpgtd  _HighBoundary+__svml_dhypot_data_internal(%rip), %xmm5, %xmm7
 122         vpor      %xmm7, %xmm6, %xmm9
 123         vpshufd   $80, %xmm9, %xmm8
 124         vmulpd    %ymm3, %ymm3, %ymm14
 125         vpshufd   $250, %xmm9, %xmm10
 126 
 127 /* _e[rror]  ~  (1.0/_z + O) * _z - 1.0 */
 128         vfmsub213pd _dOne+__svml_dhypot_data_internal(%rip), %ymm0, %ymm14
 129         vfmadd213pd _POLY_C3+__svml_dhypot_data_internal(%rip), %ymm14, %ymm15
 130         vfmadd213pd _POLY_C2+__svml_dhypot_data_internal(%rip), %ymm14, %ymm15
 131         vfmadd213pd _POLY_C1+__svml_dhypot_data_internal(%rip), %ymm14, %ymm15
 132 
 133 /* result = _z * (1.0/sqrt(_z) + O) + _p * _e[rror] * _z */
 134         vmulpd    %ymm15, %ymm14, %ymm14
 135         vmulpd    %ymm14, %ymm3, %ymm15
 136         vmulpd    %ymm15, %ymm0, %ymm4
 137         vfmadd213pd %ymm4, %ymm3, %ymm0
 138         vinsertf128 $1, %xmm10, %ymm8, %ymm11
 139         vmovmskpd %ymm11, %edx
 140 
 141 /*  The end of implementation  */
 142         testl     %edx, %edx
 143 
 144 /* Go to special inputs processing branch */
 145         jne       L(SPECIAL_VALUES_BRANCH)
 146                                 # LOE rbx r12 r13 r14 r15 edx ymm0 ymm1 ymm2
 147 
 148 /* Restore registers
 149  * and exit the function
 150  */
 151 
 152 L(EXIT):
 153         movq      %rbp, %rsp
 154         popq      %rbp
 155         cfi_def_cfa(7, 8)
 156         cfi_restore(6)
 157         ret
 158         cfi_def_cfa(6, 16)
 159         cfi_offset(6, -16)
 160 
 161 /* Branch to process
 162  * special inputs
 163  */
 164 
 165 L(SPECIAL_VALUES_BRANCH):
 166         vmovupd   %ymm1, 32(%rsp)
 167         vmovupd   %ymm2, 64(%rsp)
 168         vmovupd   %ymm0, 96(%rsp)
 169                                 # LOE rbx r12 r13 r14 r15 edx ymm0
 170 
 171         xorl      %eax, %eax
 172                                 # LOE rbx r12 r13 r14 r15 eax edx
 173 
 174         vzeroupper
 175         movq      %r12, 16(%rsp)
 176         /*  DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -112; DW_OP_plus)  */
 177         .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x90, 0xff, 0xff, 0xff, 0x22
 178         movl      %eax, %r12d
 179         movq      %r13, 8(%rsp)
 180         /*  DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -120; DW_OP_plus)  */
 181         .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x88, 0xff, 0xff, 0xff, 0x22
 182         movl      %edx, %r13d
 183         movq      %r14, (%rsp)
 184         /*  DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -128; DW_OP_plus)  */
 185         .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x80, 0xff, 0xff, 0xff, 0x22
 186                                 # LOE rbx r15 r12d r13d
 187 
 188 /* Range mask
 189  * bits check
 190  */
 191 
 192 L(RANGEMASK_CHECK):
 193         btl       %r12d, %r13d
 194 
 195 /* Call scalar math function */
 196         jc        L(SCALAR_MATH_CALL)
 197                                 # LOE rbx r15 r12d r13d
 198 
 199 /* Special inputs
 200  * processing loop
 201  */
 202 
 203 L(SPECIAL_VALUES_LOOP):
 204         incl      %r12d
 205         cmpl      $4, %r12d
 206 
 207 /* Check bits in range mask */
 208         jl        L(RANGEMASK_CHECK)
 209                                 # LOE rbx r15 r12d r13d
 210 
 211         movq      16(%rsp), %r12
 212         cfi_restore(12)
 213         movq      8(%rsp), %r13
 214         cfi_restore(13)
 215         movq      (%rsp), %r14
 216         cfi_restore(14)
 217         vmovupd   96(%rsp), %ymm0
 218 
 219 /* Go to exit */
 220         jmp       L(EXIT)
 221         /*  DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -112; DW_OP_plus)  */
 222         .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x90, 0xff, 0xff, 0xff, 0x22
 223         /*  DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -120; DW_OP_plus)  */
 224         .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x88, 0xff, 0xff, 0xff, 0x22
 225         /*  DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -128; DW_OP_plus)  */
 226         .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x80, 0xff, 0xff, 0xff, 0x22
 227                                 # LOE rbx r12 r13 r14 r15 ymm0
 228 
 229 /* Scalar math fucntion call
 230  * to process special input
 231  */
 232 
 233 L(SCALAR_MATH_CALL):
 234         movl      %r12d, %r14d
 235         movsd     32(%rsp,%r14,8), %xmm0
 236         movsd     64(%rsp,%r14,8), %xmm1
 237         call      hypot@PLT
 238                                 # LOE rbx r14 r15 r12d r13d xmm0
 239 
 240         movsd     %xmm0, 96(%rsp,%r14,8)
 241 
 242 /* Process special inputs in loop */
 243         jmp       L(SPECIAL_VALUES_LOOP)
 244                                 # LOE rbx r15 r12d r13d
 245 END(_ZGVdN4vv_hypot_avx2)
