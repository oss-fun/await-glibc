  49 // API
  50 //==============================================================
  51 // long double exp2l(long double)
  52 //
  53 // Overview of operation
  54 //==============================================================
  55 // Background
  56 //
  57 // Implementation
  58 //
  59 // Let x= K + f + r, where
  60 // K is an integer, f= 0.b1 b2... b8 (f>= 0),
  61 // and |r|<2^{-8}
  62 // T is a table that stores 2^f (256 entries) rounded to
  63 // double extended precision (only mantissa is stored)
  64 // D stores (2^f/T [ f ] - 1), rounded to single precision
  65 //
  66 // 2^x is approximated as
  67 // 2^K * T [ f ] * (1+D [ f ] +c1*r+c2*r^2+...+c6*r^6)
  68 //
  69 
  70 
  71 
  72 // Special values
  73 //==============================================================
  74 // exp2(0)= 1
  75 // exp2(+inf)= inf
  76 // exp2(-inf)= 0
  77 //
  78

 380 {.mii
 381        // get exponent
 382        getf.exp GR_EBIAS = f8
 383        // GR_D_ADDR0 = pointer to D_table
 384        addl GR_D_ADDR0 = @ltoff(D_table), gp
 385        // GR_ADDR0 = pointer to C_1...C_6 followed by T_table
 386        addl GR_ADDR0 = @ltoff(poly_coeffs), gp ;;
 387 }
 388 
 389 {.mfi
 390        // get significand

 392        // will continue only for normal/denormal numbers
 393        fclass.nm.unc p12, p7 = f8, 0x1b
 394        mov GR_63 = 63 ;;

 398        nop.m 0
 399        nop.f 0
 400        // GR_CONST2 = bias+63-8
 401        mov GR_CONST2 = 0xffff+55
 402 }
 403 {.mfi
 404        // GR_CONST1 = bias+15
 405        mov GR_CONST1 = 0xffff+15
 406        nop.f 0
 407        mov GR_CONST3 = 0x1ffff ;;
 408 }
 409 
 410 {.mfi
 411        // load start address for C_1...C_6 followed by T_table
 412        ld8 GR_ADDR = [ GR_ADDR0 ]
 413        nop.f 0
 414        // get sign of argument
 415        andcm GR_SGN = GR_EBIAS, GR_CONST3
 416 }
 417 {.mfi
 418        // GR_D_ADDR = pointer to D_table
 419        ld8 GR_D_ADDR = [ GR_D_ADDR0 ]
 420        nop.f 0
 421        // get argument exponent
 422        and GR_ARGEXP = GR_CONST3, GR_EBIAS ;;
 423 }
 424 
 425 {.mfi
 426        alloc GR_SREG = ar.pfs, 1, 4, 4, 0
 427        nop.f 0
 428        // p6 = 1 if sign = 1
 429        cmp.ne p6, p8 = GR_SGN, r0
 430 }

 432        // p7 = 1 if exponent> = 15 (argument out of range)

 434        nop.f 0
 435        sub GR_EXPON = GR_CONST2, GR_ARGEXP ;;

 439        // load C_3, C_4

 453        // load C_5, C_6
 454        ldfpd FR_COEFF5, FR_COEFF6 = [ GR_ADDR ], 16

 456        movl GR_2P14 = 0x46800000 ;;

 461        ldfe FR_COEFF1 = [ GR_ADDR ], 16
 462        fma.s0 f8 = f8, f1, f0
 463        // GR_BM63 = bias-63
 464        mov GR_BM63 = 0xffff-63 ;;
 465 }

 467 {.mlx
 468        setf.s FR_2P14 = GR_2P14
 469        // GR_UF_TEST = -2^14-62
 470        movl GR_UF_TEST = 0xc6807c00
 471 }
 472 {.mfi
 473        // load C_2
 474        ldfe FR_COEFF2 = [ GR_ADDR ], 16
 475        nop.f 0

 480        // get 8-bit index
 481        and GR_INDEX = GR_255, GR_LEADBITS

 484        nop.b 0 ;;

 488        // if sign = 1 && f>0, set p7 = 1

 504        setf.exp FR_EXP63 = GR_EM63

 506        nop.i 0 ;;

 511        // if sign = 0, set scale factor exponent S = K+bias-63
 512   (p8) add GR_K = GR_K, GR_BM63

 514        // if sign = 1, set scale factor exponent S = -K+bias-63
 515   (p6) sub GR_K = GR_BM63, GR_K ;;
