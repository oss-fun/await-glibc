 395 /* Compute sin (pi * X) for -0.25 <= X <= 0.5.  */
 396 
 397 static long double
 398 lg_sinpi (long double x)
 399 {
 400   if (x <= 0.25L)
 401     return __sinl (M_PIl * x);
 402   else
 403     return __cosl (M_PIl * (0.5L - x));
 404 }
 405 
 406 /* Compute cos (pi * X) for -0.25 <= X <= 0.5.  */
 407 
 408 static long double
 409 lg_cospi (long double x)
 410 {
 411   if (x <= 0.25L)
 412     return __cosl (M_PIl * x);
 413   else
 414     return __sinl (M_PIl * (0.5L - x));
 415 }
 416 
 417 /* Compute cot (pi * X) for -0.25 <= X <= 0.5.  */
 418 
 419 static long double
 420 lg_cotpi (long double x)
 421 {
 422   return lg_cospi (x) / lg_sinpi (x);
 423 }

 425 /* Compute lgamma of a negative argument -48 < X < -2, setting
 426    *SIGNGAMP accordingly.  */
 427 
 428 long double
 429 __lgamma_negl (long double x, int *signgamp)
 430 {
 431   /* Determine the half-integer region X lies in, handle exact
 432      integers and determine the sign of the result.  */
 433   int i = floorl (-2 * x);
 434   if ((i & 1) == 0 && i == -2 * x)
 435     return 1.0L / 0.0L;
 436   long double xn = ((i & 1) == 0 ? -i / 2 : (-i - 1) / 2);
 437   i -= 4;
 438   *signgamp = ((i & 2) == 0 ? -1 : 1);
 439 
 440   SET_RESTORE_ROUNDL (FE_TONEAREST);
 441 
 442   /* Expand around the zero X0 = X0_HI + X0_LO.  */
 443   long double x0_hi = lgamma_zeros[i][0], x0_lo = lgamma_zeros[i][1];
 444   long double xdiff = x - x0_hi - x0_lo;
 445 
 446   /* For arguments in the range -3 to -2, use polynomial
 447      approximations to an adjusted version of the gamma function.  */
 448   if (i < 2)
 449     {
 450       int j = floorl (-8 * x) - 16;
 451       long double xm = (-33 - 2 * j) * 0.0625L;
 452       long double x_adj = x - xm;
 453       size_t deg = poly_deg[j];
 454       size_t end = poly_end[j];
 455       long double g = poly_coeff[end];
 456       for (size_t j = 1; j <= deg; j++)
 457     g = g * x_adj + poly_coeff[end - j];
 458       return __log1pl (g * xdiff / (x - xn));
 459     }
 460 
 461   /* The result we want is log (sinpi (X0) / sinpi (X))
 462      + log (gamma (1 - X0) / gamma (1 - X)).  */
 463   long double x_idiff = fabsl (xn - x), x0_idiff = fabsl (xn - x0_hi - x0_lo);
 464   long double log_sinpi_ratio;
 465   if (x0_idiff < x_idiff * 0.5L)
 466     /* Use log not log1p to avoid inaccuracy from log1p of arguments
 467        close to -1.  */
 468     log_sinpi_ratio = __ieee754_logl (lg_sinpi (x0_idiff)
 469                       / lg_sinpi (x_idiff));
 470   else
 471     {
 472       /* Use log1p not log to avoid inaccuracy from log of arguments
 473      close to 1.  X0DIFF2 has positive sign if X0 is further from
 474      XN than X is from XN, negative sign otherwise.  */
 475       long double x0diff2 = ((i & 1) == 0 ? xdiff : -xdiff) * 0.5L;
 476       long double sx0d2 = lg_sinpi (x0diff2);
 477       long double cx0d2 = lg_cospi (x0diff2);
 478       log_sinpi_ratio = __log1pl (2 * sx0d2
 479                   * (-sx0d2 + cx0d2 * lg_cotpi (x_idiff)));
 480     }
 481 
 482   long double log_gamma_ratio;
 483   long double y0 = 1 - x0_hi;
 484   long double y0_eps = -x0_hi + (1 - y0) - x0_lo;
 485   long double y = 1 - x;
 486   long double y_eps = -x + (1 - y);
 487   /* We now wish to compute LOG_GAMMA_RATIO
 488      = log (gamma (Y0 + Y0_EPS) / gamma (Y + Y_EPS)).  XDIFF
 489      accurately approximates the difference Y0 + Y0_EPS - Y -
 490      Y_EPS.  Use Stirling's approximation.  First, we may need to
 491      adjust into the range where Stirling's approximation is
 492      sufficiently accurate.  */
 493   long double log_gamma_adj = 0;
 494   if (i < 18)
 495     {
 496       int n_up = (19 - i) / 2;
 497       long double ny0, ny0_eps, ny, ny_eps;
 498       ny0 = y0 + n_up;
 499       ny0_eps = y0 - (ny0 - n_up) + y0_eps;
 500       y0 = ny0;
 501       y0_eps = ny0_eps;
 502       ny = y + n_up;
 503       ny_eps = y - (ny - n_up) + y_eps;
 504       y = ny;
 505       y_eps = ny_eps;
 506       long double prodm1 = __lgamma_productl (xdiff, y - n_up, y_eps, n_up);
 507       log_gamma_adj = -__log1pl (prodm1);
 508     }
 509   long double log_gamma_high
 510     = (xdiff * __log1pl ((y0 - e_hi - e_lo + y0_eps) / e_hi)
 511        + (y - 0.5L + y_eps) * __log1pl (xdiff / y) + log_gamma_adj);
 512   /* Compute the sum of (B_2k / 2k(2k-1))(Y0^-(2k-1) - Y^-(2k-1)).  */
 513   long double y0r = 1 / y0, yr = 1 / y;
 514   long double y0r2 = y0r * y0r, yr2 = yr * yr;
 515   long double rdiff = -xdiff / (y * y0);
 516   long double bterm[NCOEFF];
 517   long double dlast = rdiff, elast = rdiff * yr * (yr + y0r);
 518   bterm[0] = dlast * lgamma_coeff[0];
 519   for (size_t j = 1; j < NCOEFF; j++)
 520     {
 521       long double dnext = dlast * y0r2 + elast;
 522       long double enext = elast * yr2;
 523       bterm[j] = dnext * lgamma_coeff[j];
 524       dlast = dnext;
 525       elast = enext;
 526     }
 527   long double log_gamma_low = 0;
 528   for (size_t j = 0; j < NCOEFF; j++)
 529     log_gamma_low += bterm[NCOEFF - 1 - j];
 530   log_gamma_ratio = log_gamma_high + log_gamma_low;
 531 
 532   return log_sinpi_ratio + log_gamma_ratio;
 533 }
