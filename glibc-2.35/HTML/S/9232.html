 185 static const _Float128 c[] = {
 186 /* 113 bits of pi/2 */
 187 #define PI_2_1 c[0]
 188  L(0x1.921fb54442d18469898cc51701b8p+0),
 189 
 190 /* pi/2 - PI_2_1 */
 191 #define PI_2_1t c[1]
 192  L(0x3.9a252049c1114cf98e804177d4c8p-116),

 195 int32_t __ieee754_rem_pio2l(_Float128 x, _Float128 *y)
 196 {
 197   _Float128 z, w, t;
 198   double tx[8];
 199   int64_t exp, n, ix, hx;
 200   uint64_t lx;
 201 
 202   GET_LDOUBLE_WORDS64 (hx, lx, x);
 203   ix = hx & 0x7fffffffffffffffLL;
 204   if (ix <= 0x3ffe921fb54442d1LL)   /* x in <-pi/4, pi/4> */
 205     {
 206       y[0] = x;
 207       y[1] = 0;
 208       return 0;
 209     }
 210 
 211   if (ix < 0x40002d97c7f3321dLL)    /* |x| in <pi/4, 3pi/4) */
 212     {
 213       if (hx > 0)
 214     {
 215       /* 113 + 113 bit PI is ok */
 216       z = x - PI_2_1;
 217       y[0] = z - PI_2_1t;
 218       y[1] = (z - y[0]) - PI_2_1t;
 219       return 1;
 220     }
 221       else
 222         {
 223       /* 113 + 113 bit PI is ok */
 224       z = x + PI_2_1;
 225       y[0] = z + PI_2_1t;
 226       y[1] = (z - y[0]) + PI_2_1t;
 227       return -1;
 228     }
 229     }
 230 
 231   if (ix >= 0x7fff000000000000LL)   /* x is +=oo or NaN */
 232     {
 233       y[0] = x - x;
 234       y[1] = y[0];
 235       return 0;
 236     }
 237 
 238   /* Handle large arguments.
 239      We split the 113 bits of the mantissa into 5 24bit integers
 240      stored in a double array.  */
 241   exp = (ix >> 48) - 16383 - 23;
 242 
 243   /* This is faster than doing this in floating point, because we
 244      have to convert it to integers anyway and like this we can keep
 245      both integer and floating point units busy.  */
 246   tx [0] = (double)(((ix >> 25) & 0x7fffff) | 0x800000);
 247   tx [1] = (double)((ix >> 1) & 0xffffff);
 248   tx [2] = (double)(((ix << 23) | (lx >> 41)) & 0xffffff);
 249   tx [3] = (double)((lx >> 17) & 0xffffff);
 250   tx [4] = (double)((lx << 7) & 0xffffff);
 251 
 252   n = __kernel_rem_pio2 (tx, tx + 5, exp, ((lx << 7) & 0xffffff) ? 5 : 4,
 253              3, two_over_pi);
 254 
 255   /* The result is now stored in 3 double values, we need to convert it into
 256      two long double values.  */
 257   t = (_Float128) tx [6] + (_Float128) tx [7];
 258   w = (_Float128) tx [5];
 259 
 260   if (hx >= 0)
 261     {
