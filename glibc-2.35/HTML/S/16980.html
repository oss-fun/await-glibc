  18 /* Finds length of a 0-terminated string.  */
  19 
  20 #include <sysdep.h>
  21 
  22     .arch ev6
  23     .set noreorder
  24     .set noat
  25 
  26 ENTRY(strlen)
  27 #ifdef PROF
  28     ldgp    gp, 0(pv)
  29     lda AT, _mcount
  30     jsr AT, (AT), _mcount
  31     .prologue 1
  32 #else
  33     .prologue 0
  34 #endif
  35 
  36     ldq_u   $1, 0($16)  # L : load first quadword ($16 may be misaligned)
  37     lda $2, -1($31) # E :
  38     insqh   $2, $16, $2 # U :
  39     andnot  $16, 7, $0  # E :
  40 
  41     or  $2, $1, $1  # E :
  42     cmpbge  $31, $1, $2 # E : $2  <- bitmask: bit i == 1 <==> i-th byte == 0
  43     nop         # E :
  44     bne $2, $found  # U :
  45 
  46 $loop:  ldq $1, 8($0)   # L :
  47     addq    $0, 8, $0   # E : addr += 8
  48     cmpbge  $31, $1, $2 # E :
  49     beq $2, $loop   # U :
  50 
  51 $found:
  52     cttz    $2, $3      # U0 :
  53     addq    $0, $3, $0  # E :
  54     subq    $0, $16, $0 # E :
  55     ret $31, ($26)  # L0 :
  56 
  57     END(strlen)
  58 libc_hidden_builtin_def (strlen)
