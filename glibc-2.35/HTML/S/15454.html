  19 /*
  20  * ALGORITHM DESCRIPTION:
  21  *
  22  *   Compute atanh(x) as 0.5 * log((1 + x)/(1 - x))
  23  *
  24  *   Special cases:
  25  *
  26  *   atanh(0)  = 0
  27  *   atanh(+1) = +INF
  28  *   atanh(-1) = -INF
  29  *   atanh(x)  = NaN if |x| > 1, or if x is a NaN or INF
  30  *
  31  */
  32 
  33 /* Offsets for data table __svml_satanh_data_internal
  34  */
  35 #define SgnMask                         0
  36 #define sOne                            32
  37 #define sPoly                           64
  38 #define iBrkValue                       320
  39 #define iOffExpoMask                    352
  40 #define sHalf                           384
  41 #define sSign                           416
  42 #define sTopMask12                      448
  43 #define TinyRange                       480
  44 #define sLn2                            512

  48         .text
  49     .section .text.avx2,"ax",@progbits
  50 ENTRY(_ZGVdN8v_atanhf_avx2)
  51         pushq     %rbp
  52         cfi_def_cfa_offset(16)
  53         movq      %rsp, %rbp
  54         cfi_def_cfa(6, 16)
  55         cfi_offset(6, -16)
  56         andq      $-32, %rsp
  57         subq      $96, %rsp
  58 
  59 /* Load constants including One = 1 */
  60         vmovups   sOne+__svml_satanh_data_internal(%rip), %ymm5
  61         vmovups   sTopMask12+__svml_satanh_data_internal(%rip), %ymm13
  62         vmovaps   %ymm0, %ymm6
  63 
  64 /* Strip off the sign, so treat X as positive until right at the end */
  65         vandps    SgnMask+__svml_satanh_data_internal(%rip), %ymm6, %ymm10
  66         vsubps    %ymm10, %ymm5, %ymm1
  67 
  68 /*
  69  * Compute V = 2 * X trivially, and UHi + U_lo = 1 - X in two pieces,
  70  * the upper part UHi being <= 12 bits long. Then we have
  71  * atanh(X) = 1/2 * log((1 + X) / (1 - X)) = 1/2 * log1p(V / (UHi + ULo)).
  72  */
  73         vaddps    %ymm10, %ymm10, %ymm14
  74 
  75 /*
  76  * Check whether |X| < 1, in which case we use the main function.
  77  * Otherwise set the rangemask so that the callout will get used.
  78  * Note that this will also use the callout for NaNs since not(NaN < 1).
  79  */
  80         vcmpnlt_uqps %ymm5, %ymm10, %ymm7
  81         vsubps    %ymm1, %ymm5, %ymm9
  82         vcmplt_oqps TinyRange+__svml_satanh_data_internal(%rip), %ymm10, %ymm4
  83         vrcpps    %ymm1, %ymm11
  84         vsubps    %ymm10, %ymm9, %ymm12
  85         vandps    %ymm13, %ymm11, %ymm0
  86 
  87 /* No need to split sU when FMA is available */
  88         vfnmadd213ps %ymm5, %ymm0, %ymm1
  89         vmovaps   %ymm6, %ymm8
  90         vfmadd213ps %ymm6, %ymm6, %ymm8
  91         vfnmadd231ps %ymm0, %ymm12, %ymm1
  92 
  93 /*
  94  * Split V as well into upper 12 bits and lower part, so that we can get
  95  * a preliminary quotient estimate without rounding error.
  96  */
  97         vandps    %ymm13, %ymm14, %ymm15
  98         vmovmskps %ymm7, %edx
  99         vsubps    %ymm15, %ymm14, %ymm7
 100 
 101 /* Hence get initial quotient estimate QHi + QLo = R * VHi + R * VLo */
 102         vmulps    %ymm15, %ymm0, %ymm10
 103 
 104 /* Compute D = E + E^2 */
 105         vfmadd213ps %ymm1, %ymm1, %ymm1
 106 
 107 /* Record the sign for eventual reincorporation. */
 108         vandps    sSign+__svml_satanh_data_internal(%rip), %ymm6, %ymm3
 109 
 110 /* Or the sign bit in with the tiny result to handle atanh(-0) correctly */
 111         vorps     %ymm3, %ymm8, %ymm2
 112         vmulps    %ymm7, %ymm0, %ymm8
 113 
 114 /*
 115  * Compute R * (VHi + VLo) * (1 + E + E^2)
 116  * = R *  (VHi + VLo) * (1 + D)
 117  * = QHi + (QHi * D + QLo + QLo * D)
 118  */
 119         vmulps    %ymm1, %ymm10, %ymm9
 120         vfmadd213ps %ymm8, %ymm8, %ymm1
 121         vaddps    %ymm1, %ymm9, %ymm1
 122 
 123 /* reduction: compute r,n */
 124         vmovups   iBrkValue+__svml_satanh_data_internal(%rip), %ymm9
 125 
 126 /*
 127  * Now finally accumulate the high and low parts of the
 128  * argument to log1p, H + L, with a final compensated summation.
 129  */
 130         vaddps    %ymm1, %ymm10, %ymm12
 131         vsubps    %ymm12, %ymm10, %ymm11
 132 
 133 /*
 134  * Now we feed into the log1p code, using H in place of _VARG1 and
 135  * later incorporating L into the reduced argument.
 136  * compute 1+x as high, low parts
 137  */
 138         vmaxps    %ymm12, %ymm5, %ymm13
 139         vminps    %ymm12, %ymm5, %ymm14
 140         vaddps    %ymm11, %ymm1, %ymm0
 141         vaddps    %ymm14, %ymm13, %ymm1
 142         vpsubd    %ymm9, %ymm1, %ymm7
 143         vsubps    %ymm1, %ymm13, %ymm15
 144         vpsrad    $23, %ymm7, %ymm10
 145         vpand     iOffExpoMask+__svml_satanh_data_internal(%rip), %ymm7, %ymm8
 146         vaddps    %ymm15, %ymm14, %ymm13
 147         vpslld    $23, %ymm10, %ymm11
 148         vpaddd    %ymm9, %ymm8, %ymm15
 149         vaddps    %ymm13, %ymm0, %ymm14
 150         vcvtdq2ps %ymm10, %ymm0
 151         vpsubd    %ymm11, %ymm5, %ymm12
 152 
 153 /* polynomial evaluation */
 154         vsubps    %ymm5, %ymm15, %ymm5
 155         vmulps    %ymm14, %ymm12, %ymm1
 156         vaddps    %ymm5, %ymm1, %ymm5
 157         vmovups   sPoly+224+__svml_satanh_data_internal(%rip), %ymm1
 158         vfmadd213ps sPoly+192+__svml_satanh_data_internal(%rip), %ymm5, %ymm1
 159         vfmadd213ps sPoly+160+__svml_satanh_data_internal(%rip), %ymm5, %ymm1
 160         vfmadd213ps sPoly+128+__svml_satanh_data_internal(%rip), %ymm5, %ymm1
 161         vfmadd213ps sPoly+96+__svml_satanh_data_internal(%rip), %ymm5, %ymm1
 162         vfmadd213ps sPoly+64+__svml_satanh_data_internal(%rip), %ymm5, %ymm1
 163         vfmadd213ps sPoly+32+__svml_satanh_data_internal(%rip), %ymm5, %ymm1
 164         vfmadd213ps sPoly+__svml_satanh_data_internal(%rip), %ymm5, %ymm1
 165         vmulps    %ymm1, %ymm5, %ymm7
 166         vfmadd213ps %ymm5, %ymm5, %ymm7
 167 
 168 /* final reconstruction */
 169         vfmadd132ps sLn2+__svml_satanh_data_internal(%rip), %ymm7, %ymm0
 170 
 171 /* Finally, halve the result and reincorporate the sign */
 172         vxorps    sHalf+__svml_satanh_data_internal(%rip), %ymm3, %ymm3
 173         vmulps    %ymm0, %ymm3, %ymm0
 174         vblendvps %ymm4, %ymm2, %ymm0, %ymm0
 175         testl     %edx, %edx
 176 
 177 /* Go to special inputs processing branch */
 178         jne       L(SPECIAL_VALUES_BRANCH)
 179                                 # LOE rbx r12 r13 r14 r15 edx ymm0 ymm6
 180 
 181 /* Restore registers
 182  * and exit the function
 183  */
 184 
 185 L(EXIT):
 186         movq      %rbp, %rsp
 187         popq      %rbp
 188         cfi_def_cfa(7, 8)
 189         cfi_restore(6)
 190         ret
 191         cfi_def_cfa(6, 16)
 192         cfi_offset(6, -16)
