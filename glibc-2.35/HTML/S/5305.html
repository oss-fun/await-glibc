  26 faccessat (int fd, const char *file, int type, int flag)
  27 {
  28   if (file == NULL || (flag & ~(AT_SYMLINK_NOFOLLOW | AT_EACCESS)) != 0
  29       || (type & ~(R_OK|W_OK|X_OK|F_OK)) != 0)
  30     {
  31       __set_errno (EINVAL);
  32       return -1;
  33     }
  34 
  35   if (fd < 0 && fd != AT_FDCWD)
  36     {
  37       __set_errno (EBADF);
  38       return -1;
  39     }
  40 
  41   __set_errno (ENOSYS);
  42   return -1;
