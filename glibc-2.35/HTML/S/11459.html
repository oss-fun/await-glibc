  21     .text
  22     .globl      __syscall_error
  23 ENTRY(__libc_vfork)
  24     LOADSYSCALL(vfork)
  25     ta  0x10
  26     bcc 2f
  27      mov    %o7, %g1
  28     call    __syscall_error
  29      mov    %g1, %o7
  30 2:  sub %o1, 1, %o1
  31     andcc   %o0, %o1, %o0
  32 1:  retl
  33      nop
  34 END(__libc_vfork)
