 260   {

 262     socklen_t msg_namelen;  /* Length of address data.  */

 291 /* Ancillary data object manipulation macros.  */
 292 #if __glibc_c99_flexarr_available
 293 # define CMSG_DATA(cmsg) ((cmsg)->__cmsg_data)
 294 #else
 295 # define CMSG_DATA(cmsg) ((unsigned char *) ((struct cmsghdr *) (cmsg) + 1))
 296 #endif
 297 #define CMSG_NXTHDR(mhdr, cmsg) __cmsg_nxthdr (mhdr, cmsg)
 298 #define CMSG_FIRSTHDR(mhdr) \
 299   ((size_t) (mhdr)->msg_controllen >= sizeof (struct cmsghdr)             \
 300    ? (struct cmsghdr *) (mhdr)->msg_control : (struct cmsghdr *) 0)
 301 #define CMSG_ALIGN(len) (((len) + sizeof (size_t) - 1) \
 302              & (size_t) ~(sizeof (size_t) - 1))
 303 #define CMSG_SPACE(len) (CMSG_ALIGN (len) \
 304              + CMSG_ALIGN (sizeof (struct cmsghdr)))
 305 #define CMSG_LEN(len)   (CMSG_ALIGN (sizeof (struct cmsghdr)) + (len))
 306 
 307 extern struct cmsghdr *__cmsg_nxthdr (struct msghdr *__mhdr,
 308                       struct cmsghdr *__cmsg) __THROW;
 309 #ifdef __USE_EXTERN_INLINES
 310 # ifndef _EXTERN_INLINE
 311 #  define _EXTERN_INLINE __extern_inline
 312 # endif
 313 _EXTERN_INLINE struct cmsghdr *
 314 __NTH (__cmsg_nxthdr (struct msghdr *__mhdr, struct cmsghdr *__cmsg))
 315 {
 316   if ((size_t) __cmsg->cmsg_len < sizeof (struct cmsghdr))
 317     /* The kernel header does this so there may be a reason.  */
 318     return (struct cmsghdr *) 0;
 319 
 320   __cmsg = (struct cmsghdr *) ((unsigned char *) __cmsg
 321                    + CMSG_ALIGN (__cmsg->cmsg_len));
 322   if ((unsigned char *) (__cmsg + 1) > ((unsigned char *) __mhdr->msg_control
 323                     + __mhdr->msg_controllen)
 324       || ((unsigned char *) __cmsg + CMSG_ALIGN (__cmsg->cmsg_len)
 325       > ((unsigned char *) __mhdr->msg_control + __mhdr->msg_controllen)))
 326     /* No more entries.  */
 327     return (struct cmsghdr *) 0;
 328   return __cmsg;
 329 }
 330 #endif  /* Use `extern inline'.  */
