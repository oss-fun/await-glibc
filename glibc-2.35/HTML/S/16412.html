  23 ! INPUT PARAMETERS
  24 ! res_ptr   o0
  25 ! s1_ptr    o1
  26 ! size      o2
  27 ! s2_limb   o3
  28 
  29 #include <sysdep.h>
  30 
  31 ENTRY(__mpn_submul_1)
  32     sub %g0,%o2,%o2     ! negate ...
  33     sll %o2,2,%o2       ! ... and scale size
  34     sub %o1,%o2,%o1     ! o1 is offset s1_ptr
  35     sub %o0,%o2,%g1     ! g1 is offset res_ptr
  36 
  37     mov 0,%o0           ! clear cy_limb
  38 
  39 LOC(loop):
  40     ld  [%o1+%o2],%o4
  41     ld  [%g1+%o2],%g2
  42     umul    %o4,%o3,%o5
  43     rd  %y,%g3
  44     addcc   %o5,%o0,%o5
  45     addx    %g3,0,%o0
  46     subcc   %g2,%o5,%g2
  47     addx    %o0,0,%o0
  48     st  %g2,[%g1+%o2]
  49 
  50     addcc   %o2,4,%o2
  51     bne LOC(loop)
  52      nop
  53 
  54     retl
  55      nop
  56 
  57 END(__mpn_submul_1)
