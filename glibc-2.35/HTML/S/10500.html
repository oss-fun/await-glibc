  27 /* Like __libc_siglongjmp(), but safe for crossing from alternate
  28    signal stack to normal stack.  Needed by NPTL.  */
  29 void
  30 __libc_unwind_longjmp (sigjmp_buf env, int val)
  31 {
  32   /* Perform any cleanups needed by the frames being unwound.  */
  33   __pthread_cleanup_upto (env->__jmpbuf, CURRENT_STACK_FRAME);
  34 
  35   if (env[0].__mask_was_saved)
  36     /* Restore the saved signal mask.  */
  37     __libc_signal_restore_set (&env[0].__saved_mask);
  38 
  39   /* Call the machine-dependent function to restore machine state.  */
  40   __sigstack_longjmp (env[0].__jmpbuf, val ?: 1);
  41 }
