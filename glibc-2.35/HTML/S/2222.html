  22 /* Look up the value of the next multibyte character and return its numerical
  23    value if it is one of the digits known in the locale.  If *DECIDED is
  24    -1 this means it is not yet decided which form it is and we have to
  25    search through all available digits.  Otherwise we know which script
  26    the digits are from.  */
  27 static inline int
  28 indigit_value (const char **s, size_t *len, int *decided)
  29 {
  30   int from_level;
  31   int to_level;
  32   const char *mbdigits[10];
  33   int i;
  34   int n;
  35 
  36   if (*decided != -1)
  37     from_level = to_level = *decided;
  38   else
  39     {
  40       from_level = 0;
  41       to_level = _NL_CURRENT_WORD (LC_CTYPE, _NL_CTYPE_INDIGITS_MB_LEN) - 1;
  42       assert (from_level <= to_level);
  43     }
  44 
  45   /* In this round we get the pointer to the digit strings and also perform
  46      the first round of comparisons.  */
  47   for (n = 0; n < 10; ++n)
  48     {
  49       size_t dlen;
  50 
  51       /* Get the string for the digits with value N.  */
  52       mbdigits[n] = _NL_CURRENT (LC_CTYPE, _NL_CTYPE_INDIGITS0_MB + n);
  53       dlen = strlen (mbdigits[n]);
  54 
  55       if (from_level == 0 && dlen <= *len
  56       && memcmp (*s, mbdigits[n], dlen) == 0)
  57     {
  58       /* Found it.  */
  59       *s += dlen;
  60       *len -= dlen;
  61       if (*decided == -1)
  62         *decided = 0;
  63       return n;
  64     }
  65 
  66       /* Advance the pointer to the next string.  */
  67       mbdigits[n] += dlen + 1;
  68     }
  69 
  70   /* Now perform the remaining tests.  */
  71   for (i = 1; i <= to_level; ++i)
  72     {
  73       /* Search all ten digits of this level.  */
  74       for (n = 0; n < 10; ++n)
  75     {
  76       size_t dlen = strlen (mbdigits[n]);
  77 
  78       if (i >= from_level && dlen <= *len
  79           && memcmp (*s, mbdigits[n], dlen) == 0)
  80         {
  81           /* Found it.  */
  82           *s += dlen;
  83           *len -= dlen;
  84           if (*decided == -1)
  85         *decided = from_level;
  86           return n;
  87         }
  88 
  89       /* Advance the pointer to the next string.  */
  90       mbdigits[n] += dlen + 1;
  91     }
  92     }
  93 
  94   /* If we reach this point no matching digit was found.  */
  95   return -1;
  96 }
