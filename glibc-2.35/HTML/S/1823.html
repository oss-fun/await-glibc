  20 /* Set *T to indicate raw mode.  */
  21 void
  22 cfmakeraw (struct termios *t)
  23 {
  24   t->c_iflag &= ~(IGNBRK|BRKINT|PARMRK|ISTRIP|INLCR|IGNCR|ICRNL|IXON);
  25   t->c_oflag &= ~OPOST;
  26   t->c_lflag &= ~(ECHO|ECHONL|ICANON|ISIG|IEXTEN);
  27   t->c_cflag &= ~(CSIZE|PARENB);
  28   t->c_cflag |= CS8;
  29   t->c_cc[VMIN] = 1;        /* read returns when one char is available.  */
  30   t->c_cc[VTIME] = 0;
  31 }
