  42 static int
  43 do_after_exec (void)
  44 {
  45   int result = 0;
  46 
  47   struct mq_attr attr;
  48   if (mq_getattr (after_exec, &attr) == 0)
  49     {
  50       puts ("mq_getattr after exec unexpectedly succeeded");
  51       result = 1;
  52     }
  53   else if (errno != EBADF)
  54     {
  55       printf ("mq_getattr after exec did not fail with EBADF: %m\n");
  56       result = 1;
  57     }
  58 
  59   return result;
  60 }

  86     {
  87       printf ("mq_getattr failed: %m\n");
  88       return 1;

  95   const char *newargv[argc + 2];

  97     newargv[i - 1] = argv[i];
