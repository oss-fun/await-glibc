  22 static void
  23 write_data (FILE *stream)
  24 {
  25   int i;
  26   for (i=0; i<100; i++)
  27     fprintf (stream, "%d\n", i);
  28   if (ferror (stream))
  29     {
  30       fprintf (stderr, "Output to stream failed.\n");
  31       exit (1);
  32     }
