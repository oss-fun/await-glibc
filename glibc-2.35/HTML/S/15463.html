  19 /*
  20  * ALGORITHM DESCRIPTION:
  21  *
  22  *   Basic formula is
  23  *    erf(x) ~ erf(x0) +
  24  *              + exp(-x0*x0)*D*(1+c0+T*P1(T)+D^2*P3(T)+D^4*P5(T)+D^6*p7+D^8*p9)
  25  *   where D=x-x0, T=x0*D
  26  *   x0 is x rounded to a specified number of fractional bits (in this case 7),
  27  *    except that x0=0 for |x|<3.5/128.0 (using x0=0 for first 4 table entries)
  28  *
  29  *   Data table packs both erf(x0)_high and a few bits of erf(x0)_low in one
  30  *   entry (in place of redundant exponent bits)
  31  *
  32  */

  56     .section .text.avx2,"ax",@progbits
  57 ENTRY(_ZGVdN4v_erf_avx2)
  58 /*
  59  * vector gather: erf(x0),
  60  * second value is exp(-x0*x0)
  61  */
  62         lea       __svml_derf_data_internal(%rip), %rdi
  63         vmovupd   _SRound+__svml_derf_data_internal(%rip), %ymm6
  64         vandpd    _AbsMask+__svml_derf_data_internal(%rip), %ymm0, %ymm5
  65 
  66 /*
  67  * erf(x) rounds to 1.0 for x>_MaxThreshold (5.9921875)
  68  * can compute all results in the main path
  69  */
  70         vminpd    _MaxThreshold+__svml_derf_data_internal(%rip), %ymm5, %ymm7
  71         vaddpd    %ymm6, %ymm7, %ymm10
  72         vcmpgt_oqpd _U2Threshold+__svml_derf_data_internal(%rip), %ymm7, %ymm9
  73         vpsllq    $4, %ymm10, %ymm11
  74         vsubpd    %ymm6, %ymm10, %ymm8
  75         vandps    _Mask32+__svml_derf_data_internal(%rip), %ymm11, %ymm12
  76         vsubpd    %ymm8, %ymm7, %ymm3
  77         vmulpd    %ymm3, %ymm8, %ymm2
  78         vandpd    %ymm9, %ymm3, %ymm1
  79 
  80 /* NaN fixup */
  81         vminpd    %ymm5, %ymm3, %ymm3
  82 
  83 /* save sign */
  84         vxorpd    %ymm0, %ymm5, %ymm4
  85 
  86 /* T^2 */
  87         vmulpd    %ymm2, %ymm2, %ymm5
  88         vextractf128 $1, %ymm12, %xmm13
  89         vmovd     %xmm12, %eax
  90         vmovd     %xmm13, %ecx
  91         vpextrd   $2, %xmm12, %edx
  92         vpextrd   $2, %xmm13, %esi
  93         movslq    %eax, %rax
  94         movslq    %edx, %rdx
  95         movslq    %ecx, %rcx
  96         movslq    %esi, %rsi
  97 
  98 /* Sign | Diff */
  99         vxorpd    %ymm4, %ymm3, %ymm12
 100 
 101 /*
 102  * _LA_ polynomial computation
 103  * Start polynomial evaluation
 104  */
 105         vmovupd   _poly1_0+__svml_derf_data_internal(%rip), %ymm3
 106         vmovupd   (%rdi,%rax), %xmm6
 107         vmovupd   (%rdi,%rdx), %xmm7
 108         vmovupd   (%rdi,%rcx), %xmm8
 109         vmovupd   (%rdi,%rsi), %xmm9
 110         vunpcklpd %xmm7, %xmm6, %xmm14
 111         vunpcklpd %xmm9, %xmm8, %xmm15
 112 
 113 /* D2 = Diff^2 */
 114         vmulpd    %ymm1, %ymm1, %ymm13
 115         vfmadd213pd _poly1_1+__svml_derf_data_internal(%rip), %ymm2, %ymm3
 116         vmovupd   _poly5_0+__svml_derf_data_internal(%rip), %ymm1
 117         vunpckhpd %xmm9, %xmm8, %xmm10
 118         vfmadd213pd _poly1_2+__svml_derf_data_internal(%rip), %ymm2, %ymm3
 119         vfmadd213pd _poly5_1+__svml_derf_data_internal(%rip), %ymm2, %ymm1
 120         vfmadd213pd _poly1_3+__svml_derf_data_internal(%rip), %ymm2, %ymm3
 121         vfmadd213pd _poly3_3+__svml_derf_data_internal(%rip), %ymm13, %ymm1
 122 
 123 /* P1 = T^2*P1 - T */
 124         vfmsub213pd %ymm2, %ymm5, %ymm3
 125         vinsertf128 $1, %xmm15, %ymm14, %ymm0
 126         vunpckhpd %xmm7, %xmm6, %xmm14
 127         vmovupd   _poly3_0+__svml_derf_data_internal(%rip), %ymm6
 128         vfmadd213pd _poly3_1+__svml_derf_data_internal(%rip), %ymm2, %ymm6
 129         vfmadd213pd _poly3_2+__svml_derf_data_internal(%rip), %ymm2, %ymm6
 130         vfmadd213pd %ymm1, %ymm2, %ymm6
 131 
 132 /* P1 + P3*D2 */
 133         vfmadd213pd %ymm3, %ymm13, %ymm6
 134 
 135 /* Sign | _Erf_H */
 136         vxorpd    %ymm4, %ymm0, %ymm0
 137         vinsertf128 $1, %xmm10, %ymm14, %ymm11
 138 
 139 /* exp_h(x0) * Diff */
 140         vmulpd    %ymm12, %ymm11, %ymm2
 141 
 142 /*
 143  * branch-free
 144  * low part of result: exp_h(x0) * Diff*(1+P1)
 145  */
 146         vfmadd213pd %ymm2, %ymm2, %ymm6
 147 
 148 /* Final result */
 149         vaddpd    %ymm6, %ymm0, %ymm15
 150 
 151 /* Fix erf(-0) = -0 */
 152         vorpd     %ymm4, %ymm15, %ymm0
 153         ret
 154 
 155 END(_ZGVdN4v_erf_avx2)
