 312 static int
 313 do_test (void)
 314 {
 315   const struct ltest *lt;
 316   char *ep;
 317   int status = 0;
 318   int save_errno;
 319 
 320   for (lt = tests; lt->str != NULL; ++lt)
 321     {
 322       long long int l;
 323 
 324       errno = 0;
 325       l = strtoll (lt->str, &ep, lt->base);
 326       save_errno = errno;
 327       printf ("strtoll(\"%s\", , %d) test %u",
 328           lt->str, lt->base, (unsigned int) (lt - tests));
 329       if (l == (long long int) lt->expect && *ep == lt->left
 330       && save_errno == lt->err)
 331     puts("\tOK");
 332       else
 333     {
 334       puts("\tBAD");
 335       if (l != (long long int) lt->expect)
 336         printf("  returns %lld, expected %lld\n",
 337            l, (long long int) lt->expect);
 338       if (lt->left != *ep)
 339         {
 340           char exp1[5], exp2[5];
 341           expand (exp1, *ep);
 342           expand (exp2, lt->left);
 343           printf ("  leaves '%s', expected '%s'\n", exp1, exp2);
 344         }
 345       if (save_errno != lt->err)
 346         printf ("  errno %d (%s)  instead of %d (%s)\n",
 347             save_errno, strerror (save_errno),
 348             lt->err, strerror (lt->err));
 349       status = 1;
 350     }
 351     }
 352 
 353   for (++lt; lt->str != NULL; lt++)
 354     {
 355       unsigned long long int ul;
 356 
 357       errno = 0;
 358       ul = strtoull (lt->str, &ep, lt->base);
 359       save_errno = errno;
 360       printf ("strtoull(\"%s\", , %d) test %u",
 361           lt->str, lt->base, (unsigned int) (lt - tests));
 362       if (ul == lt->expect && *ep == lt->left && save_errno == lt->err)
 363     puts("\tOK");
 364       else
 365     {
 366       puts ("\tBAD");
 367       if (ul != lt->expect)
 368         printf ("  returns %llu, expected %llu\n",
 369             ul, lt->expect);
 370       if (lt->left != *ep)
 371         {
 372           char exp1[5], exp2[5];
 373           expand (exp1, *ep);
 374           expand (exp2, lt->left);
 375           printf ("  leaves '%s', expected '%s'\n", exp1, exp2);
 376         }
 377       if (save_errno != lt->err)
 378         printf ("  errno %d (%s) instead of %d (%s)\n",
 379             save_errno, strerror (save_errno),
 380             lt->err, strerror (lt->err));
 381       status = 1;
 382     }
 383     }
 384 
 385   return status ? EXIT_FAILURE : EXIT_SUCCESS;
 386 }

 388 static void
 389 expand (char *dst, int c)
 390 {
 391   if (isprint (c))
 392     {
 393       dst[0] = c;
 394       dst[1] = '\0';
 395     }
 396   else
 397     (void) sprintf (dst, "%#.3o", (unsigned int) c);
 398 }
