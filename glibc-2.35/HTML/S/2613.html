  23 /* This is used from only one place: dl-misc.c:_dl_debug_vdprintf.
  24    Hence it's in a header with the expectation it will be inlined.
  25 
  26    This is writev, but with a constraint added and others loosened:
  27 
  28    1. Under RTLD_PRIVATE_ERRNO, it must not clobber the private errno
  29       when another thread holds the dl_load_lock.
  30    2. It is not obliged to detect and report errors at all.
  31    3. It's not really obliged to deliver a single atomic write
  32       (though it may be preferable).  */
  33 
  34 static inline void
  35 _dl_writev (int fd, const struct iovec *iov, size_t niov)
  36 {
  37   /* Note that if __writev is an implementation that calls malloc,
  38      this will cause linking problems building the dynamic linker.  */
  39 
  40 #if RTLD_PRIVATE_ERRNO
  41   /* We have to take this lock just to be sure we don't clobber the private
  42      errno when it's being used by another thread that cares about it.
  43      Yet we must be sure not to try calling the lock functions before
  44      the thread library is fully initialized.  */
  45   if (__glibc_unlikely (_dl_starting_up))
  46     __writev (fd, iov, niov);
  47   else
  48     {
  49       __rtld_lock_lock_recursive (GL(dl_load_lock));
  50       __writev (fd, iov, niov);
  51       __rtld_lock_unlock_recursive (GL(dl_load_lock));
  52     }
  53 #else
  54   __writev (fd, iov, niov);
  55 #endif
  56 }
