  64 #define atomic_decrement_if_positive(mem) \
  65   ({ __typeof (*mem) __oldval, __val;                         \
  66      __typeof (mem) __memp = (mem);                       \
  67                                           \
  68      __val = (*__memp);                               \
  69      do                                       \
  70        {                                      \
  71      __oldval = __val;                            \
  72      if (__builtin_expect (__val <= 0, 0))                    \
  73        break;                                 \
  74      __val = atomic_compare_and_exchange_val_acq (__memp,   __oldval - 1, \
  75                               __oldval);          \
  76        }                                      \
  77      while (__builtin_expect (__val != __oldval, 0));                 \
  78      __oldval; })

  80 #define atomic_bit_test_set(mem, bit) \
  81   ({ __typeof (*mem) __oldval, __val;                         \
  82      __typeof (mem) __memp = (mem);                       \
  83      __typeof (*mem) __mask = ((__typeof (*mem)) 1 << (bit));             \
  84                                           \
  85      __val = (*__memp);                               \
  86      do                                       \
  87        {                                      \
  88      __oldval = __val;                            \
  89      __val = atomic_compare_and_exchange_val_acq (__memp,             \
  90                               __oldval | __mask,      \
  91                               __oldval);          \
  92        }                                      \
  93      while (__builtin_expect (__val != __oldval, 0));                 \
  94      __oldval & __mask; })
