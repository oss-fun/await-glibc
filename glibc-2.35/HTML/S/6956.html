  36 static void
  37 do_one_test (impl_t *impl, const char *s1, const char *s2, char *exp_result)
  38 {
  39   size_t i, iters = INNER_LOOP_ITERS_SMALL;
  40   timing_t start, stop, cur;
  41 
  42   TIMING_NOW (start);
  43   for (i = 0; i < iters; ++i)
  44     {
  45       CALL (impl, s1, s2);
  46     }
  47   TIMING_NOW (stop);
  48 
  49   TIMING_DIFF (cur, start, stop);
  50 
  51   TIMING_PRINT_MEAN ((double) cur, (double) iters);
  52 }

  55 static void
  56 do_test (size_t align1, size_t align2, size_t len1, size_t len2,
  57      int fail)
  58 {
  59   char *s1 = (char *) (buf1 + align1);
  60   char *s2 = (char *) (buf2 + align2);
  61 
  62   static const char d[] = "1234567890abcxyz";
  63 #define dl (sizeof (d) - 1)
  64   char *ss2 = s2;
  65   for (size_t l = len2; l > 0; l = l > dl ? l - dl : 0)
  66     {
  67       size_t t = l > dl ? dl : l;
  68       ss2 = mempcpy (ss2, d, t);
  69     }
  70   s2[len2] = '\0';
  71 
  72   if (fail)
  73     {
  74       char *ss1 = s1;
  75       for (size_t l = len1; l > 0; l = l > dl ? l - dl : 0)
  76     {
  77       size_t t = l > dl ? dl : l;
  78       memcpy (ss1, d, t);
  79       ++ss1[len2 > 7 ? 7 : len2 - 1];
  80       ss1 += t;
  81     }
  82     }
  83   else
  84     {
  85       memset (s1, '0', len1);
  86       for (size_t i = 0; i < len2; ++i)
  87     s1[len1 - len2 + i] = toupper (s2[i]);
  88     }
  89   s1[len1] = '\0';
  90 
  91   printf ("Length %4zd/%zd, alignment %2zd/%2zd, %s:",
  92       len1, len2, align1, align2, fail ? "fail" : "found");
  93 
  94   FOR_EACH_IMPL (impl, 0)
  95     do_one_test (impl, s1, s2, fail ? NULL : s1 + len1 - len2);
  96 
  97   putchar ('\n');
  98 }
