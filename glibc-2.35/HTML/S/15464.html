  19 /*
  20  * ALGORITHM DESCRIPTION:
  21  *
  22  *   Compute asinh(x) as log(x + sqrt(x*x + 1))
  23  *   using RSQRT instructions for starting the
  24  *   square root approximation, and small table lookups for log
  25  *   that map to AVX-512 permute instructions
  26  *
  27  *   Special cases:
  28  *
  29  *   asinh(NaN) = quiet NaN, and raise invalid exception
  30  *   asinh(INF) = that INF
  31  *   asinh(0)   = that 0
  32  *
  33  */
  34

  66         cfi_offset(6, -16)
  67         andq      $-64, %rsp
  68         subq      $192, %rsp
  69         vmovaps   %zmm0, %zmm10

  71 /* x^2 */

  73         vmovups   One+__svml_sasinh_data_internal_avx512(%rip), %zmm2
  74 
  75 /* polynomial computation for small inputs */
  76         vmovups   ca1+__svml_sasinh_data_internal_avx512(%rip), %zmm1
  77 
  78 /* not a very small input ? */

  80 
  81 /* 1+x^2 */

  83 
  84 /* |input| */

  89         vrsqrt14ps %zmm7, %zmm8

  93         vcmpps    $21, {sae}, %zmm11, %zmm12, %k2
  94

  97 
  98 /* sign bit */

 113         vmulps    {rn-sae}, %zmm1, %zmm0, %zmm9

 115         vmovups   c1s+__svml_sasinh_data_internal_avx512(%rip), %zmm1

 123         vfmsub213ps {rn-sae}, %zmm0, %zmm10, %zmm4
 124

 129         vmovaps   %zmm2, %zmm0

 131 
 132 /* Sl = (Yh*R0)_low+(R0*Yl) */

 134

 137 
 138 /* rel. error term: Eh=(1-Sh*R0)-Sl*R0 */
 139         vfnmadd231ps {rn-sae}, %zmm5, %zmm8, %zmm0
 140

 142         vmovups   c2s+__svml_sasinh_data_internal_avx512(%rip), %zmm8
 143         vcmpps    $21, {sae}, %zmm7, %zmm12, %k1
 144 
 145 /* Sh*Eh */
 146         vmulps    {rn-sae}, %zmm0, %zmm6, %zmm4
 147         vfmadd231ps {rn-sae}, %zmm0, %zmm8, %zmm1
 148

 150         vfmadd213ps {rn-sae}, %zmm5, %zmm1, %zmm4
 151 
 152 /* Xh */
 153         vsubps    {rn-sae}, %zmm6, %zmm15, %zmm5
 154 
 155 /* fixup for very large inputs */
 156         vmovups   OneEighth+__svml_sasinh_data_internal_avx512(%rip), %zmm6
 157

 159         vaddps    {rn-sae}, %zmm4, %zmm15, %zmm3
 160 
 161 /* Xl */
 162         vsubps    {rn-sae}, %zmm5, %zmm12, %zmm5

 166         vmulps    {rn-sae}, %zmm6, %zmm12, %zmm3{%k1}

 176         vmovups   __svml_sasinh_data_internal_avx512(%rip), %zmm0

 179         vaddps    {rn-sae}, %zmm5, %zmm1, %zmm7

 188 
 189 /* polynomial */
 190         vmovups   poly_coeff3+__svml_sasinh_data_internal_avx512(%rip), %zmm4

 192 /* reduced argument for log(): (Rcp*Xin-1)+Rcp*Xin_low */

 194         vmovups   Four+__svml_sasinh_data_internal_avx512(%rip), %zmm3
 195 
 196 /* exponents */
 197         vgetexpps {sae}, %zmm8, %zmm1
 198 
 199 /* Prepare table index */
 200         vpsrld    $18, %zmm8, %zmm14
 201         vfmadd231ps {rn-sae}, %zmm8, %zmm7, %zmm2
 202         vmovups   poly_coeff2+__svml_sasinh_data_internal_avx512(%rip), %zmm7
 203         vsubps    {rn-sae}, %zmm3, %zmm1, %zmm1{%k1}
 204         vpermt2ps Log_tbl_H+64+__svml_sasinh_data_internal_avx512(%rip), %zmm14, %zmm0
 205         vmovups   Log_tbl_L+__svml_sasinh_data_internal_avx512(%rip), %zmm3
 206         vfmadd231ps {rn-sae}, %zmm2, %zmm4, %zmm7
 207         vfnmadd231ps {rn-sae}, %zmm1, %zmm15, %zmm0
 208 
 209 /* R^2 */
 210         vmulps    {rn-sae}, %zmm2, %zmm2, %zmm6
 211         vfmadd213ps {rn-sae}, %zmm5, %zmm2, %zmm7
 212         vpermt2ps Log_tbl_L+64+__svml_sasinh_data_internal_avx512(%rip), %zmm14, %zmm3
 213 
 214 /* -K*L2L + Tl */
 215         vmovups   L2L+__svml_sasinh_data_internal_avx512(%rip), %zmm14
 216         vfnmadd213ps {rn-sae}, %zmm3, %zmm14, %zmm1
 217 
 218 /* Tl + R^2*Poly */
 219         vfmadd213ps {rn-sae}, %zmm1, %zmm6, %zmm7
 220 
 221 /* R+Tl + R^2*Poly */
 222         vaddps    {rn-sae}, %zmm2, %zmm7, %zmm2
 223         vaddps    {rn-sae}, %zmm2, %zmm0, %zmm9{%k2}
 224         vxorps    %zmm13, %zmm9, %zmm0
 225         testl     %edx, %edx
 226 
 227 /* Go to special inputs processing branch */
 228         jne       L(SPECIAL_VALUES_BRANCH)
 229                                 # LOE rbx r12 r13 r14 r15 edx zmm0 zmm10
