  39   const char *p = strrchr (map->l_name, '/');
  40   const char *l_name = p == NULL ? map->l_name : p + 1;
  41   uintptr_t ck = -1;
  42   if (strncmp (l_name, TEST_NAME, strlen (TEST_NAME)) == 0)
