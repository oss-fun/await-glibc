 118 __attribute ((always_inline))
 119 check_avx (void)
 120 {
 121   if (avx == -1)
 122     {
 123       unsigned int eax, ebx, ecx, edx;
 124 
 125       if (__get_cpuid (1, &eax, &ebx, &ecx, &edx)
 126       && (ecx & bit_AVX))
 127     avx = 1;
 128       else
 129     avx = 0;
 130     }
 131   return avx;
