  28 /* Structure for MTIOCTOP - magnetic tape operation command.  */
  29 struct mtop
  30   {
  31     short int mt_op;        /* Operations defined below.  */
  32     int mt_count;       /* How many of them.  */
  33   };
  34 #define _IOT_mtop /* Hurd ioctl type field.  */ \
  35   _IOT (_IOTS (short), 1, _IOTS (int), 1, 0, 0)

  37 /* Magnetic Tape operations [Not all operations supported by all drivers].  */
  38 #define MTRESET 0   /* +reset drive in case of problems.  */
  39 #define MTFSF   1   /* Forward space over FileMark,
  40              * position at first record of next file.  */
  41 #define MTBSF   2   /* Backward space FileMark (position before FM).  */
  42 #define MTFSR   3   /* Forward space record.  */
  43 #define MTBSR   4   /* Backward space record.  */
  44 #define MTWEOF  5   /* Write an end-of-file record (mark).  */
  45 #define MTREW   6   /* Rewind.  */
  46 #define MTOFFL  7   /* Rewind and put the drive offline (eject?).  */
  47 #define MTNOP   8   /* No op, set status only (read with MTIOCGET).  */
  48 #define MTRETEN 9   /* Retension tape.  */
  49 #define MTBSFM  10  /* +backward space FileMark, position at FM.  */
  50 #define MTFSFM  11  /* +forward space FileMark, position at FM.  */
  51 #define MTEOM   12  /* Goto end of recorded media (for appending files).
  52                MTEOM positions after the last FM, ready for
  53                appending another file.  */
  54 #define MTERASE 13  /* Erase tape -- be careful!  */
  55 
  56 #define MTRAS1  14  /* Run self test 1 (nondestructive).  */
  57 #define MTRAS2  15  /* Run self test 2 (destructive).  */
  58 #define MTRAS3  16  /* Reserved for self test 3.  */
  59 
  60 #define MTSETBLK 20 /* Set block length (SCSI).  */
  61 #define MTSETDENSITY 21 /* Set tape density (SCSI).  */
  62 #define MTSEEK  22  /* Seek to block (Tandberg, etc.).  */
  63 #define MTTELL  23  /* Tell block (Tandberg, etc.).  */
  64 #define MTSETDRVBUFFER 24 /* Set the drive buffering according to SCSI-2.
  65                  Ordinary buffered operation with code 1.  */
  66 #define MTFSS   25  /* Space forward over setmarks.  */
  67 #define MTBSS   26  /* Space backward over setmarks.  */
  68 #define MTWSM   27  /* Write setmarks.  */
  69 
  70 #define MTLOCK  28  /* Lock the drive door.  */
  71 #define MTUNLOCK 29 /* Unlock the drive door.  */
  72 #define MTLOAD  30  /* Execute the SCSI load command.  */
  73 #define MTUNLOAD 31 /* Execute the SCSI unload command.  */
  74 #define MTCOMPRESSION 32/* Control compression with SCSI mode page 15.  */
  75 #define MTSETPART 33    /* Change the active tape partition.  */
  76 #define MTMKPART  34    /* Format the tape with one or two partitions.  */

  78 /* structure for MTIOCGET - mag tape get status command */
  79 
  80 struct mtget
  81   {
  82     long int mt_type;       /* Type of magtape device.  */
  83     long int mt_resid;      /* Residual count: (not sure)
  84                    number of bytes ignored, or
  85                    number of files not skipped, or
  86                    number of records not skipped.  */
  87     /* The following registers are device dependent.  */
  88     long int mt_dsreg;      /* Status register.  */
  89     long int mt_gstat;      /* Generic (device independent) status.  */
  90     long int mt_erreg;      /* Error register.  */
  91     /* The next two fields are not always used.  */
  92     __daddr_t mt_fileno;    /* Number of current file on tape.  */
  93     __daddr_t mt_blkno;     /* Current block number.  */
  94   };
  95 #define _IOT_mtget /* Hurd ioctl type field.  */ \
  96   _IOT (_IOTS (long), 7, 0, 0, 0, 0)
  97

 173 
 174     unsigned long int debug;    /* Debugging flags.  */
 175 
 176     unsigned have_dens:1;
 177     unsigned have_bsf:1;
 178     unsigned have_fsr:1;
 179     unsigned have_bsr:1;
 180     unsigned have_eod:1;
 181     unsigned have_seek:1;
 182     unsigned have_tell:1;
 183     unsigned have_ras1:1;
 184     unsigned have_ras2:1;
 185     unsigned have_ras3:1;
 186     unsigned have_qfa:1;
 187 
 188     unsigned pad1:5;
 189     char reserved[10];
 190   };

 195 /* Magnetic tape I/O control commands.  */
 196 #define MTIOCTOP    _IOW('m', 1, struct mtop)   /* Do a mag tape op. */
 197 #define MTIOCGET    _IOR('m', 2, struct mtget)  /* Get tape status.  */
 198 #define MTIOCPOS    _IOR('m', 3, struct mtpos)  /* Get tape position.*/
 199 
 200 /* The next two are used by the QIC-02 driver for runtime reconfiguration.
 201    See tpqic02.h for struct mtconfiginfo.  */
 202 #define MTIOCGETCONFIG  _IOR('m', 4, struct mtconfiginfo) /* Get tape config.*/
 203 #define MTIOCSETCONFIG  _IOW('m', 5, struct mtconfiginfo) /* Set tape config.*/

 205 /* Generic Mag Tape (device independent) status macros for examining
 206    mt_gstat -- HP-UX compatible.
 207    There is room for more generic status bits here, but I don't
 208    know which of them are reserved. At least three or so should
 209    be added to make this really useful.  */
 210 #define GMT_EOF(x)              ((x) & 0x80000000)
 211 #define GMT_BOT(x)              ((x) & 0x40000000)
 212 #define GMT_EOT(x)              ((x) & 0x20000000)
 213 #define GMT_SM(x)               ((x) & 0x10000000)  /* DDS setmark */
 214 #define GMT_EOD(x)              ((x) & 0x08000000)  /* DDS EOD */
 215 #define GMT_WR_PROT(x)          ((x) & 0x04000000)
 216 /* #define GMT_ ?       ((x) & 0x02000000) */
 217 #define GMT_ONLINE(x)           ((x) & 0x01000000)
 218 #define GMT_D_6250(x)           ((x) & 0x00800000)
 219 #define GMT_D_1600(x)           ((x) & 0x00400000)
 220 #define GMT_D_800(x)            ((x) & 0x00200000)
 221 /* #define GMT_ ?       ((x) & 0x00100000) */
 222 /* #define GMT_ ?       ((x) & 0x00080000) */
 223 #define GMT_DR_OPEN(x)          ((x) & 0x00040000)  /* Door open (no tape).  */
 224 /* #define GMT_ ?       ((x) & 0x00020000) */
 225 #define GMT_IM_REP_EN(x)        ((x) & 0x00010000)  /* Immediate report mode.*/
 226 /* 16 generic status bits unused.  */
