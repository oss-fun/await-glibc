  25 static ssize_t
  26 do_write (void *cookie, const char *buf, size_t n)
  27 {
  28   error_t error = __io_write ((io_t) cookie, buf, n, -1,
  29                   (mach_msg_type_number_t *) &n);
  30   if (error)
  31     return __hurd_fail (error);
  32   return n;
  33 }

  35 /* Write formatted output to PORT, a Mach port supporting the i/o protocol,
  36    according to the format string FORMAT, using the argument list in ARG.  */
  37 int
  38 vpprintf (io_t port, const char *format, va_list arg)
  39 {
  40   int done;
  41 
  42   struct locked_FILE
  43   {
  44     struct _IO_cookie_file cfile;
  45 #ifdef _IO_MTSAFE_IO
  46     _IO_lock_t lock;
  47 #endif
  48   } temp_f;
  49 #ifdef _IO_MTSAFE_IO
  50   temp_f.cfile.__fp.file._lock = &temp_f.lock;
  51 #endif
  52 
  53   _IO_cookie_init (&temp_f.cfile, _IO_NO_READS,
  54            (void *) port, (cookie_io_functions_t) { write: do_write });
  55 
  56   done = __vfprintf_internal (&temp_f.cfile.__fp.file, format, arg, 0);
  57 
  58   return done;
  59 }
