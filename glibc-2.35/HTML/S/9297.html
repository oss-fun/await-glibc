 414 /* Compute sin (pi * X) for -0.25 <= X <= 0.5.  */
 415 
 416 static _Float128
 417 lg_sinpi (_Float128 x)
 418 {
 419   if (x <= L(0.25))
 420     return __sinl (M_PIl * x);
 421   else
 422     return __cosl (M_PIl * (L(0.5) - x));
 423 }
 424 
 425 /* Compute cos (pi * X) for -0.25 <= X <= 0.5.  */
 426 
 427 static _Float128
 428 lg_cospi (_Float128 x)
 429 {
 430   if (x <= L(0.25))
 431     return __cosl (M_PIl * x);
 432   else
 433     return __sinl (M_PIl * (L(0.5) - x));
 434 }

 436 /* Compute cot (pi * X) for -0.25 <= X <= 0.5.  */
 437 
 438 static _Float128
 439 lg_cotpi (_Float128 x)
 440 {
 441   return lg_cospi (x) / lg_sinpi (x);
 442 }

 444 /* Compute lgamma of a negative argument -50 < X < -2, setting
 445    *SIGNGAMP accordingly.  */
 446 
 447 _Float128
 448 __lgamma_negl (_Float128 x, int *signgamp)
 449 {
 450   /* Determine the half-integer region X lies in, handle exact
 451      integers and determine the sign of the result.  */
 452   int i = floorl (-2 * x);
 453   if ((i & 1) == 0 && i == -2 * x)
 454     return L(1.0) / L(0.0);
 455   _Float128 xn = ((i & 1) == 0 ? -i / 2 : (-i - 1) / 2);
 456   i -= 4;
 457   *signgamp = ((i & 2) == 0 ? -1 : 1);
 458 
 459   SET_RESTORE_ROUNDL (FE_TONEAREST);
 460 
 461   /* Expand around the zero X0 = X0_HI + X0_LO.  */
 462   _Float128 x0_hi = lgamma_zeros[i][0], x0_lo = lgamma_zeros[i][1];
 463   _Float128 xdiff = x - x0_hi - x0_lo;
 464 
 465   /* For arguments in the range -3 to -2, use polynomial
 466      approximations to an adjusted version of the gamma function.  */
 467   if (i < 2)
 468     {
 469       int j = floorl (-8 * x) - 16;
 470       _Float128 xm = (-33 - 2 * j) * L(0.0625);
 471       _Float128 x_adj = x - xm;
 472       size_t deg = poly_deg[j];
 473       size_t end = poly_end[j];
 474       _Float128 g = poly_coeff[end];
 475       for (size_t j = 1; j <= deg; j++)
 476     g = g * x_adj + poly_coeff[end - j];
 477       return __log1pl (g * xdiff / (x - xn));
 478     }
 479 
 480   /* The result we want is log (sinpi (X0) / sinpi (X))
 481      + log (gamma (1 - X0) / gamma (1 - X)).  */
 482   _Float128 x_idiff = fabsl (xn - x), x0_idiff = fabsl (xn - x0_hi - x0_lo);
 483   _Float128 log_sinpi_ratio;
 484   if (x0_idiff < x_idiff * L(0.5))
 485     /* Use log not log1p to avoid inaccuracy from log1p of arguments
 486        close to -1.  */
 487     log_sinpi_ratio = __ieee754_logl (lg_sinpi (x0_idiff)
 488                       / lg_sinpi (x_idiff));
 489   else
 490     {
 491       /* Use log1p not log to avoid inaccuracy from log of arguments
 492      close to 1.  X0DIFF2 has positive sign if X0 is further from
 493      XN than X is from XN, negative sign otherwise.  */
 494       _Float128 x0diff2 = ((i & 1) == 0 ? xdiff : -xdiff) * L(0.5);
 495       _Float128 sx0d2 = lg_sinpi (x0diff2);
 496       _Float128 cx0d2 = lg_cospi (x0diff2);
 497       log_sinpi_ratio = __log1pl (2 * sx0d2
 498                   * (-sx0d2 + cx0d2 * lg_cotpi (x_idiff)));
 499     }
 500 
 501   _Float128 log_gamma_ratio;
 502   _Float128 y0 = 1 - x0_hi;
 503   _Float128 y0_eps = -x0_hi + (1 - y0) - x0_lo;
 504   _Float128 y = 1 - x;
 505   _Float128 y_eps = -x + (1 - y);
 506   /* We now wish to compute LOG_GAMMA_RATIO
 507      = log (gamma (Y0 + Y0_EPS) / gamma (Y + Y_EPS)).  XDIFF
 508      accurately approximates the difference Y0 + Y0_EPS - Y -
 509      Y_EPS.  Use Stirling's approximation.  First, we may need to
 510      adjust into the range where Stirling's approximation is
 511      sufficiently accurate.  */
 512   _Float128 log_gamma_adj = 0;
 513   if (i < 20)
 514     {
 515       int n_up = (21 - i) / 2;
 516       _Float128 ny0, ny0_eps, ny, ny_eps;
 517       ny0 = y0 + n_up;
 518       ny0_eps = y0 - (ny0 - n_up) + y0_eps;
 519       y0 = ny0;
 520       y0_eps = ny0_eps;
 521       ny = y + n_up;
 522       ny_eps = y - (ny - n_up) + y_eps;
 523       y = ny;
 524       y_eps = ny_eps;
 525       _Float128 prodm1 = __lgamma_productl (xdiff, y - n_up, y_eps, n_up);
 526       log_gamma_adj = -__log1pl (prodm1);
 527     }
 528   _Float128 log_gamma_high
 529     = (xdiff * __log1pl ((y0 - e_hi - e_lo + y0_eps) / e_hi)
 530        + (y - L(0.5) + y_eps) * __log1pl (xdiff / y) + log_gamma_adj);
 531   /* Compute the sum of (B_2k / 2k(2k-1))(Y0^-(2k-1) - Y^-(2k-1)).  */
 532   _Float128 y0r = 1 / y0, yr = 1 / y;
 533   _Float128 y0r2 = y0r * y0r, yr2 = yr * yr;
 534   _Float128 rdiff = -xdiff / (y * y0);
 535   _Float128 bterm[NCOEFF];
 536   _Float128 dlast = rdiff, elast = rdiff * yr * (yr + y0r);
 537   bterm[0] = dlast * lgamma_coeff[0];
 538   for (size_t j = 1; j < NCOEFF; j++)
 539     {
 540       _Float128 dnext = dlast * y0r2 + elast;
 541       _Float128 enext = elast * yr2;
 542       bterm[j] = dnext * lgamma_coeff[j];
 543       dlast = dnext;
 544       elast = enext;
 545     }
 546   _Float128 log_gamma_low = 0;
 547   for (size_t j = 0; j < NCOEFF; j++)
 548     log_gamma_low += bterm[NCOEFF - 1 - j];
 549   log_gamma_ratio = log_gamma_high + log_gamma_low;
 550 
 551   return log_sinpi_ratio + log_gamma_ratio;
 552 }
