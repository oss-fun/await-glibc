  25 ssize_t
  26 __libc_pread (int fd, void *buf, size_t nbyte, off_t offset)
  27 {
  28   /* Since we must not change the file pointer preserve the value so that
  29      we can restore it later.  */
  30   int save_errno;
  31   ssize_t result;
  32   off_t old_offset = __libc_lseek (fd, 0, SEEK_CUR);
  33   if (old_offset == (off_t) -1)
  34     return -1;
  35 
  36   /* Set to wanted position.  */
  37   if (__libc_lseek (fd, offset, SEEK_SET) == (off_t) -1)
  38     return -1;
  39 
  40   /* Write out the data.  */
  41   result = __libc_read (fd, buf, nbyte);
  42 
  43   /* Now we have to restore the position.  If this fails we have to
  44      return this as an error.  But if the writing also failed we
  45      return this error.  */
  46   save_errno = errno;
  47   if (__libc_lseek (fd, old_offset, SEEK_SET) == (off_t) -1)
  48     {
  49       if (result == -1)
  50     __set_errno (save_errno);
  51       return -1;
  52     }
  53   __set_errno (save_errno);
  54 
  55   return result;
  56 }
