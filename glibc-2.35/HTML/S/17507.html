  30 int
  31 __ffs (int x)
  32 {
  33     int r;
  34 
  35     if (x == 0)
  36       return 0;
  37     __asm__("    lr   %%r1,%1\n"
  38         "    sr   %0,%0\n"
  39         "    tml  %%r1,0xFFFF\n"
  40         "    jnz  0f\n"
  41         "    ahi  %0,16\n"
  42         "    srl  %%r1,16\n"
  43         "0:  tml  %%r1,0x00FF\n"
  44         "    jnz  1f\n"
  45         "    ahi  %0,8\n"
  46         "    srl  %%r1,8\n"
  47         "1:  tml  %%r1,0x000F\n"
  48         "    jnz  2f\n"
  49         "    ahi  %0,4\n"
  50         "    srl  %%r1,4\n"
  51         "2:  tml  %%r1,0x0003\n"
  52         "    jnz  3f\n"
  53         "    ahi  %0,2\n"
  54         "    srl  %%r1,2\n"
  55         "3:  tml  %%r1,0x0001\n"
  56         "    jnz  4f\n"
  57         "    ahi  %0,1\n"
  58         "4:"
  59         : "=&d" (r) : "d" (x) : "cc", "1" );
  60     return r+1;
  61 }
