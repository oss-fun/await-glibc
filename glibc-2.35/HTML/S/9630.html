  42     .p2align ALIGN

  46 0:  .asciz "GNU"        /* vendor name */

  48     .p2align ALIGN

  50     .long 1         /* pr_type.  */

  57     .long 0x08000800
  58 #endif
  59 4:
  60     .p2align ALIGN
  61 5:
  62
