  60 int
  61 getrpcport (const char *host, u_long prognum, u_long versnum, u_int proto)
  62 {
  63   struct sockaddr_in addr;
  64 
  65   if (__libc_rpc_gethostbyname (host, &addr) != 0)
  66     return 0;
  67   return pmap_getport (&addr, prognum, versnum, proto);
  68 }
