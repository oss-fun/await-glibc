  52 /* Print part of the library search path (from a single source).  */
  53 static void
  54 print_search_path_for_help_1 (struct r_search_path_elem **list)
  55 {
  56   if (list == NULL || list == (void *) -1)
  57     /* Path is missing or marked as inactive.  */
  58     return;
  59 
  60   for (; *list != NULL; ++list)
  61     {
  62       _dl_write (STDOUT_FILENO, "  ", 2);
  63       const char *name = (*list)->dirname;
  64       size_t namelen = (*list)->dirnamelen;
  65       if (namelen == 0)
  66         {
  67           /* The empty string denotes the current directory.  */
  68           name = ".";
  69           namelen = 1;
  70         }
  71       else if (namelen > 1)
  72         /* Remove the trailing slash.  */
  73         --namelen;
  74       _dl_write (STDOUT_FILENO, name, namelen);
  75       _dl_printf (" (%s)\n", (*list)->what);
  76     }
  77 }

  79 /* Prints the library search path.  See _dl_init_paths in dl-load.c
  80    how this information is populated.  */
  81 static void
  82 print_search_path_for_help (struct dl_main_state *state)
  83 {
  84   if (__rtld_search_dirs.dirs == NULL)
  85     /* The run-time search paths have not yet been initialized.  */
  86     call_init_paths (state);
  87 
  88   _dl_printf ("\nShared library search path:\n");
  89 
  90   /* The print order should reflect the processing in
  91      _dl_map_object.  */
  92 
  93   struct link_map *map = GL(dl_ns)[LM_ID_BASE]._ns_loaded;
  94   if (map != NULL)
  95     print_search_path_for_help_1 (map->l_rpath_dirs.dirs);
  96 
  97   print_search_path_for_help_1 (__rtld_env_path_list.dirs);
  98 
  99   if (map != NULL)
 100     print_search_path_for_help_1 (map->l_runpath_dirs.dirs);
 101 
 102   _dl_printf ("  (libraries located via %s)\n", LD_SO_CACHE);
 103 
 104   print_search_path_for_help_1 (__rtld_search_dirs.dirs);
 105 }

 107 /* Helper function for printing flags associated with a HWCAP name.  */
 108 static void
 109 print_hwcap_1 (bool *first, bool active, const char *label)
 110 {
 111   if (active)
 112     {
 113       if (*first)
 114         {
 115           _dl_printf (" (");
 116           *first = false;
 117         }
 118       else
 119         _dl_printf (", ");
 120       _dl_printf ("%s", label);
 121     }
 122 }

 155 /* Print the list of recognized glibc-hwcaps subdirectories.  */
 156 static void
 157 print_hwcaps_subdirectories (const struct dl_main_state *state)
 158 {
 159   bool nothing_printed = true;
 160   struct dl_hwcaps_split split;
 161 
 162   /* The prepended glibc-hwcaps subdirectories.  */
 163   _dl_hwcaps_split_init (&split, state->glibc_hwcaps_prepend);
 164   while (_dl_hwcaps_split (&split))
 165     {
 166       print_hwcaps_subdirectories_header (&nothing_printed);
 167       print_hwcaps_subdirectories_name (&split);
 168       bool first = true;
 169       print_hwcap_1 (&first, true, "searched");
 170       print_hwcap_1_finish (&first);
 171     }
 172 
 173   /* The built-in glibc-hwcaps subdirectories.  Do the filtering
 174      manually, so that more precise diagnostics are possible.  */
 175   uint32_t mask = _dl_hwcaps_subdirs_active ();
 176   _dl_hwcaps_split_init (&split, _dl_hwcaps_subdirs);
 177   while (_dl_hwcaps_split (&split))
 178     {
 179       print_hwcaps_subdirectories_header (&nothing_printed);
 180       print_hwcaps_subdirectories_name (&split);
 181       bool first = true;
 182       print_hwcap_1 (&first, mask & 1, "supported");
 183       bool listed = _dl_hwcaps_contains (state->glibc_hwcaps_mask,
 184                                          split.segment, split.length);
 185       print_hwcap_1 (&first, !listed, "masked");
 186       print_hwcap_1 (&first, (mask & 1) && listed, "searched");
 187       print_hwcap_1_finish (&first);
 188       mask >>= 1;
 189     }
 190 
 191   if (nothing_printed)
 192     _dl_printf ("\n\
 193 No subdirectories of glibc-hwcaps directories are searched.\n");
 194 }

 196 /* Write a list of hwcap subdirectories to standard output.  See
 197  _dl_important_hwcaps in dl-hwcaps.c.  */
 198 static void
 199 print_legacy_hwcap_directories (void)
 200 {
 201   _dl_printf ("\n\
 202 Legacy HWCAP subdirectories under library search path directories:\n");
 203 
 204   const char *platform = GLRO (dl_platform);
 205   if (platform != NULL)
 206     _dl_printf ("  %s (AT_PLATFORM; supported, searched)\n", platform);
 207 
 208   _dl_printf ("  tls (supported, searched)\n");
 209 
 210   uint64_t hwcap_mask = GET_HWCAP_MASK();
 211   uint64_t searched = GLRO (dl_hwcap) & hwcap_mask;
 212   for (int n = 63; n >= 0; --n)
 213     {
 214       uint64_t bit = 1ULL << n;
 215       if (HWCAP_IMPORTANT & bit)
 216         {
 217           _dl_printf ("  %s", _dl_hwcap_string (n));
 218           bool first = true;
 219           print_hwcap_1 (&first, GLRO (dl_hwcap) & bit, "supported");
 220           print_hwcap_1 (&first, !(hwcap_mask & bit), "masked");
 221           print_hwcap_1 (&first, searched & bit, "searched");
 222           print_hwcap_1_finish (&first);
 223         }
 224     }
 225 }
