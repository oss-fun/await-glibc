  22 static inline uintptr_t __attribute__ ((always_inline))
  23 _dl_setup_stack_chk_guard (void *dl_random)
  24 {
  25   union
  26   {
  27     uintptr_t num;
  28     unsigned char bytes[sizeof (uintptr_t)];
  29   } ret = { 0 };
  30 
  31   if (dl_random == NULL)
  32     {
  33       ret.bytes[sizeof (ret) - 1] = 255;
  34       ret.bytes[sizeof (ret) - 2] = '\n';
  35     }
  36   else
  37     {
  38       memcpy (ret.bytes, dl_random, sizeof (ret));
  39 #if BYTE_ORDER == LITTLE_ENDIAN
  40       ret.num &= ~(uintptr_t) 0xff;
  41 #elif BYTE_ORDER == BIG_ENDIAN
  42       ret.num &= ~((uintptr_t) 0xff << (8 * (sizeof (ret) - 1)));
  43 #else
  44 # error "BYTE_ORDER unknown"
  45 #endif
  46     }
  47   return ret.num;
  48 }
