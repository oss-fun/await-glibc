  19 /*
  20  * ALGORITHM DESCRIPTION:
  21  *
  22  *    1+x = 2^k*(xh + xl) is computed in high-low parts; xh in [1,2)
  23  *    Get short reciprocal approximation Rcp ~ 1/xh
  24  *    R = (Rcp*xh - 1.0) + Rcp*xl
  25  *    log1p(x) = k*log(2.0) - log(Rcp) + poly(R)
  26  *       log(Rcp) is tabulated
  27  *
  28  *
  29  */

  63 /* reduction: compute r,n */

  65         vmovups   SgnMask+__svml_slog1p_data_internal(%rip), %zmm4

  68 /* compute 1+x as high, low parts */

  70         vminps    {sae}, %zmm3, %zmm2, %zmm7
  71         vandnps   %zmm3, %zmm4, %zmm1
  72         vpternlogd $255, %zmm4, %zmm4, %zmm4
  73         vaddps    {rn-sae}, %zmm7, %zmm5, %zmm9
  74         vpsubd    %zmm12, %zmm9, %zmm10
  75         vsubps    {rn-sae}, %zmm9, %zmm5, %zmm6

  77 /* check argument value ranges */

  79         vpsrad    $23, %zmm10, %zmm13
  80         vmovups   sPoly_5+__svml_slog1p_data_internal(%rip), %zmm9

  87         vmovups   sPoly_1+__svml_slog1p_data_internal(%rip), %zmm14
  88         vmulps    {rn-sae}, %zmm13, %zmm15, %zmm6

  91         vmovups   sPoly_4+__svml_slog1p_data_internal(%rip), %zmm10
  92         vmovups   sPoly_3+__svml_slog1p_data_internal(%rip), %zmm11
  93         vmovups   sPoly_2+__svml_slog1p_data_internal(%rip), %zmm12
  94 
  95 /* polynomial evaluation */
  96         vsubps    {rn-sae}, %zmm2, %zmm5, %zmm2
  97         vaddps    {rn-sae}, %zmm6, %zmm2, %zmm15
  98         vmovups   sPoly_7+__svml_slog1p_data_internal(%rip), %zmm2
  99         vfmadd231ps {rn-sae}, %zmm15, %zmm7, %zmm2
 100         vpandnd   %zmm8, %zmm8, %zmm4{%k1}
 101         vmovups   sPoly_6+__svml_slog1p_data_internal(%rip), %zmm8
 102 
 103 /* combine and get argument value range mask */
 104         vptestmd  %zmm4, %zmm4, %k0
 105         vfmadd213ps {rn-sae}, %zmm8, %zmm15, %zmm2
 106         kmovw     %k0, %edx
 107         vfmadd213ps {rn-sae}, %zmm9, %zmm15, %zmm2
 108         vfmadd213ps {rn-sae}, %zmm10, %zmm15, %zmm2
 109         vfmadd213ps {rn-sae}, %zmm11, %zmm15, %zmm2
 110         vfmadd213ps {rn-sae}, %zmm12, %zmm15, %zmm2
 111         vfmadd213ps {rn-sae}, %zmm14, %zmm15, %zmm2
 112         vmulps    {rn-sae}, %zmm15, %zmm2, %zmm4
 113         vfmadd213ps {rn-sae}, %zmm15, %zmm15, %zmm4
 114 
 115 /* final reconstruction */
 116         vmovups   sLn2+__svml_slog1p_data_internal(%rip), %zmm15
 117         vfmadd213ps {rn-sae}, %zmm4, %zmm15, %zmm0
 118         vorps     %zmm1, %zmm0, %zmm0
