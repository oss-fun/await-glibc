  24 int
  25 __sem_destroy (sem_t *sem)
  26 {
  27   struct new_sem *isem = (struct new_sem *) sem;
  28   if (
  29 #if __HAVE_64B_ATOMICS
  30       atomic_load_relaxed (&isem->data) >> SEM_NWAITERS_SHIFT
  31 #else
  32       atomic_load_relaxed (&isem->value) & SEM_NWAITERS_MASK
  33       || isem->nwaiters
  34 #endif
  35       )
  36     /* There are threads waiting on *SEM.  */
  37     {
  38       errno = EBUSY;
  39       return -1;
  40     }
  41 
  42   return 0;
  43 }
