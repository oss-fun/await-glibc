  19 #ifndef _BITS_CANCELATION_H
  20 #define _BITS_CANCELATION_H 1
  21 
  22 struct __pthread_cancelation_handler
  23 {
  24   void (*__handler) (void *);
  25   void *__arg;
  26   struct __pthread_cancelation_handler *__next;
  27 };
  28 
  29 /* Returns the thread local location of the cleanup handler stack.  */
  30 struct __pthread_cancelation_handler **__pthread_get_cleanup_stack (void);
  31 
  32 #define __pthread_cleanup_push(rt, rtarg) \
  33     { \
  34       struct __pthread_cancelation_handler **__handlers \
  35         = __pthread_get_cleanup_stack (); \
  36       struct __pthread_cancelation_handler __handler = \
  37         { \
  38           (rt), \
  39           (rtarg), \
  40           *__handlers \
  41         }; \
  42       *__handlers = &__handler;
  43 
  44 #define __pthread_cleanup_pop(execute) \
  45       if (execute) \
  46         __handler.__handler (__handler.__arg); \
  47       *__handlers = __handler.__next; \
  48     }
  49 
  50 #endif /* _BITS_CANCELATION_H */
