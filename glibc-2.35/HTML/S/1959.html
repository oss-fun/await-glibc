 101 static void __attribute__ ((used))
 102 cmdline_process_function (int c)
 103 {
 104   switch (c)
 105     {
 106       ITERATIONS_PROCESS
 107       case OPT_RANDOM:
 108         {
 109           int fdr = open ("/dev/urandom", O_RDONLY);
 110           if (fdr < 0 || read (fdr, &seed, sizeof (seed)) != sizeof (seed))
 111             seed = time (NULL);
 112           if (fdr >= 0)
 113             close (fdr);
 114           do_srandom = 1;
 115           break;
 116         }
 117 
 118       case OPT_SEED:
 119         seed = strtoul (optarg, NULL, 0);
 120         do_srandom = 1;
 121         break;
 122     }
 123 }
