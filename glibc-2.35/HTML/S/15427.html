  19 /*
  20  * ALGORITHM DESCRIPTION:
  21  *
  22  *   exp2(x)  = 2^n * T[j] * (1 + P(y))
  23  *   where
  24  *        x = m*(1/K) + y,    y in [-1/K..1/K]
  25  *        m = n*K + j,           m,n,j - signed integer, j in [-K/2..K/2]
  26  *
  27  *        values of 2^j/K are tabulated
  28  *
  29  *        P(y) is a minimax polynomial approximation of exp2(x)-1
  30  *        on small interval [-1/K..1/K]
  31  *
  32  *  Special cases:
  33  *
  34  *   exp2(NaN)  = NaN
  35  *   exp2(+INF) = +INF
  36  *   exp2(-INF) = 0
  37  *   exp2(x)    = 1 for subnormals
  38  *   For IEEE double
  39  *     if x >= 1024.0 then exp2(x) overflows
  40  *     if x < -1076.0 then exp2(x) underflows
  41  *
  42  */
  43

  62         cfi_def_cfa_offset(16)
  63         movq      %rsp, %rbp
  64         cfi_def_cfa(6, 16)
  65         cfi_offset(6, -16)
  66         andq      $-32, %rsp
  67         subq      $96, %rsp
  68 
  69 /* out, basePtr, iIndex, iBaseOfs, iSize, iGran, iOfs */
  70         lea       __svml_dexp2_data_internal(%rip), %r8
  71         vmovupd   _dbShifter+__svml_dexp2_data_internal(%rip), %ymm4
  72         vmovupd   _lIndexMask+__svml_dexp2_data_internal(%rip), %ymm3
  73         vmovapd   %ymm0, %ymm1
  74 
  75 /*  Load arument  */
  76         vaddpd    %ymm4, %ymm1, %ymm2
  77         vsubpd    %ymm4, %ymm2, %ymm0
  78 
  79 /*  Index and lookup  */
  80         vandps    %ymm3, %ymm2, %ymm9
  81         vpandn    %ymm2, %ymm3, %ymm2
  82 
  83 /*  2^N  */
  84         vpsllq    $45, %ymm2, %ymm3
  85 
  86 /*  R  */
  87         vsubpd    %ymm0, %ymm1, %ymm15
  88 
  89 /* Check for overflow\underflow  */
  90         vextractf128 $1, %ymm1, %xmm5
  91 
  92 /*
  93  *  Polynomial
  94  * poly(dN) = a1*dR+...+a4*dR^4
  95  */
  96         vmovupd   _dPC4+__svml_dexp2_data_internal(%rip), %ymm0
  97         vshufps   $221, %xmm5, %xmm1, %xmm6
  98         vandps    _iAbsMask+__svml_dexp2_data_internal(%rip), %xmm6, %xmm7
  99         vpcmpgtd  _iDomainRange+__svml_dexp2_data_internal(%rip), %xmm7, %xmm8
 100         vfmadd213pd _dPC3+__svml_dexp2_data_internal(%rip), %ymm15, %ymm0
 101         vmovmskps %xmm8, %eax
 102         vfmadd213pd _dPC2+__svml_dexp2_data_internal(%rip), %ymm15, %ymm0
 103 
 104 /* a1+...+a4*dR^3 ! */
 105         vfmadd213pd _dPC1+__svml_dexp2_data_internal(%rip), %ymm15, %ymm0
 106         vextractf128 $1, %ymm9, %xmm12
 107         vmovd     %xmm9, %edx
 108         vmovd     %xmm12, %esi
 109         shll      $3, %edx
 110         vpextrd   $2, %xmm9, %ecx
 111         shll      $3, %esi
 112         vpextrd   $2, %xmm12, %edi
 113         shll      $3, %ecx
 114         vmovq     (%r8,%rdx), %xmm10
 115         shll      $3, %edi
 116         vmovq     (%r8,%rsi), %xmm13
 117         vmovhpd   (%r8,%rcx), %xmm10, %xmm11
 118         vmovhpd   (%r8,%rdi), %xmm13, %xmm14
 119         vinsertf128 $1, %xmm14, %ymm11, %ymm4
 120 
 121 /* dR=dR*dT */
 122         vmulpd    %ymm15, %ymm4, %ymm15
 123 
 124 /*
 125  *  Reconstruction
 126  * exp2 = {2^N later}*(Tj+Tj*poly)
 127  * dN = dT+dT*dR*(a1+...+a4*dR^3)
 128  */
 129         vfmadd213pd %ymm4, %ymm15, %ymm0
 130 
 131 /* quick 2^N */
 132         vpaddq    %ymm3, %ymm0, %ymm0
 133 
 134 /*  Finish   */
 135         testl     %eax, %eax
 136 
 137 /* Go to special inputs processing branch */
 138         jne       L(SPECIAL_VALUES_BRANCH)
 139                                 # LOE rbx r12 r13 r14 r15 eax ymm0 ymm1
 140 
 141 /* Restore registers
 142  * and exit the function
 143  */
 144 
 145 L(EXIT):
 146         movq      %rbp, %rsp
 147         popq      %rbp
 148         cfi_def_cfa(7, 8)
 149         cfi_restore(6)
 150         ret
 151         cfi_def_cfa(6, 16)
 152         cfi_offset(6, -16)
 153

 155  * special inputs
 156  */
 157

 178                                 # LOE rbx r15 r12d r13d
 179 
 180 /* Range mask
 181  * bits check
 182  */
 183

 185         btl       %r12d, %r13d
 186 
 187 /* Call scalar math function */
 188         jc        L(SCALAR_MATH_CALL)
 189                                 # LOE rbx r15 r12d r13d

 214         .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0xb0, 0xff, 0xff, 0xff, 0x22
