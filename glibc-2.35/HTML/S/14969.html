  19 /*
  20  * ALGORITHM DESCRIPTION:
  21  *
  22  *   exp2(x)  = 2^n * T[j] * (1 + P(y))
  23  *   where
  24  *        x = m*(1/K) + y,    y in [-1/K..1/K]
  25  *        m = n*K + j,           m,n,j - signed integer, j in [-K/2..K/2]
  26  *
  27  *        values of 2^j/K are tabulated
  28  *
  29  *        P(y) is a minimax polynomial approximation of exp2(x)-1
  30  *        on small interval [-1/K..1/K]
  31  *
  32  *  Special cases:
  33  *
  34  *   exp2(NaN)  = NaN
  35  *   exp2(+INF) = +INF
  36  *   exp2(-INF) = 0
  37  *   exp2(x)    = 1 for subnormals
  38  *   For IEEE double
  39  *     if x >= 1024.0 then exp2(x) overflows
  40  *     if x < -1076.0 then exp2(x) underflows
  41  *
  42  */
  43 
  44 /* Offsets for data table __svml_dexp2_data_internal
  45  */
  46 #define _dbT                            0
  47 #define _dbShifter                      1024
  48 #define _dPC1                           1040
  49 #define _dPC2                           1056
  50 #define _dPC3                           1072
  51 #define _dPC4                           1088
  52 #define _lIndexMask                     1104
  53 #define _iAbsMask                       1120
  54 #define _iDomainRange                   1136

  56 #include <sysdep.h>

  58         .text
  59     .section .text.sse4,"ax",@progbits
  60 ENTRY(_ZGVbN2v_exp2_sse4)
  61         subq      $72, %rsp
  62         cfi_def_cfa_offset(80)
  63 
  64 /*  R  */
  65         movaps    %xmm0, %xmm7
  66         movups    _dbShifter+__svml_dexp2_data_internal(%rip), %xmm1
  67 
  68 /* out, basePtr, iIndex, iBaseOfs, iSize, iGran, iOfs */
  69         lea       __svml_dexp2_data_internal(%rip), %rsi
  70 
  71 /*  Load arument  */
  72         movaps    %xmm1, %xmm10
  73         addpd     %xmm0, %xmm10
  74         movaps    %xmm10, %xmm6
  75         subpd     %xmm1, %xmm6
  76         subpd     %xmm6, %xmm7
  77 
  78 /*
  79  *  Polynomial
  80  * poly(dN) = a1*dR+...+a4*dR^4
  81  */
  82         movups    _dPC4+__svml_dexp2_data_internal(%rip), %xmm8
  83         mulpd     %xmm7, %xmm8
  84         addpd     _dPC3+__svml_dexp2_data_internal(%rip), %xmm8
  85         mulpd     %xmm7, %xmm8
  86         addpd     _dPC2+__svml_dexp2_data_internal(%rip), %xmm8
  87         movdqu    _lIndexMask+__svml_dexp2_data_internal(%rip), %xmm9
  88 
  89 /*  Index and lookup  */
  90         movdqa    %xmm9, %xmm5
  91         pandn     %xmm10, %xmm9
  92         pand      %xmm10, %xmm5
  93 
  94 /*  2^N  */
  95         psllq     $45, %xmm9
  96         movd      %xmm5, %eax
  97         movq      _iAbsMask+__svml_dexp2_data_internal(%rip), %xmm2
  98 
  99 /* Check for overflow\underflow  */
 100         pshufd    $221, %xmm0, %xmm4
 101         pextrw    $4, %xmm5, %ecx
 102 
 103 /* a1+...+a4*dR^3 ! */
 104         mulpd     %xmm7, %xmm8
 105         shll      $3, %eax
 106         pand      %xmm2, %xmm4
 107         shll      $3, %ecx
 108         movq      (%rsi,%rax), %xmm1
 109         movhpd    (%rsi,%rcx), %xmm1
 110 
 111 /* dR=dR*dT */
 112         mulpd     %xmm1, %xmm7
 113         addpd     _dPC1+__svml_dexp2_data_internal(%rip), %xmm8
 114 
 115 /*
 116  *  Reconstruction
 117  * exp2 = {2^N later}*(Tj+Tj*poly)
 118  * dN = dT+dT*dR*(a1+...+a4*dR^3)
 119  */
 120         mulpd     %xmm7, %xmm8
 121         addpd     %xmm8, %xmm1
 122         movq      _iDomainRange+__svml_dexp2_data_internal(%rip), %xmm3
 123         pcmpgtd   %xmm3, %xmm4
 124         movmskps  %xmm4, %edx
 125 
 126 /* quick 2^N */
 127         paddq     %xmm9, %xmm1
 128         andl      $3, %edx
 129 
 130 /*  Finish   */
 131 
 132 /* Go to special inputs processing branch */
 133         jne       L(SPECIAL_VALUES_BRANCH)
 134                                 # LOE rbx rbp r12 r13 r14 r15 edx xmm0 xmm1
 135 
 136 /* Restore registers
 137  * and exit the function
 138  */
 139 
 140 L(EXIT):
 141         movaps    %xmm1, %xmm0
 142         addq      $72, %rsp
 143         cfi_def_cfa_offset(8)
 144         ret
 145         cfi_def_cfa_offset(80)
 146 
 147 /* Branch to process
 148  * special inputs
 149  */
 150 
 151 L(SPECIAL_VALUES_BRANCH):
 152         movups    %xmm0, 32(%rsp)
 153         movups    %xmm1, 48(%rsp)
 154                                 # LOE rbx rbp r12 r13 r14 r15 edx xmm1
 155 
 156         xorl      %eax, %eax
 157         movq      %r12, 16(%rsp)
 158         cfi_offset(12, -64)
 159         movl      %eax, %r12d
 160         movq      %r13, 8(%rsp)
 161         cfi_offset(13, -72)
 162         movl      %edx, %r13d
 163         movq      %r14, (%rsp)
 164         cfi_offset(14, -80)
 165                                 # LOE rbx rbp r15 r12d r13d
 166 
 167 /* Range mask
 168  * bits check
 169  */
 170 
 171 L(RANGEMASK_CHECK):
 172         btl       %r12d, %r13d
 173 
 174 /* Call scalar math function */
 175         jc        L(SCALAR_MATH_CALL)
 176                                 # LOE rbx rbp r15 r12d r13d
 177 
 178 /* Special inputs
 179  * processing loop
 180  */
 181 
 182 L(SPECIAL_VALUES_LOOP):
 183         incl      %r12d
 184         cmpl      $2, %r12d
 185 
 186 /* Check bits in range mask */
 187         jl        L(RANGEMASK_CHECK)
 188                                 # LOE rbx rbp r15 r12d r13d
 189 
 190         movq      16(%rsp), %r12
 191         cfi_restore(12)
 192         movq      8(%rsp), %r13
 193         cfi_restore(13)
 194         movq      (%rsp), %r14
 195         cfi_restore(14)
 196         movups    48(%rsp), %xmm1
 197 
 198 /* Go to exit */
 199         jmp       L(EXIT)
 200         cfi_offset(12, -64)
 201         cfi_offset(13, -72)
 202         cfi_offset(14, -80)
 203                                 # LOE rbx rbp r12 r13 r14 r15 xmm1
 204 
 205 /* Scalar math fucntion call
 206  * to process special input
 207  */
 208 
 209 L(SCALAR_MATH_CALL):
 210         movl      %r12d, %r14d
 211         movsd     32(%rsp,%r14,8), %xmm0
 212         call      exp2@PLT
 213                                 # LOE rbx rbp r14 r15 r12d r13d xmm0
 214 
 215         movsd     %xmm0, 48(%rsp,%r14,8)
 216 
 217 /* Process special inputs in loop */
 218         jmp       L(SPECIAL_VALUES_LOOP)
 219                                 # LOE rbx rbp r15 r12d r13d
 220 END(_ZGVbN2v_exp2_sse4)
