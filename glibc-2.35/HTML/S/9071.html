  23 #define CBRT2 1.2599210498948731648     /* 2^(1/3) */
  24 #define SQR_CBRT2 1.5874010519681994748     /* 2^(2/3) */

  26 static const double factor[5] =
  27 {
  28   1.0 / SQR_CBRT2,
  29   1.0 / CBRT2,
  30   1.0,
  31   CBRT2,
  32   SQR_CBRT2
  33 };

  36 double
  37 __cbrt (double x)
  38 {
  39   double xm, ym, u, t2;
  40   int xe;
  41 
  42   /* Reduce X.  XM now is an range 1.0 to 0.5.  */
  43   xm = __frexp (fabs (x), &xe);
  44 
  45   /* If X is not finite or is null return it (with raising exceptions
  46      if necessary.
  47      Note: *Our* version of `frexp' sets XE to zero if the argument is
  48      Inf or NaN.  This is not portable but faster.  */
  49   if (xe == 0 && fpclassify (x) <= FP_ZERO)
  50     return x + x;
  51 
  52   u = (0.354895765043919860
  53        + ((1.50819193781584896
  54        + ((-2.11499494167371287
  55            + ((2.44693122563534430
  56            + ((-1.83469277483613086
  57                + (0.784932344976639262 - 0.145263899385486377 * xm)
  58                       * xm)
  59               * xm))
  60           * xm))
  61           * xm))
  62       * xm));
  63 
  64   t2 = u * u * u;
  65 
  66   ym = u * (t2 + 2.0 * xm) / (2.0 * t2 + xm) * factor[2 + xe % 3];
  67 
  68   return __ldexp (x > 0.0 ? ym : -ym, xe / 3);
  69 }
