  34 #define CHECK_PRINTF(EXPECTED, FMT, ...)                \
  35   do                                    \
  36     {                                   \
  37       int ret = SNPRINTF (buf, sizeof buf / sizeof buf[0], L_(FMT), \
  38               __VA_ARGS__);                 \
  39       TEST_COMPARE_STRING_MACRO (buf, L_(EXPECTED));            \
  40       TEST_COMPARE (ret, STRLEN (L_(EXPECTED)));            \
  41     }                                   \
  42   while (0)
