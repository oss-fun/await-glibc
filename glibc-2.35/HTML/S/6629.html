  83      version here.  */

  86 # error "Assumed value of MB_LEN_MAX wrong"
  87 #endif
  88   if (__glibc_objsize (__s) != (size_t) -1
  89       && __STDLIB_MB_LEN_MAX > __glibc_objsize (__s))
  90     return __wctomb_chk (__s, __wchar, __glibc_objsize (__s));
