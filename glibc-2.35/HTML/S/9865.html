  57 /* Linux uses a negative return value to indicate syscall errors,
  58    unlike most Unices, which use the condition codes' carry flag.
  59 
  60    Since version 2.1 the return value of a system call might be
  61    negative even if the call succeeded.  E.g., the lseek system call
  62    might return a large offset.  Therefore we must not anymore test
  63    for < 0, but test for a real error by making sure the value in R0
  64    is a real error number.  Linus said he will make sure the no syscall
  65    returns a value in -1 .. -4095 as a valid result so we can safely
  66    test with -4095.  */

  74 #undef  DO_CALL
  75 #define DO_CALL(syscall_name) \
  76   l.addi r11, r0, SYS_ify (syscall_name); \
  77   l.sys 1; \
  78    l.nop
  79 
  80 #undef  PSEUDO
  81 #define PSEUDO(name, syscall_name, args) \
  82   ENTRY (name); \
  83   DO_CALL(syscall_name); \
  84   /* if -4096 < ret < 0 holds, it's an error */ \
  85   l.sfgeui r11, 0xf001; \
  86   l.bf L(pseudo_end); \
  87    l.nop
  88 
  89 #undef  PSEUDO_NOERRNO
  90 #define PSEUDO_NOERRNO(name, syscall_name, args)  \
  91   ENTRY (name);           \
  92   DO_CALL(syscall_name)
  93 
  94 #undef  PSEUDO_END
  95 #define PSEUDO_END(name) \
  96 L(pseudo_end): \
  97   l.j SYSCALL_ERROR_NAME; \

  99   END (name)

 126 #undef INTERNAL_SYSCALL
 127 #define INTERNAL_SYSCALL(name, nr, args...) \
 128     INTERNAL_SYSCALL_NCS (SYS_ify (name), nr, args)
 129 
 130 /* The _NCS variant allows non-constant syscall numbers.  */
 131 #undef INTERNAL_SYSCALL_NCS
 132 #define INTERNAL_SYSCALL_NCS(number, nr, args...) \
 133     ({ unsigned long int __sys_result;              \
 134       {                             \
 135         long int _sc_ret = (long int) number;           \
 136         LOAD_ARGS_##nr (args)                   \
 137         register long int __sc_ret __asm__ ("r11") = _sc_ret;   \
 138         __asm__ __volatile__ ("l.sys 1\n\t"             \
 139                   " l.nop\n\t"              \
 140                   : "+r" (__sc_ret)         \
 141                   : ASM_ARGS_##nr           \
 142                   : ASM_CLOBBERS_##nr           \
 143                     "r12", "r13", "r15", "r17", "r19",  \
 144                     "r21", "r23", "r25", "r27", "r29",  \
 145                     "r31", "memory");           \
 146         __sys_result = __sc_ret;                    \
 147       }                             \
 148       (long int) __sys_result; })
