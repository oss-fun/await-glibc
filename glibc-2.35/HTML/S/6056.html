  24 /* Print out on stderr a line consisting of the test in S, a colon, a space,
  25    a message describing the meaning of the signal number SIG and a newline.
  26    If S is NULL or "", the colon and space are omitted.  */
  27 void
  28 psignal (int sig, const char *s)
  29 {
  30   const char *colon, *desc;
  31 
  32   if (s == NULL || *s == '\0')
  33     s = colon = "";
  34   else
  35     colon = ": ";
  36 
  37   if (sig >= 0 && sig < NSIG && (desc = __sys_siglist[sig]) != NULL)
  38     (void) __fxprintf (NULL, "%s%s%s\n", s, colon, _(desc));
  39   else
  40     {
  41       char *buf;
  42 
  43       if (__asprintf (&buf, _("%s%sUnknown signal %d\n"), s, colon, sig) < 0)
  44     (void) __fxprintf (NULL, "%s%s%s\n", s, colon, _("Unknown signal"));
  45       else
  46     {
  47       (void) __fxprintf (NULL, "%s", buf);
  48 
  49       free (buf);
  50     }
  51     }
  52 }
