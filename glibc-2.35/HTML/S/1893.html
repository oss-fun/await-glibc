  27 static int
  28 do_test (void)
  29 {
  30   /* The test below assumes that pvalloc rounds up the allocation size
  31      to at least 8.  */
  32   TEST_VERIFY (xsysconf (_SC_PAGESIZE) >= 8);
  33 
  34   void *p = pvalloc (5);
  35   TEST_VERIFY_EXIT (p != NULL);
  36 
  37   /* This is valid assuming the page size is at least 8 because
  38      pvalloc rounds up the allocation size to a multiple of the page
  39      size.  Due to bug 25041, this used to trigger a compiler
  40      warning.  */
  41   strcpy (p, "abcdefg");
  42 
  43   asm ("" : : "g" (p) : "memory"); /* Optimization barrier.  */
  44   TEST_VERIFY (malloc_usable_size (p) >= xsysconf (_SC_PAGESIZE));
  45   return 0;
  46 }
