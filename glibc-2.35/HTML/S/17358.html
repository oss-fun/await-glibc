  25 float
  26 __floorf (float x)
  27 {
  28   int flags = riscv_getflags ();
  29   bool nan = isnan (x);
  30   float mag = fabsf (x);
  31 
  32   if (nan)
  33     return x + x;
  34 
  35   if (mag < (1 << __FLT_MANT_DIG__))
  36     {
  37       int32_t i;
  38       float new_x;
  39 
  40       asm volatile ("fcvt.w.s %0, %1, rdn" : "=r" (i) : "f" (x));
  41       asm volatile ("fcvt.s.w %0, %1, rdn" : "=f" (new_x) : "r" (i));
  42 
  43       /* floor(-0) == -0, and in general we'll always have the same
  44      sign as our input.  */
  45       x = copysignf (new_x, x);
  46 
  47       riscv_setflags (flags);
  48     }
  49 
  50   return x;
  51 }
