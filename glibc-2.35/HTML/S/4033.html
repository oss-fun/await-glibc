  28 static int
  29 clock_test (clockid_t cl)
  30 {
  31   struct timespec old_ts;
  32   struct timespec ts;
  33   struct timespec waitit;
  34   int result = 0;
  35   int i;
  36 
  37   memset (&ts, '\0', sizeof ts);
  38 
  39   waitit.tv_sec = 0;
  40   waitit.tv_nsec = 500000000;
  41 
  42   /* Get and print resolution of the clock.  */
  43   if (clock_getres (cl, &ts) == 0)
  44     {
  45       if (ts.tv_nsec < 0 || ts.tv_nsec >= 1000000000)
  46     {
  47       printf ("clock %d: nanosecond value of resolution wrong\n", cl);
  48       result = 1;
  49     }
  50       else
  51     printf ("clock %d: resolution = %jd.%09jd secs\n",
  52         cl, (intmax_t) ts.tv_sec, (intmax_t) ts.tv_nsec);
  53     }
  54   else
  55     {
  56       printf ("clock %d: cannot get resolution\n", cl);
  57       result = 1;
  58     }
  59 
  60   memset (&ts, '\0', sizeof ts);
  61   memset (&old_ts, '\0', sizeof old_ts);
  62 
  63   /* Next get the current time value a few times.  */
  64   for (i = 0; i < 10; ++i)
  65     {
  66       if (clock_gettime (cl, &ts) == 0)
  67     {
  68       if (ts.tv_nsec < 0 || ts.tv_nsec >= 1000000000)
  69         {
  70           printf ("clock %d: nanosecond value of time wrong (try %d)\n",
  71               cl, i);
  72           result = 1;
  73         }
  74       else
  75         {
  76           printf ("clock %d: time = %jd.%09jd secs\n",
  77               cl, (intmax_t) ts.tv_sec, (intmax_t) ts.tv_nsec);
  78 
  79           if (memcmp (&ts, &old_ts, sizeof ts) == 0)
  80         {
  81           printf ("clock %d: time hasn't changed (try %d)\n", cl, i);
  82           result = 1;
  83 
  84           old_ts = ts;
  85         }
  86         }
  87     }
  88       else
  89     {
  90       printf ("clock %d: cannot get time (try %d)\n", cl, i);
  91       result = 1;
  92     }
  93 
  94       /* Wait a bit before the next iteration.  */
  95       nanosleep (&waitit, NULL);
  96     }
  97 
  98   return result;
  99 }
