  55   do {                              \
  56     if ((_FP_FRAC_HIGH_RAW_##fs(X) & _FP_QNANBIT_##fs)      \
  57     && !(_FP_FRAC_HIGH_RAW_##fs(Y) & _FP_QNANBIT_##fs)) \
  58       {                             \
  59     R##_s = Y##_s;                      \
  60     _FP_FRAC_COPY_##wc(R,Y);                \
  61       }                             \
  62     else                            \
  63       {                             \
  64     R##_s = X##_s;                      \
  65     _FP_FRAC_COPY_##wc(R,X);                \
  66       }                             \
  67     R##_c = FP_CLS_NAN;                     \
  68   } while (0)
