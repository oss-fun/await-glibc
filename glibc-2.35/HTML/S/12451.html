  45 // API
  46 //==============================================================
  47 // long double fmal  (long double x, long double y, long double z)
  48 //
  49 // Overview of operation
  50 //==============================================================
  51 // returns x * y + z with one rounding error
  52 
  53 // All the special cases are handled by the fma instruction itself
  54 
  55 // floating-point registers used: 3
  56 // f8,  input x, output
  57 // f9,  input y
  58 // f10, input z
  59 
  60 .section .text
  61 GLOBAL_LIBM_ENTRY(fmal)
  62 
  63 { .mfb
  64       nop.m 999
  65       fma.s0  f8 = f8, f9, f10  // Result = x * y + z
  66       br.ret.sptk    b0
  67 }
  68 ;;
  69 
  70 GLOBAL_LIBM_END(fmal)
