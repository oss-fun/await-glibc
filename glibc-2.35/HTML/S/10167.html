  18 #include <sysdep.h>

  20 #include <sys/asm.h>
  21 
  22 /* Sign-extend the ioctl number, since the kernel wants it as a
  23    sign-extended 32-bit value, but our prototype is that of a long.  */
  24 
  25     .text
  26 ENTRY (__ioctl)
  27     sll a1, a1, 0
  28     li v0, __NR_ioctl
  29     syscall         /* Do the system call.  */
  30     bne a3, zero, L(error)
  31     ret
  32 
  33 L(error):
  34     SETUP_GP64_REG (a0, __ioctl)
  35     PTR_LA t9, __syscall_error
  36     RESTORE_GP64_REG
  37     jr t9
  38 
  39 PSEUDO_END (__ioctl)
  40 
  41 libc_hidden_def (__ioctl)
  42 weak_alias (__ioctl, ioctl)
