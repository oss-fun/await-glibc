  34 int
  35 __aio_error (const struct aiocb *aiocbp)
  36 {
  37   int ret;
  38 
  39   /* Acquire the mutex to make sure all operations for this request are
  40      complete.  */
  41   __pthread_mutex_lock (&__aio_requests_mutex);
  42   ret = aiocbp->__error_code;
  43   __pthread_mutex_unlock (&__aio_requests_mutex);
  44 
  45   return ret;
  46 }
