  28 {
  29     float y,z;
  30     int32_t i,k,hx;
  31 
  32     GET_FLOAT_WORD(hx,x);

  34     k=0;
  35     if (hx < 0x00800000) {          /* x < 2**-126  */
  36         if (__builtin_expect((hx&0x7fffffff)==0, 0))
  37           return -two25/fabsf (x);  /* log(+-0)=-inf  */

  39         return (x-x)/(x-x); /* log(-#) = NaN */

  41         GET_FLOAT_WORD(hx,x);

  44     k += (hx>>23)-127;
