  27 char *
  28 support_format_netent (struct netent *e)
  29 {
  30   if (e == NULL)
  31     {
  32       char *value = support_format_herrno (h_errno);
  33       char *result = xasprintf ("error: %s\n", value);
  34       free (value);
  35       return result;
  36     }
  37 
  38   struct xmemstream mem;
  39   xopen_memstream (&mem);
  40 
  41   if (e->n_name != NULL)
  42     fprintf (mem.out, "name: %s\n", e->n_name);
  43   for (char **ap = e->n_aliases; *ap != NULL; ++ap)
  44     fprintf (mem.out, "alias: %s\n", *ap);
  45   if (e->n_addrtype != AF_INET)
  46     fprintf (mem.out, "addrtype: %d\n", e->n_addrtype);
  47   /* On alpha, e->n_net is an unsigned long.  */
  48   unsigned int n_net = e->n_net;
  49   fprintf (mem.out, "net: 0x%08x\n", n_net);
  50 
  51   xfclose_memstream (&mem);
  52   return mem.buffer;
  53 }
