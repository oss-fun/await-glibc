  19 /*
  20  * ALGORITHM DESCRIPTION:
  21  *
  22  *      SelMask = (|x| >= 0.5) ? 1 : 0;
  23  *      R = SelMask ? sqrt(0.5 - 0.5*|x|) : |x|
  24  *      acos(|x|) = SelMask ? 2*Poly(R) : (Pi/2 - Poly(R))
  25  *      acos(x) = sign(x) ? (Pi - acos(|x|)) : acos(|x|)
  26  *
  27  *
  28  */
  29 
  30 /* Offsets for data table __svml_sacos_data_internal
  31  */
  32 #define SgnBit                          0
  33 #define OneHalf                         16
  34 #define SmallNorm                       32
  35 #define MOne                            48
  36 #define Two                             64
  37 #define sqrt_coeff                      80
  38 #define poly_coeff                      112
  39 #define Pi2H                            192
  40 #define PiH                             208
  41 
  42 #include <sysdep.h>

  44         .text
  45     .section .text.sse4,"ax",@progbits
  46 ENTRY(_ZGVbN4v_acosf_sse4)
  47         subq      $72, %rsp
  48         cfi_def_cfa_offset(80)
  49 
  50 /* X<X^2 iff X<0 */
  51         movaps    %xmm0, %xmm14
  52 
  53 /*
  54  * 2*sqrt(X) ~ Sh - Sl  (to 24+ bits)
  55  * SQ ~ 2*sqrt(X)
  56  */
  57         movups    __svml_sacos_data_internal(%rip), %xmm3
  58         movups    OneHalf+__svml_sacos_data_internal(%rip), %xmm5
  59 
  60 /* x = -|arg| */
  61         movaps    %xmm3, %xmm4
  62         orps      %xmm0, %xmm4
  63 
  64 /* Y = 0.5 + 0.5*(-x) */
  65         movaps    %xmm5, %xmm6
  66         mulps     %xmm4, %xmm6
  67 
  68 /* x^2 */
  69         movaps    %xmm4, %xmm13
  70         mulps     %xmm4, %xmm13
  71         addps     %xmm6, %xmm5
  72 
  73 /* SQ ~ 2*sqrt(Y) */
  74         rsqrtps   %xmm5, %xmm8
  75         minps     %xmm5, %xmm13
  76         movaps    %xmm5, %xmm2
  77         movaps    %xmm13, %xmm1
  78         cmpltps   SmallNorm+__svml_sacos_data_internal(%rip), %xmm2
  79         cmpnltps  %xmm5, %xmm1
  80         cmpltps   %xmm13, %xmm14
  81         addps     %xmm5, %xmm5
  82         andnps    %xmm8, %xmm2
  83         movaps    %xmm13, %xmm11
  84         movaps    %xmm2, %xmm9
  85         movaps    %xmm1, %xmm6
  86         mulps     %xmm2, %xmm9
  87         andnps    %xmm4, %xmm6
  88         mulps     %xmm5, %xmm2
  89         mulps     %xmm13, %xmm11
  90         mulps     %xmm9, %xmm5
  91         movups    sqrt_coeff+__svml_sacos_data_internal(%rip), %xmm10
  92         andps     %xmm0, %xmm3
  93 
  94 /* polynomial */
  95         movups    poly_coeff+__svml_sacos_data_internal(%rip), %xmm12
  96         movaps    %xmm1, %xmm15
  97         mulps     %xmm13, %xmm12
  98         subps     Two+__svml_sacos_data_internal(%rip), %xmm5
  99         mulps     %xmm5, %xmm10
 100         addps     poly_coeff+16+__svml_sacos_data_internal(%rip), %xmm12
 101         mulps     %xmm2, %xmm5
 102         mulps     %xmm11, %xmm12
 103         addps     sqrt_coeff+16+__svml_sacos_data_internal(%rip), %xmm10
 104         mulps     %xmm5, %xmm10
 105         movups    poly_coeff+32+__svml_sacos_data_internal(%rip), %xmm5
 106         subps     %xmm10, %xmm2
 107         mulps     %xmm13, %xmm5
 108         movups    MOne+__svml_sacos_data_internal(%rip), %xmm7
 109         andps     %xmm1, %xmm2
 110         cmpnleps  %xmm4, %xmm7
 111         addps     poly_coeff+48+__svml_sacos_data_internal(%rip), %xmm5
 112         movmskps  %xmm7, %edx
 113         orps      %xmm2, %xmm6
 114         addps     %xmm12, %xmm5
 115         mulps     %xmm13, %xmm5
 116         pxor      %xmm3, %xmm6
 117         movups    PiH+__svml_sacos_data_internal(%rip), %xmm7
 118         andps     %xmm1, %xmm7
 119         addps     poly_coeff+64+__svml_sacos_data_internal(%rip), %xmm5
 120         mulps     %xmm13, %xmm5
 121         andps     %xmm14, %xmm7
 122         mulps     %xmm6, %xmm5
 123         andnps    Pi2H+__svml_sacos_data_internal(%rip), %xmm15
 124         addps     %xmm5, %xmm6
 125         addps     %xmm15, %xmm7
 126         addps     %xmm6, %xmm7
 127         testl     %edx, %edx
 128 
 129 /* Go to special inputs processing branch */
 130         jne       L(SPECIAL_VALUES_BRANCH)
 131                                 # LOE rbx rbp r12 r13 r14 r15 edx xmm0 xmm7
 132 
 133 /* Restore registers
 134  * and exit the function
 135  */
 136 
 137 L(EXIT):
 138         movaps    %xmm7, %xmm0
 139         addq      $72, %rsp
 140         cfi_def_cfa_offset(8)
 141         ret
 142         cfi_def_cfa_offset(80)
 143 
 144 /* Branch to process
 145  * special inputs
 146  */
 147 
 148 L(SPECIAL_VALUES_BRANCH):
 149         movups    %xmm0, 32(%rsp)
 150         movups    %xmm7, 48(%rsp)
 151                                 # LOE rbx rbp r12 r13 r14 r15 edx
 152 
 153         xorl      %eax, %eax
 154         movq      %r12, 16(%rsp)
 155         cfi_offset(12, -64)
 156         movl      %eax, %r12d
 157         movq      %r13, 8(%rsp)
 158         cfi_offset(13, -72)
 159         movl      %edx, %r13d
 160         movq      %r14, (%rsp)
 161         cfi_offset(14, -80)
 162                                 # LOE rbx rbp r15 r12d r13d
 163 
 164 /* Range mask
 165  * bits check
 166  */
 167 
 168 L(RANGEMASK_CHECK):
 169         btl       %r12d, %r13d
 170 
 171 /* Call scalar math function */
 172         jc        L(SCALAR_MATH_CALL)
 173                                 # LOE rbx rbp r15 r12d r13d
 174 
 175 /* Special inputs
 176  * processing loop
 177  */
 178 
 179 L(SPECIAL_VALUES_LOOP):
 180         incl      %r12d
 181         cmpl      $4, %r12d
 182 
 183 /* Check bits in range mask */
 184         jl        L(RANGEMASK_CHECK)
 185                                 # LOE rbx rbp r15 r12d r13d
 186 
 187         movq      16(%rsp), %r12
 188         cfi_restore(12)
 189         movq      8(%rsp), %r13
 190         cfi_restore(13)
 191         movq      (%rsp), %r14
 192         cfi_restore(14)
 193         movups    48(%rsp), %xmm7
 194 
 195 /* Go to exit */
 196         jmp       L(EXIT)
 197         cfi_offset(12, -64)
 198         cfi_offset(13, -72)
 199         cfi_offset(14, -80)
 200                                 # LOE rbx rbp r12 r13 r14 r15 xmm7
 201 
 202 /* Scalar math fucntion call
 203  * to process special input
 204  */
 205 
 206 L(SCALAR_MATH_CALL):
 207         movl      %r12d, %r14d
 208         movss     32(%rsp,%r14,4), %xmm0
 209         call      acosf@PLT
 210                                 # LOE rbx rbp r14 r15 r12d r13d xmm0
 211 
 212         movss     %xmm0, 48(%rsp,%r14,4)
 213 
 214 /* Process special inputs in loop */
 215         jmp       L(SPECIAL_VALUES_LOOP)
 216                                 # LOE rbx rbp r15 r12d r13d
 217 END(_ZGVbN4v_acosf_sse4)
