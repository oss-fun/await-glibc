  19 /*
  20  * ALGORITHM DESCRIPTION:
  21  *
  22  *   Compute asinh(x) as log(x + sqrt(x*x + 1))
  23  *   using RSQRT instructions for starting the
  24  *   square root approximation, and small table lookups for log
  25  *   that map to AVX-512 permute instructions
  26  *
  27  *   Special cases:
  28  *
  29  *   asinh(NaN) = quiet NaN, and raise invalid exception
  30  *   asinh(INF) = that INF
  31  *   asinh(0)   = that 0
  32  *
  33  */
  34

  55 #define poly_coeff8                     1280
  56 #define poly_coeff7                     1344
  57 #define poly_coeff6                     1408

  70 ENTRY(_ZGVeN8v_asinh_skx)
  71         pushq     %rbp
  72         cfi_def_cfa_offset(16)
  73         movq      %rsp, %rbp
  74         cfi_def_cfa(6, 16)
  75         cfi_offset(6, -16)
  76         andq      $-64, %rsp
  77         subq      $192, %rsp
  78         vmovaps   %zmm0, %zmm3
  79 
  80 /* x^2 */
  81         vmulpd    {rn-sae}, %zmm3, %zmm3, %zmm14
  82         vmovups   One+__svml_dasinh_data_internal_avx512(%rip), %zmm9
  83 
  84 /* polynomial computation for small inputs */
  85         vmovups   ca2+__svml_dasinh_data_internal_avx512(%rip), %zmm10
  86         vmovups   ca1+__svml_dasinh_data_internal_avx512(%rip), %zmm11
  87 
  88 /* not a very small input ? */
  89         vmovups   SmallThreshold+__svml_dasinh_data_internal_avx512(%rip), %zmm0
  90 
  91 /* A=max(x^2, 1); */
  92         vmaxpd    {sae}, %zmm14, %zmm9, %zmm4
  93 
  94 /* B=min(x^2, 1); */
  95         vminpd    {sae}, %zmm14, %zmm9, %zmm5
  96         vfmadd231pd {rn-sae}, %zmm14, %zmm10, %zmm11
  97 
  98 /* 1+x^2 */
  99         vaddpd    {rn-sae}, %zmm9, %zmm14, %zmm8
 100 
 101 /* |input| */
 102         vandpd    AbsMask+__svml_dasinh_data_internal_avx512(%rip), %zmm3, %zmm1
 103         vrsqrt14pd %zmm8, %zmm6
 104         vcmppd    $21, {sae}, %zmm0, %zmm1, %k2
 105 
 106 /* B_high */
 107         vsubpd    {rn-sae}, %zmm4, %zmm8, %zmm7
 108 
 109 /* sign bit */
 110         vxorpd    %zmm3, %zmm1, %zmm2
 111         vmulpd    {rn-sae}, %zmm14, %zmm11, %zmm4
 112 
 113 /* B_low */
 114         vsubpd    {rn-sae}, %zmm7, %zmm5, %zmm13
 115         vmovups   c2s+__svml_dasinh_data_internal_avx512(%rip), %zmm5
 116         vmovups   c1s+__svml_dasinh_data_internal_avx512(%rip), %zmm7
 117 
 118 /* polynomial computation for small inputs */
 119         vfmadd213pd {rn-sae}, %zmm1, %zmm1, %zmm4
 120 
 121 /* (x^2)_low */
 122         vmovaps   %zmm3, %zmm15
 123         vfmsub213pd {rn-sae}, %zmm14, %zmm3, %zmm15
 124 
 125 /* Sh ~sqrt(1+x^2) */
 126         vmulpd    {rn-sae}, %zmm6, %zmm8, %zmm14
 127 
 128 /* Yl = (x^2)_low + B_low */
 129         vaddpd    {rn-sae}, %zmm15, %zmm13, %zmm13
 130 
 131 /* very large inputs ? */
 132         vmovups   Threshold+__svml_dasinh_data_internal_avx512(%rip), %zmm15
 133 
 134 /* (Yh*R0)_low */
 135         vfmsub213pd {rn-sae}, %zmm14, %zmm6, %zmm8
 136         vcmppd    $21, {sae}, %zmm15, %zmm1, %k1
 137 
 138 /* Sl = (Yh*R0)_low+(R0*Yl) */
 139         vfmadd213pd {rn-sae}, %zmm8, %zmm6, %zmm13
 140         vmovups   LargeThreshold+__svml_dasinh_data_internal_avx512(%rip), %zmm8
 141 
 142 /* rel. error term: Eh=1-Sh*R0 */
 143         vmovaps   %zmm9, %zmm12
 144         vfnmadd231pd {rn-sae}, %zmm14, %zmm6, %zmm12
 145         vcmppd    $22, {sae}, %zmm8, %zmm1, %k0
 146 
 147 /* rel. error term: Eh=(1-Sh*R0)-Sl*R0 */
 148         vfnmadd231pd {rn-sae}, %zmm13, %zmm6, %zmm12
 149 
 150 /*
 151  * sqrt(1+x^2) ~ Sh + Sl + Sh*Eh*poly_s
 152  * poly_s = c1+c2*Eh+c3*Eh^2
 153  */
 154         vmovups   c4s+__svml_dasinh_data_internal_avx512(%rip), %zmm6
 155         vmovups   c3s+__svml_dasinh_data_internal_avx512(%rip), %zmm8
 156 
 157 /* Sh*Eh */
 158         vmulpd    {rn-sae}, %zmm12, %zmm14, %zmm11
 159         vfmadd231pd {rn-sae}, %zmm12, %zmm6, %zmm8
 160 
 161 /* Sh+x */
 162         vaddpd    {rn-sae}, %zmm1, %zmm14, %zmm6
 163         kmovw     %k0, %edx
 164         vfmadd213pd {rn-sae}, %zmm5, %zmm12, %zmm8
 165         vfmadd213pd {rn-sae}, %zmm7, %zmm12, %zmm8
 166 
 167 /* Xh */
 168         vsubpd    {rn-sae}, %zmm14, %zmm6, %zmm12
 169 
 170 /* Sl + Sh*Eh*poly_s */
 171         vfmadd213pd {rn-sae}, %zmm13, %zmm8, %zmm11
 172 
 173 /* fixup for very large inputs */
 174         vmovups   OneEighth+__svml_dasinh_data_internal_avx512(%rip), %zmm8
 175 
 176 /* Xl */
 177         vsubpd    {rn-sae}, %zmm12, %zmm1, %zmm12
 178 
 179 /* Xin0+Sl+Sh*Eh*poly_s ~ x+sqrt(1+x^2) */
 180         vaddpd    {rn-sae}, %zmm11, %zmm6, %zmm10
 181 
 182 /* Sl_high */
 183         vsubpd    {rn-sae}, %zmm6, %zmm10, %zmm5
 184         vmulpd    {rn-sae}, %zmm8, %zmm1, %zmm10{%k1}
 185 
 186 /* Table lookups */
 187         vmovups   __svml_dasinh_data_internal_avx512(%rip), %zmm6
 188 
 189 /* Sl_l */
 190         vsubpd    {rn-sae}, %zmm5, %zmm11, %zmm7
 191         vrcp14pd  %zmm10, %zmm13
 192 
 193 /* Xin_low */
 194         vaddpd    {rn-sae}, %zmm12, %zmm7, %zmm14
 195         vmovups   Log_tbl_L+__svml_dasinh_data_internal_avx512(%rip), %zmm7
 196         vmovups   poly_coeff6+__svml_dasinh_data_internal_avx512(%rip), %zmm12
 197 
 198 /* round reciprocal to 1+4b mantissas */
 199         vpaddq    AddB5+__svml_dasinh_data_internal_avx512(%rip), %zmm13, %zmm11
 200 
 201 /* fixup for very large inputs */
 202         vxorpd    %zmm14, %zmm14, %zmm14{%k1}
 203         vmovups   poly_coeff5+__svml_dasinh_data_internal_avx512(%rip), %zmm13
 204         vandpd    RcpBitMask+__svml_dasinh_data_internal_avx512(%rip), %zmm11, %zmm15
 205         vmovups   poly_coeff7+__svml_dasinh_data_internal_avx512(%rip), %zmm11
 206 
 207 /* Prepare table index */
 208         vpsrlq    $48, %zmm15, %zmm5
 209 
 210 /* reduced argument for log(): (Rcp*Xin-1)+Rcp*Xin_low */
 211         vfmsub231pd {rn-sae}, %zmm15, %zmm10, %zmm9
 212 
 213 /* exponents */
 214         vgetexppd {sae}, %zmm15, %zmm8
 215         vmovups   Four+__svml_dasinh_data_internal_avx512(%rip), %zmm10
 216         vpermt2pd Log_tbl_H+64+__svml_dasinh_data_internal_avx512(%rip), %zmm5, %zmm6
 217         vpermt2pd Log_tbl_L+64+__svml_dasinh_data_internal_avx512(%rip), %zmm5, %zmm7
 218         vsubpd    {rn-sae}, %zmm10, %zmm8, %zmm8{%k1}
 219         vfmadd231pd {rn-sae}, %zmm15, %zmm14, %zmm9
 220 
 221 /* polynomials */
 222         vmovups   poly_coeff9+__svml_dasinh_data_internal_avx512(%rip), %zmm10
 223         vmovups   poly_coeff8+__svml_dasinh_data_internal_avx512(%rip), %zmm5
 224         vmovups   poly_coeff4+__svml_dasinh_data_internal_avx512(%rip), %zmm14
 225 
 226 /* -K*L2H + Th */
 227         vmovups   L2H+__svml_dasinh_data_internal_avx512(%rip), %zmm15
 228         vfmadd231pd {rn-sae}, %zmm9, %zmm10, %zmm5
 229 
 230 /* -K*L2L + Tl */
 231         vmovups   L2L+__svml_dasinh_data_internal_avx512(%rip), %zmm10
 232         vfnmadd231pd {rn-sae}, %zmm8, %zmm15, %zmm6
 233         vfmadd213pd {rn-sae}, %zmm11, %zmm9, %zmm5
 234         vfnmadd213pd {rn-sae}, %zmm7, %zmm10, %zmm8
 235         vmovups   poly_coeff3+__svml_dasinh_data_internal_avx512(%rip), %zmm7
 236         vmovups   poly_coeff1+__svml_dasinh_data_internal_avx512(%rip), %zmm10
 237 
 238 /* R^2 */
 239         vmulpd    {rn-sae}, %zmm9, %zmm9, %zmm11
 240         vfmadd213pd {rn-sae}, %zmm12, %zmm9, %zmm5
 241         vfmadd213pd {rn-sae}, %zmm13, %zmm9, %zmm5
 242         vfmadd213pd {rn-sae}, %zmm14, %zmm9, %zmm5
 243         vfmadd213pd {rn-sae}, %zmm7, %zmm9, %zmm5
 244         vmovups   poly_coeff2+__svml_dasinh_data_internal_avx512(%rip), %zmm7
 245         vfmadd213pd {rn-sae}, %zmm7, %zmm9, %zmm5
 246         vfmadd213pd {rn-sae}, %zmm10, %zmm9, %zmm5
 247 
 248 /* Tl + R^2*Poly */
 249         vfmadd213pd {rn-sae}, %zmm8, %zmm11, %zmm5
 250 
 251 /* R+Tl + R^2*Poly */
 252         vaddpd    {rn-sae}, %zmm9, %zmm5, %zmm9
 253         vaddpd    {rn-sae}, %zmm9, %zmm6, %zmm4{%k2}
 254         vxorpd    %zmm2, %zmm4, %zmm0
 255         testl     %edx, %edx
 256 
 257 /* Go to special inputs processing branch */
 258         jne       L(SPECIAL_VALUES_BRANCH)
 259                                 # LOE rbx r12 r13 r14 r15 edx zmm0 zmm3
 260 
 261 /* Restore registers
 262  * and exit the function
 263  */
 264 
 265 L(EXIT):
 266         movq      %rbp, %rsp
 267         popq      %rbp
 268         cfi_def_cfa(7, 8)
 269         cfi_restore(6)
 270         ret
 271         cfi_def_cfa(6, 16)
 272         cfi_offset(6, -16)
 273

 275  * special inputs
 276  */
 277

 279         vmovups   %zmm3, 64(%rsp)
 280         vmovups   %zmm0, 128(%rsp)
 281                                 # LOE rbx r12 r13 r14 r15 edx zmm0
 282

 284                                 # LOE rbx r12 r13 r14 r15 eax edx

 287         movq      %r12, 16(%rsp)

 290         movl      %eax, %r12d
 291         movq      %r13, 8(%rsp)
 292         /*  DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -184; DW_OP_plus)  */

 295         movq      %r14, (%rsp)
 296         /*  DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -192; DW_OP_plus)  */

 298                                 # LOE rbx r15 r12d r13d

 301  * bits check

 305         btl       %r12d, %r13d
 306 
 307 /* Call scalar math function */
 308         jc        L(SCALAR_MATH_CALL)
