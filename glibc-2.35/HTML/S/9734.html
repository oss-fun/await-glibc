  32 int
  33 __sched_getaffinity_new (pid_t pid, size_t cpusetsize, cpu_set_t *cpuset)
  34 {
  35   int res = INLINE_SYSCALL (sched_getaffinity, 3, pid,
  36                 MIN (INT_MAX, cpusetsize), cpuset);
  37   if (res != -1)
  38     {
  39       /* Clean the rest of the memory the kernel didn't do.  */
  40       memset ((char *) cpuset + res, '\0', cpusetsize - res);
  41 
  42       res = 0;
  43     }
  44   return res;
  45 }
