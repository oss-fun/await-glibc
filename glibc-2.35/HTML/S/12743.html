  46 // API
  47 //==============================================================
  48 // double erfc(double)
  49 //
  50 // Overview of operation
  51 //==============================================================
  52 // 1. 0 <= x <= 28.0
  53 //
  54 //    erfc(x)  = P14(z) * exp( -x^2 ), z = x - x(i).
  55 //
  56 //    Comment:
  57 //
  58 //    Let x(i) = -1.0 + 2^(i/4),i=0,...19. So we have 20 unequal
  59 //    argument intervals [x(i),x(i+1)] with length ratio q = 2^(1/4).
  60 //    Values x(i) we have in the table erfc_xb_table.
  61 //
  62 //    Let x(i)<= x < x(i+1).
  63 //    We can find i as exponent of number (x + 1)^4.
  64 //
  65 //    Let P14(z) - polynomial approximation of degree 14 for function
  66 //    erfc(z+x(i)) * exp( (z+x(i))^2) and 0 <= z <= x(i+1)-x(i).
  67 //    Polynomial coefficients we have in the table erfc_p_table.
  68 //
  69 //    So we can find result for erfc(x) as above.
  70 //    Algorithm description for exp function see below.
  71 //
  72 // 2. -6 <= x < 0
  73 //
  74 //      erfc(x)  = 2.0 - erfc(-x)
  75 //
  76 // 3. x > 28.0
  77 //    erfc(x)  ~=~ 0.0
  78 //
  79 // 4. x < -6.0
  80 //    erfc(x)  ~=~ 2.0
  81 
  82 // Special values
  83 //==============================================================
  84 // erfc(+0)    = 1.0
  85 // erfc(-0)    = 1.0
  86 
  87 // erfc(+qnan) = +qnan
  88 // erfc(-qnan) = -qnan
  89 // erfc(+snan) = +qnan
  90 // erfc(-snan) = -qnan
  91 
  92 // erfc(-inf)  = 2.0
  93 // erfc(+inf)  = +0
  94 
  95 //==============================================================
  96 // Take double exp(double) from libm_64.
  97 //
  98 // Overview of operation
  99 //==============================================================
 100 // Take the input x. w is "how many log2/128 in x?"
 101 //  w = x * 128/log2
 102 //  n = int(w)
 103 //  x = n log2/128 + r + delta
 104 
 105 //  n = 128M + index_1 + 2^4 index_2
 106 //  x = M log2 + (log2/128) index_1 + (log2/8) index_2 + r + delta
 107 
 108 //  exp(x) = 2^M  2^(index_1/128)  2^(index_2/8) exp(r) exp(delta)
 109 //       Construct 2^M
 110 //       Get 2^(index_1/128) from table_1;
 111 //       Get 2^(index_2/8)   from table_2;
 112 //       Calculate exp(r) by series
 113 //          r = x - n (log2/128)_high
 114 //          delta = - n (log2/128)_low
 115 //       Calculate exp(delta) as 1 + delta
 116 //==============================================================
 117 //  Comment for exp for erfc:
 118 //
 119 //  We use quad precision for calculate input argument -x^2 and add
 120 //  result low bits to value delta in exp.
