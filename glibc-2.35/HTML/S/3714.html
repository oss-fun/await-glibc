  40 /* wrapper y1 */
  41 double
  42 __y1 (double x)
  43 {
  44   if (__builtin_expect (islessequal (x, 0.0) || isgreater (x, X_TLOSS), 0)
  45       && _LIB_VERSION != _IEEE_)
  46     {
  47       if (x < 0.0)
  48     {
  49       /* d = zero/(x-x) */
  50       __feraiseexcept (FE_INVALID);
  51       return __kernel_standard (x, x, 11);
  52     }
  53       else if (x == 0.0)
  54     {
  55       /* d = -one/(x-x) */
  56       __feraiseexcept (FE_DIVBYZERO);
  57       return __kernel_standard (x, x, 10);
  58     }
  59       else if (_LIB_VERSION != _POSIX_)
  60     /* y1(x>X_TLOSS) */
  61     return __kernel_standard (x, x, 37);
  62     }
  63 
  64   return __ieee754_y1 (x);
  65 }
