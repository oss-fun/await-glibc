 136 /* Function used to test for correct thread agent pointer.  */
 137 static inline bool
 138 ta_ok (const td_thragent_t *ta)
 139 {
 140   list_t *runp;
 141 
 142   list_for_each (runp, &__td_agent_list)
 143     if (list_entry (runp, td_thragent_t, list) == ta)
 144       return true;
 145 
 146   return false;
 147 }
