  22 /* Send COUNT bytes from file associated with IN_FD starting at OFFSET to
  23    descriptor OUT_FD.  */
  24 ssize_t
  25 sendfile (int out_fd, int in_fd, off_t *offset, size_t count)
  26 {
  27   if (offset == NULL || sizeof (off_t) == sizeof (off64_t))
  28     return __sendfile64 (out_fd, in_fd, (off64_t *) offset, count);
  29   else
  30     {
  31       off64_t ofs = *offset;
  32       ssize_t ret = __sendfile64 (out_fd, in_fd, &ofs, count);
  33       *offset = ofs;
  34       return ret;
  35     }
  36 }
