  19 /*
  20  * ALGORITHM DESCRIPTION:
  21  *
  22  *   exp10(x)  = 2^x/log10(2) = 2^n * (1 + T[j]) * (1 + P(y))
  23  *   where
  24  *        x = m*log10(2)/K + y,  y in [-log10(2)/K..log10(2)/K]
  25  *        m = n*K + j,           m,n,j - signed integer, j in [-K/2..K/2]
  26  *
  27  *        values of 2^j/K are tabulated
  28  *
  29  *        P(y) is a minimax polynomial approximation of exp10(x)-1
  30  *        on small interval [-log10(2)/K..log10(2)/K]
  31  *
  32  *  Special cases:
  33  *
  34  *   exp10(NaN)  = NaN
  35  *   exp10(+INF) = +INF
  36  *   exp10(-INF) = 0
  37  *   exp10(x)    = 1 for subnormals
  38  *   For IEEE float
  39  *     if x >  38.5318412780761720 then exp10f(x) overflow
  40  *     if x < -45.4555282592773440 then exp10f(x) underflow
  41  *
  42  */
  43

  62 ENTRY(_ZGVdN8v_exp10f_avx2)
  63         pushq     %rbp
  64         cfi_def_cfa_offset(16)
  65         movq      %rsp, %rbp
  66         cfi_def_cfa(6, 16)
  67         cfi_offset(6, -16)
  68         andq      $-32, %rsp
  69         subq      $96, %rsp
  70         lea       __svml_sexp10_data_internal(%rip), %rax
  71         vmovups   _sShifter+__svml_sexp10_data_internal(%rip), %ymm4
  72 
  73 /*  Load arument  */
  74         vmovups   _sLg2_10+__svml_sexp10_data_internal(%rip), %ymm1
  75         vmovups   _iIndexMask+__svml_sexp10_data_internal(%rip), %ymm2
  76         vmovaps   %ymm0, %ymm3
  77         vfmadd213ps %ymm4, %ymm3, %ymm1
  78 
  79 /*  Index and lookup  */
  80         vandps    %ymm2, %ymm1, %ymm7
  81 
  82 /* iIndex *= sizeof(S); */
  83         vpslld    $2, %ymm7, %ymm10
  84         vsubps    %ymm4, %ymm1, %ymm0
  85 
  86 /* Check for overflow\underflow  */
  87         vandps    _iAbsMask+__svml_sexp10_data_internal(%rip), %ymm3, %ymm5
  88         vpcmpgtd  _iDomainRange+__svml_sexp10_data_internal(%rip), %ymm5, %ymm6
  89         vmovmskps %ymm6, %edx
  90         vmovd     %xmm10, %ecx
  91         vextractf128 $1, %ymm10, %xmm6
  92         vpextrd   $1, %xmm10, %esi
  93         vpextrd   $2, %xmm10, %edi
  94         vpextrd   $3, %xmm10, %r8d
  95         movslq    %ecx, %rcx
  96         movslq    %esi, %rsi
  97         movslq    %edi, %rdi
  98         movslq    %r8d, %r8
  99         vmovd     (%rax,%rcx), %xmm8
 100         vmovd     (%rax,%rsi), %xmm9
 101         vmovd     (%rax,%rdi), %xmm11
 102         vmovd     (%rax,%r8), %xmm12
 103         vpunpckldq %xmm9, %xmm8, %xmm13
 104         vpunpckldq %xmm12, %xmm11, %xmm14
 105         vpunpcklqdq %xmm14, %xmm13, %xmm15
 106 
 107 /*  R  */
 108         vmovups   _sInvLg2_10hi+__svml_sexp10_data_internal(%rip), %ymm13
 109         vmovd     %xmm6, %r9d
 110         vfnmadd213ps %ymm3, %ymm0, %ymm13
 111         vpextrd   $1, %xmm6, %r10d
 112         movslq    %r9d, %r9
 113         movslq    %r10d, %r10
 114         vfnmadd132ps _sInvLg2_10lo+__svml_sexp10_data_internal(%rip), %ymm13, %ymm0
 115         vmovd     (%rax,%r9), %xmm4
 116         vmovd     (%rax,%r10), %xmm5
 117         vpunpckldq %xmm5, %xmm4, %xmm9
 118 
 119 /*
 120  *  Polynomial
 121  * exp10 = 2^N*(Tj+Tj*poly)
 122  * poly(sN) = {1+later} a0+a1*sR
 123  */
 124         vmovups   _sPC2+__svml_sexp10_data_internal(%rip), %ymm4
 125         vfmadd213ps _sPC1+__svml_sexp10_data_internal(%rip), %ymm0, %ymm4
 126         vpextrd   $2, %xmm6, %r11d
 127         vpextrd   $3, %xmm6, %ecx
 128         movslq    %r11d, %r11
 129         movslq    %ecx, %rcx
 130         vfmadd213ps _sPC0+__svml_sexp10_data_internal(%rip), %ymm0, %ymm4
 131         vmovd     (%rax,%r11), %xmm7
 132         vmovd     (%rax,%rcx), %xmm8
 133         vpunpckldq %xmm8, %xmm7, %xmm11
 134 
 135 /* remove index bits */
 136         vpandn    %ymm1, %ymm2, %ymm0
 137         vpunpcklqdq %xmm11, %xmm9, %xmm12
 138 
 139 /*  2^N  */
 140         vpslld    $18, %ymm0, %ymm1
 141         vinsertf128 $1, %xmm12, %ymm15, %ymm14
 142 
 143 /* Tj_l+Tj_h*poly */
 144         vfmadd213ps %ymm14, %ymm14, %ymm4
 145 
 146 /* quick mul 2^N */
 147         vpaddd    %ymm1, %ymm4, %ymm0
 148 
 149 /*  Finish   */
 150         testl     %edx, %edx
 151 
 152 /* Go to special inputs processing branch */
 153         jne       L(SPECIAL_VALUES_BRANCH)
 154                                 # LOE rbx r12 r13 r14 r15 edx ymm0 ymm3
 155 
 156 /* Restore registers
 157  * and exit the function
 158  */
 159 
 160 L(EXIT):
 161         movq      %rbp, %rsp
 162         popq      %rbp
 163         cfi_def_cfa(7, 8)
 164         cfi_restore(6)
 165         ret
 166         cfi_def_cfa(6, 16)
 167         cfi_offset(6, -16)
 168 
 169 /* Branch to process
 170  * special inputs
 171  */
 172 
 173 L(SPECIAL_VALUES_BRANCH):
 174         vmovups   %ymm3, 32(%rsp)
 175         vmovups   %ymm0, 64(%rsp)
 176                                 # LOE rbx r12 r13 r14 r15 edx ymm0
 177 
 178         xorl      %eax, %eax
 179                                 # LOE rbx r12 r13 r14 r15 eax edx

 185         movl      %eax, %r12d
 186         movq      %r13, 8(%rsp)
 187         /*  DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -88; DW_OP_plus)  */
 188         .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0xa8, 0xff, 0xff, 0xff, 0x22
 189         movl      %edx, %r13d
 190         movq      %r14, (%rsp)
 191         /*  DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -96; DW_OP_plus)  */
 192         .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0xa0, 0xff, 0xff, 0xff, 0x22
 193                                 # LOE rbx r15 r12d r13d
 194 
 195 /* Range mask
 196  * bits check
 197  */
 198 
 199 L(RANGEMASK_CHECK):
 200         btl       %r12d, %r13d
 201 
 202 /* Call scalar math function */
 203         jc        L(SCALAR_MATH_CALL)
 204                                 # LOE rbx r15 r12d r13d
 205 
 206 /* Special inputs
 207  * processing loop
 208  */

 213

 219         cfi_restore(12)
 220         movq      8(%rsp), %r13
 221         cfi_restore(13)
 222         movq      (%rsp), %r14
 223         cfi_restore(14)

 225

 228         /*  DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -80; DW_OP_plus)  */
 229         .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0xb0, 0xff, 0xff, 0xff, 0x22
 230         /*  DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -88; DW_OP_plus)  */

 232         /*  DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -96; DW_OP_plus)  */

 234                                 # LOE rbx r12 r13 r14 r15 ymm0
