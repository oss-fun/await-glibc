  37 #define MCOUNT asm(\
  38     ".globl _mcount;\n\t" \
  39     ".align 2;\n\t" \
  40     ".set push;\n\t" \
  41     ".set nomips16;\n\t" \
  42     ".type _mcount,@function;\n\t" \
  43     ".ent _mcount\n\t" \
  44         "_mcount:\n\t" \
  45         ".frame $sp,44,$31\n\t" \
  46         ".set noreorder;\n\t" \
  47         ".set noat;\n\t" \
  48         CPLOAD \
  49     "subu $29,$29,48;\n\t" \
  50     CPRESTORE \

  52         "sw $5,28($29);\n\t" \
  53         "sw $6,32($29);\n\t" \

  55         "sw $2,40($29);\n\t" \
