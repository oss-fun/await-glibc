 378 #define MIN_NEEDED_INPUT    MIN_NEEDED_FROM

 382 #define BODY \
 383   {                                       \
 384     uint32_t ch = *inptr;                             \
 385     int increment = 1;                                \
 386                                           \
 387     if (__builtin_expect (ch >= 0xc1, 0) && ch <= 0xcf)               \
 388       {                                       \
 389     /* Composed character.  First test whether the next byte          \
 390        is also available.  */                         \
 391     uint32_t ch2;                                 \
 392                                           \
 393     if (__glibc_unlikely (inptr + 1 >= inend))                \
 394       {                                   \
 395         /* The second character is not available.  */             \
 396         result = __GCONV_INCOMPLETE_INPUT;                    \
 397         break;                                \
 398       }                                   \
 399                                           \
 400     ch2 = inptr[1];                               \
 401                                           \
 402     if (__builtin_expect (ch2 < 0x20, 0)                      \
 403         || __builtin_expect (ch2 >= 0x80, 0))                 \
 404       {                                   \
 405         /* This is illegal.  */                       \
 406         STANDARD_FROM_LOOP_ERR_HANDLER (1);                   \
 407       }                                   \
 408                                           \
 409     ch = to_ucs4_comb[ch - 0xc1][ch2 - 0x20];                 \
 410                                           \
 411     increment = 2;                                \
 412       }                                       \
 413     else                                      \
 414       ch = to_ucs4[ch];                               \
 415                                           \
 416     if (__builtin_expect (ch == 0, 0) && *inptr != '\0')              \
 417       {                                       \
 418     /* This is an illegal character.  */                      \
 419     STANDARD_FROM_LOOP_ERR_HANDLER (increment);               \
 420       }                                       \
 421     else                                      \
 422       {                                       \
 423     put32 (outptr, ch);                           \
 424     outptr += 4;                                  \
 425       }                                       \
 426     inptr += increment;                               \
