  30 void
  31 __freelocale (locale_t dataset)
  32 {
  33   int cnt;
  34 
  35   /* This static object is returned for newlocale (LC_ALL_MASK, "C").  */
  36   if (dataset == _nl_C_locobj_ptr)
  37     return;
  38 
  39   /* We modify global data (the usage counts).  */
  40   __libc_rwlock_wrlock (__libc_setlocale_lock);
  41 
  42   for (cnt = 0; cnt < __LC_LAST; ++cnt)
  43     if (cnt != LC_ALL && dataset->__locales[cnt]->usage_count != UNDELETABLE)
  44       /* We can remove the data.  */
  45       _nl_remove_locale (cnt, dataset->__locales[cnt]);
  46 
  47   /* It's done.  */
  48   __libc_rwlock_unlock (__libc_setlocale_lock);
  49 
  50   /* Free the locale_t handle itself.  */
  51   free (dataset);
  52 }
