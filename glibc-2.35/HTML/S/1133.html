  22 static int
  23 do_test (void)
  24 {
  25   int ret = do_test_common ();
  26   if (ret == 1)
  27     return 1;
  28 
  29   /* Test passing a negative length.  The compat fadvise64 might use
  30      off64_t for size argument passing, so using -1 for len without
  31      _FILE_OFFSET_BITS might not trigger the length issue.  */
  32   if (posix_fadvise (temp_fd, 0, -1, POSIX_FADV_NORMAL) != EINVAL)
  33     FAIL_EXIT1 ("posix_fadvise with negative length did not return EINVAL");
  34 
  35   /* Check with some offset values larger than 32-bits.  */
  36   off_t offset = UINT32_MAX + 2048LL;
  37   if (posix_fadvise (temp_fd, 0, offset, POSIX_FADV_NORMAL) != 0)
  38     FAIL_EXIT1 ("posix_fadvise failed (offset = 0, len = %zd) failed",
  39         (ssize_t)offset);
  40 
  41   if (posix_fadvise (temp_fd, offset, 0, POSIX_FADV_NORMAL) != 0)
  42     FAIL_EXIT1 ("posix_fadvise failed (offset = %zd, len = 0) failed",
  43         (ssize_t)offset);
  44 
  45   return 0;
  46 }
