  27 /* Maximum number of IFUNC implementations.  */
  28 #define MAX_IFUNC   7
  29 
  30 size_t
  31 __libc_ifunc_impl_list (const char *name, struct libc_ifunc_impl *array,
  32             size_t max)
  33 {
  34   assert (max >= MAX_IFUNC);
  35 
  36   size_t i = 0;
  37 
  38   INIT_ARCH ();
  39 
  40   /* Support sysdeps/aarch64/multiarch/memcpy.c, memmove.c and memset.c.  */
  41   IFUNC_IMPL (i, name, memcpy,
  42           IFUNC_IMPL_ADD (array, i, memcpy, 1, __memcpy_thunderx)
  43           IFUNC_IMPL_ADD (array, i, memcpy, !bti, __memcpy_thunderx2)
  44           IFUNC_IMPL_ADD (array, i, memcpy, 1, __memcpy_falkor)
  45           IFUNC_IMPL_ADD (array, i, memcpy, 1, __memcpy_simd)
  46 #if HAVE_AARCH64_SVE_ASM
  47           IFUNC_IMPL_ADD (array, i, memcpy, sve, __memcpy_a64fx)
  48 #endif
  49           IFUNC_IMPL_ADD (array, i, memcpy, 1, __memcpy_generic))
  50   IFUNC_IMPL (i, name, memmove,
  51           IFUNC_IMPL_ADD (array, i, memmove, 1, __memmove_thunderx)
  52           IFUNC_IMPL_ADD (array, i, memmove, !bti, __memmove_thunderx2)
  53           IFUNC_IMPL_ADD (array, i, memmove, 1, __memmove_falkor)
  54           IFUNC_IMPL_ADD (array, i, memmove, 1, __memmove_simd)
  55 #if HAVE_AARCH64_SVE_ASM
  56           IFUNC_IMPL_ADD (array, i, memmove, sve, __memmove_a64fx)
  57 #endif
  58           IFUNC_IMPL_ADD (array, i, memmove, 1, __memmove_generic))
  59   IFUNC_IMPL (i, name, memset,
  60           /* Enable this on non-falkor processors too so that other cores
  61          can do a comparative analysis with __memset_generic.  */
  62           IFUNC_IMPL_ADD (array, i, memset, (zva_size == 64), __memset_falkor)
  63           IFUNC_IMPL_ADD (array, i, memset, (zva_size == 64), __memset_emag)
  64           IFUNC_IMPL_ADD (array, i, memset, 1, __memset_kunpeng)
  65 #if HAVE_AARCH64_SVE_ASM
  66           IFUNC_IMPL_ADD (array, i, memset, sve, __memset_a64fx)
  67 #endif
  68           IFUNC_IMPL_ADD (array, i, memset, 1, __memset_generic))
  69   IFUNC_IMPL (i, name, memchr,
  70           IFUNC_IMPL_ADD (array, i, memchr, !mte, __memchr_nosimd)
  71           IFUNC_IMPL_ADD (array, i, memchr, 1, __memchr_generic))
  72 
  73   IFUNC_IMPL (i, name, strlen,
  74           IFUNC_IMPL_ADD (array, i, strlen, !mte, __strlen_asimd)
  75           IFUNC_IMPL_ADD (array, i, strlen, 1, __strlen_mte))
  76 
  77   return i;
  78 }
