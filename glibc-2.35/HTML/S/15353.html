  19 /*
  20  * ALGORITHM DESCRIPTION:
  21  *
  22  *   Compute acosh(x) as log(x + sqrt(x*x - 1))
  23  *   using RSQRT instructions for starting the
  24  *   square root approximation, and small table lookups for log
  25  *   that map to AVX-512 permute instructions
  26  *
  27  *   Special cases:
  28  *
  29  *   acosh(NaN)  = quiet NaN, and raise invalid exception
  30  *   acosh(-INF) = NaN
  31  *   acosh(+INF) = +INF
  32  *   acosh(x)    = NaN if x < 1
  33  *   acosh(1)    = +0
  34  *
  35  */
  36

  74         movq      %rsp, %rbp

  76         cfi_offset(6, -16)

  78         subq      $192, %rsp
  79         vmovups   One+__svml_dacosh_data_internal_avx512(%rip), %zmm5
  80 
  81 /* polynomial computation for small inputs */

  83         vmovups   ca1+__svml_dacosh_data_internal_avx512(%rip), %zmm14
  84 
  85 /*
  86  * sqrt(1+x^2) ~ Sh + Sl + Sh*Eh*poly_s
  87  * poly_s = c1+c2*Eh+c3*Eh^2
  88  */

  90         vmovups   c2s+__svml_dacosh_data_internal_avx512(%rip), %zmm2
  91         vmovups   c1s+__svml_dacosh_data_internal_avx512(%rip), %zmm6
  92 
  93 /* very large inputs ? */
  94         vmovups   Threshold+__svml_dacosh_data_internal_avx512(%rip), %zmm15
  95 
  96 /* out of range inputs? */
  97         vmovups   LargeThreshold+__svml_dacosh_data_internal_avx512(%rip), %zmm3
  98 
  99 /* not a very small input ? */
 100         vmovups   SmallThreshold+__svml_dacosh_data_internal_avx512(%rip), %zmm10
 101         vmovaps   %zmm0, %zmm12
 102 
 103 /* x^2 - 1 */
 104         vmovaps   %zmm5, %zmm11
 105         vfmsub231pd {rn-sae}, %zmm12, %zmm12, %zmm11
 106         vcmppd    $21, {sae}, %zmm15, %zmm12, %k2
 107         vcmppd    $22, {sae}, %zmm3, %zmm12, %k0
 108         vcmppd    $18, {sae}, %zmm5, %zmm12, %k1
 109         vrsqrt14pd %zmm11, %zmm4
 110         vcmppd    $21, {sae}, %zmm10, %zmm11, %k3
 111         vfmadd231pd {rn-sae}, %zmm11, %zmm13, %zmm14
 112         vmovups   c3s+__svml_dacosh_data_internal_avx512(%rip), %zmm13
 113 
 114 /* Sh ~sqrt(-1+x^2) */
 115         vmulpd    {rn-sae}, %zmm4, %zmm11, %zmm9
 116         vmulpd    {rn-sae}, %zmm11, %zmm14, %zmm8
 117 
 118 /* Sh+x */
 119         vaddpd    {rn-sae}, %zmm12, %zmm9, %zmm15
 120 
 121 /* Shh */
 122         vsubpd    {rn-sae}, %zmm12, %zmm15, %zmm14
 123 
 124 /* (Yh*R0)_low */
 125         vmovaps   %zmm11, %zmm0
 126         korw      %k0, %k1, %k0
 127 
 128 /* rel. error term: Eh=1-Sh*R0 */
 129         vmovaps   %zmm5, %zmm7
 130         vfmsub213pd {rn-sae}, %zmm9, %zmm4, %zmm0
 131         vfnmadd231pd {rn-sae}, %zmm9, %zmm4, %zmm7
 132 
 133 /* rel. error term: Eh=(1-Sh*R0)-Sl*R0 */
 134         vfnmadd231pd {rn-sae}, %zmm0, %zmm4, %zmm7
 135 
 136 /* Shl */
 137         vsubpd    {rn-sae}, %zmm14, %zmm9, %zmm4
 138         vmovups   poly_coeff7+__svml_dacosh_data_internal_avx512(%rip), %zmm14
 139         vfmadd231pd {rn-sae}, %zmm7, %zmm1, %zmm13
 140         vfmadd213pd {rn-sae}, %zmm2, %zmm7, %zmm13
 141         vfmadd213pd {rn-sae}, %zmm6, %zmm7, %zmm13
 142 
 143 /* Sh*Eh */
 144         vmulpd    {rn-sae}, %zmm7, %zmm9, %zmm7
 145 
 146 /* Sl + Sh*Eh*poly_s */
 147         vfmadd213pd {rn-sae}, %zmm0, %zmm13, %zmm7
 148 
 149 /* polynomials */
 150         vmovups   poly_coeff9+__svml_dacosh_data_internal_avx512(%rip), %zmm13
 151 
 152 /* polynomial computation for small inputs */
 153         vaddpd    {rn-sae}, %zmm7, %zmm9, %zmm0
 154 
 155 /* Xin0+Sl+Sh*Eh*poly_s ~ x+sqrt(1+x^2) */
 156         vaddpd    {rn-sae}, %zmm7, %zmm15, %zmm6
 157         vfmadd231pd {rn-sae}, %zmm0, %zmm8, %zmm0
 158 
 159 /* fixup for very large inputs */
 160         vmovups   OneEighth+__svml_dacosh_data_internal_avx512(%rip), %zmm8
 161 
 162 /* Sl_high */
 163         vsubpd    {rn-sae}, %zmm15, %zmm6, %zmm9
 164         vmovups   poly_coeff6+__svml_dacosh_data_internal_avx512(%rip), %zmm15
 165         vmulpd    {rn-sae}, %zmm8, %zmm12, %zmm6{%k2}
 166 
 167 /* Sl_l */
 168         vsubpd    {rn-sae}, %zmm9, %zmm7, %zmm3
 169         vrcp14pd  %zmm6, %zmm1
 170 
 171 /* Xin_low */
 172         vaddpd    {rn-sae}, %zmm4, %zmm3, %zmm7
 173 
 174 /* Table lookups */
 175         vmovups   __svml_dacosh_data_internal_avx512(%rip), %zmm3
 176 
 177 /* round reciprocal to 1+4b mantissas */
 178         vpaddq    AddB5+__svml_dacosh_data_internal_avx512(%rip), %zmm1, %zmm2
 179 
 180 /* fixup for very large inputs */
 181         vxorpd    %zmm7, %zmm7, %zmm7{%k2}
 182         vmovups   poly_coeff8+__svml_dacosh_data_internal_avx512(%rip), %zmm1
 183         vandpd    RcpBitMask+__svml_dacosh_data_internal_avx512(%rip), %zmm2, %zmm8
 184         vmovups   Log_tbl_L+__svml_dacosh_data_internal_avx512(%rip), %zmm2
 185 
 186 /* Prepare table index */
 187         vpsrlq    $48, %zmm8, %zmm9
 188 
 189 /* reduced argument for log(): (Rcp*Xin-1)+Rcp*Xin_low */
 190         vfmsub231pd {rn-sae}, %zmm8, %zmm6, %zmm5
 191 
 192 /* exponents */
 193         vgetexppd {sae}, %zmm8, %zmm4
 194         vmovups   Four+__svml_dacosh_data_internal_avx512(%rip), %zmm6
 195         vpermt2pd Log_tbl_H+64+__svml_dacosh_data_internal_avx512(%rip), %zmm9, %zmm3
 196         vpermt2pd Log_tbl_L+64+__svml_dacosh_data_internal_avx512(%rip), %zmm9, %zmm2
 197         vsubpd    {rn-sae}, %zmm6, %zmm4, %zmm4{%k2}
 198         vfmadd231pd {rn-sae}, %zmm8, %zmm7, %zmm5
 199         vmovups   poly_coeff5+__svml_dacosh_data_internal_avx512(%rip), %zmm6
 200         vmovups   poly_coeff4+__svml_dacosh_data_internal_avx512(%rip), %zmm7
 201 
 202 /* -K*L2H + Th */
 203         vmovups   L2H+__svml_dacosh_data_internal_avx512(%rip), %zmm8
 204 
 205 /* -K*L2L + Tl */
 206         vmovups   L2L+__svml_dacosh_data_internal_avx512(%rip), %zmm9
 207         vfmadd231pd {rn-sae}, %zmm5, %zmm13, %zmm1
 208         vmovups   poly_coeff2+__svml_dacosh_data_internal_avx512(%rip), %zmm13
 209         vfnmadd231pd {rn-sae}, %zmm4, %zmm8, %zmm3
 210         vfnmadd213pd {rn-sae}, %zmm2, %zmm9, %zmm4
 211         vfmadd213pd {rn-sae}, %zmm14, %zmm5, %zmm1
 212         vmovups   poly_coeff3+__svml_dacosh_data_internal_avx512(%rip), %zmm2
 213         vmovups   poly_coeff1+__svml_dacosh_data_internal_avx512(%rip), %zmm14
 214         vfmadd213pd {rn-sae}, %zmm15, %zmm5, %zmm1
 215 
 216 /* R^2 */
 217         vmulpd    {rn-sae}, %zmm5, %zmm5, %zmm15
 218         vfmadd213pd {rn-sae}, %zmm6, %zmm5, %zmm1
 219         vfmadd213pd {rn-sae}, %zmm7, %zmm5, %zmm1
 220         vfmadd213pd {rn-sae}, %zmm2, %zmm5, %zmm1
 221         vfmadd213pd {rn-sae}, %zmm13, %zmm5, %zmm1
 222         vfmadd213pd {rn-sae}, %zmm14, %zmm5, %zmm1
 223 
 224 /* Tl + R^2*Poly */
 225         vfmadd213pd {rn-sae}, %zmm4, %zmm15, %zmm1
 226 
 227 /* R+Tl + R^2*Poly */
 228         vaddpd    {rn-sae}, %zmm5, %zmm1, %zmm5
 229         vaddpd    {rn-sae}, %zmm5, %zmm3, %zmm0{%k3}
 230 
 231 /* Go to special inputs processing branch */
 232         jne       L(SPECIAL_VALUES_BRANCH)
 233                                 # LOE rbx r12 r13 r14 r15 k0 zmm0 zmm12

 236  * and exit the function
