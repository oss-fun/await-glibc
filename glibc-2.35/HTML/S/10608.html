  18 #include <sysdep.h>
  19 #define _ERRNO_H    1
  20 #include <bits/errno.h>
  21 #include <tcb-offsets.h>
  22 
  23 /* Clone the calling process, but without copying the whole address space.
  24    The calling process is suspended until the new process exits or is
  25    replaced by a call to `execve'.  Return -1 for errors, 0 to the new process,
  26    and the process ID of the new process to the old process.  */
  27 
  28 ENTRY (__vfork)
  29 
  30     /* Pop the return PC value into ECX.  */
  31     popl    %ecx
  32     cfi_adjust_cfa_offset (-4)
  33     cfi_register (%eip, %ecx)
  34 
  35     /* Stuff the syscall number in EAX and enter into the kernel.  */
  36     movl    $SYS_ify (vfork), %eax
  37     int $0x80
  38 
  39     /* Jump to the return PC.  Don't jump directly since this
  40        disturbs the branch target cache.  Instead push the return
  41        address back on the stack.  */
  42     pushl   %ecx
  43     cfi_adjust_cfa_offset (4)
  44 
  45     cmpl    $-4095, %eax
  46     /* Branch forward if it failed.  */
  47     jae SYSCALL_ERROR_LABEL
  48 
  49 #if SHSTK_ENABLED
  50     /* Check if shadow stack is in use.  */
  51     xorl    %edx, %edx
  52     rdsspd  %edx
  53     testl   %edx, %edx
  54     /* Normal return if shadow stack isn't in use.  */
  55     je  L(no_shstk)
  56 
  57     testl   %eax, %eax
  58     /* In parent, normal return.  */
  59     jnz L(no_shstk)
  60 
  61     /* NB: In child, jump back to caller via indirect branch without
  62        popping shadow stack which is shared with parent.  Keep shadow
  63        stack mismatched so that child returns in the vfork-calling
  64        function will trigger SIGSEGV.  */
  65     popl    %ecx
  66     cfi_adjust_cfa_offset (-4)
  67     jmp *%ecx
  68 
  69 L(no_shstk):
  70 #endif
  71 
  72     ret
  73 
  74 PSEUDO_END (__vfork)
  75 libc_hidden_def (__vfork)
  76 
  77 weak_alias (__vfork, vfork)
  78 strong_alias (__vfork, __libc_vfork)
