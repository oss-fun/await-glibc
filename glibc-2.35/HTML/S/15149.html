  19 /*
  20  * ALGORITHM DESCRIPTION:
  21  *
  22  *      SelMask = (|x| >= 0.5) ? 1 : 0;
  23  *      R = SelMask ? sqrt(0.5 - 0.5*|x|) : |x|
  24  *      asin(x) = (SelMask ? (Pi/2 - 2*Poly(R)) : Poly(R))*(-1)^sign(x)
  25  *
  26  *
  27  */

  29 /* Offsets for data table __svml_sasin_data_internal
  30  */
  31 #define AbsMask                         0
  32 #define OneHalf                         64
  33 #define SmallNorm                       128
  34 #define One                             192
  35 #define Two                             256
  36 #define sqrt_coeff_1                    320
  37 #define sqrt_coeff_2                    384
  38 #define poly_coeff_1                    448
  39 #define poly_coeff_2                    512
  40 #define poly_coeff_3                    576
  41 #define poly_coeff_4                    640
  42 #define poly_coeff_5                    704
  43 #define Pi2H                            768

  48     .section .text.exex512,"ax",@progbits
  49 ENTRY(_ZGVeN16v_asinf_skx)
  50         pushq     %rbp
  51         cfi_def_cfa_offset(16)
  52         movq      %rsp, %rbp
  53         cfi_def_cfa(6, 16)
  54         cfi_offset(6, -16)
  55         andq      $-64, %rsp
  56         subq      $192, %rsp
  57         vmovups   __svml_sasin_data_internal(%rip), %zmm4
  58         vmovups   OneHalf+__svml_sasin_data_internal(%rip), %zmm6
  59 
  60 /* SQ ~ -2*sqrt(Y) */
  61         vmovups   SmallNorm+__svml_sasin_data_internal(%rip), %zmm8
  62         vmovups   Two+__svml_sasin_data_internal(%rip), %zmm12
  63         vmovups   sqrt_coeff_1+__svml_sasin_data_internal(%rip), %zmm13
  64         vmovups   One+__svml_sasin_data_internal(%rip), %zmm7
  65         vmovaps   %zmm0, %zmm3
  66 
  67 /* x = |arg| */
  68         vandps    %zmm3, %zmm4, %zmm2
  69         vandnps   %zmm3, %zmm4, %zmm1
  70 
  71 /* x^2 */
  72         vmulps    {rn-sae}, %zmm2, %zmm2, %zmm5
  73         vcmpps    $17, {sae}, %zmm2, %zmm7, %k0
  74         vcmpps    $21, {sae}, %zmm6, %zmm2, %k2
  75         vmovups   poly_coeff_2+__svml_sasin_data_internal(%rip), %zmm7
  76         kmovw     %k0, %edx
  77 
  78 /* Y = 0.5 - 0.5*x */
  79         vmovaps   %zmm6, %zmm9
  80         vfnmadd231ps {rn-sae}, %zmm2, %zmm6, %zmm9
  81         vmovups   poly_coeff_5+__svml_sasin_data_internal(%rip), %zmm6
  82         vrsqrt14ps %zmm9, %zmm10
  83         vcmpps    $17, {sae}, %zmm8, %zmm9, %k1
  84         vminps    {sae}, %zmm9, %zmm5, %zmm0
  85         vmovups   sqrt_coeff_2+__svml_sasin_data_internal(%rip), %zmm8
  86         vmovups   poly_coeff_4+__svml_sasin_data_internal(%rip), %zmm5
  87         vxorps    %zmm10, %zmm10, %zmm10{%k1}
  88         vaddps    {rn-sae}, %zmm9, %zmm9, %zmm14
  89         vmulps    {rn-sae}, %zmm10, %zmm10, %zmm11
  90         vmulps    {rn-sae}, %zmm10, %zmm14, %zmm4
  91         vfmsub213ps {rn-sae}, %zmm12, %zmm11, %zmm14
  92         vmulps    {rn-sae}, %zmm14, %zmm4, %zmm15
  93         vfmadd231ps {rn-sae}, %zmm14, %zmm13, %zmm8
  94         vmovups   poly_coeff_3+__svml_sasin_data_internal(%rip), %zmm14
  95 
  96 /* polynomial */
  97         vmovups   poly_coeff_1+__svml_sasin_data_internal(%rip), %zmm13
  98         vfmsub213ps {rn-sae}, %zmm4, %zmm15, %zmm8
  99         vfmadd231ps {rn-sae}, %zmm0, %zmm14, %zmm5
 100         vfmadd231ps {rn-sae}, %zmm0, %zmm13, %zmm7
 101         vmulps    {rn-sae}, %zmm0, %zmm0, %zmm15
 102         vblendmps %zmm8, %zmm2, %zmm2{%k2}
 103         vfmadd213ps {rn-sae}, %zmm5, %zmm15, %zmm7
 104         vfmadd213ps {rn-sae}, %zmm6, %zmm0, %zmm7
 105         vmulps    {rn-sae}, %zmm0, %zmm7, %zmm9
 106         vmovups   Pi2H+__svml_sasin_data_internal(%rip), %zmm0
 107         vfmadd213ps {rn-sae}, %zmm2, %zmm2, %zmm9
 108         vaddps    {rn-sae}, %zmm0, %zmm9, %zmm9{%k2}
 109         vxorps    %zmm1, %zmm9, %zmm0
 110         testl     %edx, %edx
 111 
 112 /* Go to special inputs processing branch */
 113         jne       L(SPECIAL_VALUES_BRANCH)
 114                                 # LOE rbx r12 r13 r14 r15 edx zmm0 zmm3
 115 
 116 /* Restore registers
 117  * and exit the function
 118  */
 119 
 120 L(EXIT):

 122         popq      %rbp
 123         cfi_def_cfa(7, 8)
 124         cfi_restore(6)
 125         ret

 127         cfi_offset(6, -16)
