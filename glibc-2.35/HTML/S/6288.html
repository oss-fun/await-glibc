  24 __res_enable_icmp (int family, int fd)
  25 {
  26   int one = 1;
  27   switch (family)
  28     {
  29     case AF_INET:
  30       return __setsockopt (fd, SOL_IP, IP_RECVERR, &one, sizeof (one));
  31     case AF_INET6:
  32       return __setsockopt (fd, SOL_IPV6, IPV6_RECVERR, &one, sizeof (one));
  33     default:

  35       return -1;
  36     }
