  19 /*
  20  * ALGORITHM DESCRIPTION:
  21  *
  22  *   Compute asinh(x) as log(x + sqrt(x*x + 1))
  23  *
  24  *   Special cases:
  25  *
  26  *   asinh(NaN) = quiet NaN, and raise invalid exception
  27  *   asinh(INF) = that INF
  28  *   asinh(0)   = that 0
  29  *
  30  */
  31 
  32 /* Offsets for data table __svml_sasinh_data_internal
  33  */

  35 #define sOne                            32
  36 #define sPoly                           64
  37 #define iBrkValue                       320
  38 #define iOffExpoMask                    352
  39 #define sBigThreshold                   384
  40 #define sC2                             416
  41 #define sC3                             448
  42 #define sHalf                           480
  43 #define sLargestFinite                  512
  44 #define sLittleThreshold                544
  45 #define sSign                           576
  46 #define sThirtyOne                      608
  47 #define sTopMask8                       640
  48 #define XScale                          672
  49 #define sLn2                            704

  53         .text
  54     .section .text.avx2,"ax",@progbits
  55 ENTRY(_ZGVdN8v_asinhf_avx2)
  56         pushq     %rbp
  57         cfi_def_cfa_offset(16)
  58         movq      %rsp, %rbp
  59         cfi_def_cfa(6, 16)
  60         cfi_offset(6, -16)
  61         andq      $-32, %rsp
  62         subq      $96, %rsp
  63         vmovaps   %ymm0, %ymm9
  64 
  65 /* Load the constant 1 and a sign mask */
  66         vmovups   sOne+__svml_sasinh_data_internal(%rip), %ymm8
  67 
  68 /* No need to split X when FMA is available in hardware. */
  69         vmulps    %ymm9, %ymm9, %ymm5
  70         vmovups   sTopMask8+__svml_sasinh_data_internal(%rip), %ymm1
  71 
  72 /*
  73  * Finally, express Y + W = X^2 + 1 accurately where Y has <= 8 bits.
  74  * If |X| <= 1 then |XHi| <= 1 and so |X2Hi| <= 1, so we can treat 1
  75  * as the dominant component in the compensated summation. Otherwise,
  76  * if |X| >= 1, then since X2Hi only has 22 significant bits, the basic
  77  * addition will be exact anyway until we get to |X| >= 2^24. But by
  78  * that time the log function is well-conditioned enough that the
  79  * rounding error doesn't matter. Hence we can treat 1 as dominant even
  80  * if it literally isn't.
  81  */
  82         vaddps    %ymm5, %ymm8, %ymm13
  83         vandps    %ymm1, %ymm13, %ymm2
  84         vmovaps   %ymm9, %ymm4
  85         vsubps    %ymm13, %ymm8, %ymm11
  86         vsubps    %ymm2, %ymm13, %ymm15
  87 
  88 /*
  89  * Compute R = 1/sqrt(Y + W) * (1 + d)
  90  * Force R to <= 8 significant bits.
  91  * This means that R * Y and R^2 * Y are exactly representable.
  92  */
  93         vrsqrtps  %ymm2, %ymm0
  94         vfmsub213ps %ymm5, %ymm9, %ymm4
  95         vaddps    %ymm11, %ymm5, %ymm12
  96 
  97 /*
  98  * Get the absolute value of the input, since we will exploit antisymmetry
  99  * and mostly assume X >= 0 in the core computation
 100  */
 101         vandps    SgnMask+__svml_sasinh_data_internal(%rip), %ymm9, %ymm6
 102 
 103 /*
 104  * Check whether the input is finite, by checking |X| <= MaxFloat
 105  * Otherwise set the rangemask so that the callout will get used.
 106  * Note that this will also use the callout for NaNs since not(NaN <= MaxFloat)
 107  */
 108         vcmpnle_uqps sLargestFinite+__svml_sasinh_data_internal(%rip), %ymm6, %ymm10
 109         vaddps    %ymm12, %ymm4, %ymm14
 110 
 111 /*
 112  * Unfortunately, we can still be in trouble if |X| <= 2^-5, since
 113  * the absolute error 2^-(7+24)-ish in sqrt(1 + X^2) gets scaled up
 114  * by 1/X and comes close to our threshold. Hence if |X| <= 2^-4,
 115  * perform an alternative computation
 116  * sqrt(1 + X^2) - 1 = X^2/2 - X^4/8 + X^6/16
 117  * X2 = X^2
 118  */
 119         vaddps    %ymm4, %ymm5, %ymm4
 120 
 121 /*
 122  * The following computation can go wrong for very large X, basically
 123  * because X^2 overflows. But for large X we have
 124  * asinh(X) / log(2 X) - 1 =~= 1/(4 * X^2), so for X >= 2^30
 125  * we can just later stick X back into the log and tweak up the exponent.
 126  * Actually we scale X by 2^-30 and tweak the exponent up by 31,
 127  * to stay in the safe range for the later log computation.
 128  * Compute a flag now telling us when do do this.
 129  */
 130         vcmplt_oqps sBigThreshold+__svml_sasinh_data_internal(%rip), %ymm6, %ymm7
 131         vaddps    %ymm15, %ymm14, %ymm3
 132 
 133 /*
 134  * Now       1 / (1 + d)
 135  * = 1 / (1 + (sqrt(1 - e) - 1))
 136  * = 1 / sqrt(1 - e)
 137  * = 1 + 1/2 * e + 3/8 * e^2 + 5/16 * e^3 + 35/128 * e^4 + ...
 138  * So compute the first three nonconstant terms of that, so that
 139  * we have a relative correction (1 + Corr) to apply to S etc.
 140  * C1 = 1/2
 141  * C2 = 3/8
 142  * C3 = 5/16
 143  */
 144         vmovups   sC3+__svml_sasinh_data_internal(%rip), %ymm12
 145         vmovmskps %ymm10, %edx
 146         vandps    %ymm1, %ymm0, %ymm10
 147 
 148 /*
 149  * Compute S = (Y/sqrt(Y + W)) * (1 + d)
 150  * and T = (W/sqrt(Y + W)) * (1 + d)
 151  * so that S + T = sqrt(Y + W) * (1 + d)
 152  * S is exact, and the rounding error in T is OK.
 153  */
 154         vmulps    %ymm10, %ymm2, %ymm15
 155         vmulps    %ymm3, %ymm10, %ymm14
 156         vmovups   sHalf+__svml_sasinh_data_internal(%rip), %ymm3
 157         vsubps    %ymm8, %ymm15, %ymm0
 158 
 159 /*
 160  * Obtain sqrt(1 + X^2) - 1 in two pieces
 161  * sqrt(1 + X^2) - 1
 162  * = sqrt(Y + W) - 1
 163  * = (S + T) * (1 + Corr) - 1
 164  * = [S - 1] + [T + (S + T) * Corr]
 165  * We need a compensated summation for the last part. We treat S - 1
 166  * as the larger part; it certainly is until about X < 2^-4, and in that
 167  * case, the error is affordable since X dominates over sqrt(1 + X^2) - 1
 168  * Final sum is dTmp5 (hi) + dTmp7 (lo)
 169  */
 170         vaddps    %ymm14, %ymm15, %ymm13
 171 
 172 /*
 173  * Compute e = -(2 * d + d^2)
 174  * The first FMR is exact, and the rounding error in the other is acceptable
 175  * since d and e are ~ 2^-8
 176  */
 177         vmovaps   %ymm8, %ymm11
 178         vfnmadd231ps %ymm15, %ymm10, %ymm11
 179         vfnmadd231ps %ymm14, %ymm10, %ymm11
 180         vfmadd213ps sC2+__svml_sasinh_data_internal(%rip), %ymm11, %ymm12
 181         vfmadd213ps %ymm3, %ymm11, %ymm12
 182         vmulps    %ymm12, %ymm11, %ymm1
 183 
 184 /* Now multiplex the two possible computations */
 185         vcmple_oqps sLittleThreshold+__svml_sasinh_data_internal(%rip), %ymm6, %ymm11
 186         vfmadd213ps %ymm14, %ymm13, %ymm1
 187         vaddps    %ymm0, %ymm1, %ymm2
 188         vsubps    %ymm2, %ymm0, %ymm10
 189 
 190 /* sX2over2 = X^2/2 */
 191         vmulps    %ymm4, %ymm3, %ymm0
 192         vaddps    %ymm10, %ymm1, %ymm1
 193 
 194 /* sX4over4 = X^4/4 */
 195         vmulps    %ymm0, %ymm0, %ymm5
 196 
 197 /* sX46 = -X^4/4 + X^6/8 */
 198         vfmsub231ps %ymm0, %ymm5, %ymm5
 199 
 200 /* sX46over2 = -X^4/8 + x^6/16 */
 201         vmulps    %ymm5, %ymm3, %ymm3
 202         vaddps    %ymm3, %ymm0, %ymm5
 203         vblendvps %ymm11, %ymm5, %ymm2, %ymm2
 204         vsubps    %ymm5, %ymm0, %ymm4
 205 
 206 /*
 207  * Now do another compensated sum to add |X| + [sqrt(1 + X^2) - 1].
 208  * It's always safe to assume |X| is larger.
 209  * This is the final 2-part argument to the log1p function
 210  */
 211         vaddps    %ymm2, %ymm6, %ymm14
 212 
 213 /*
 214  * Now resume the main code.
 215  * reduction: compute r,n
 216  */
 217         vmovups   iBrkValue+__svml_sasinh_data_internal(%rip), %ymm5
 218         vaddps    %ymm4, %ymm3, %ymm10
 219 
 220 /*
 221  * Now we feed into the log1p code, using H in place of _VARG1 and
 222  * also adding L into Xl.
 223  * compute 1+x as high, low parts
 224  */
 225         vmaxps    %ymm14, %ymm8, %ymm15
 226         vminps    %ymm14, %ymm8, %ymm0
 227         vblendvps %ymm11, %ymm10, %ymm1, %ymm12
 228         vsubps    %ymm14, %ymm6, %ymm1
 229         vaddps    %ymm0, %ymm15, %ymm3
 230 
 231 /* Now multiplex to the case X = 2^-30 * input, Xl = sL = 0 in the "big" case. */
 232         vmulps    XScale+__svml_sasinh_data_internal(%rip), %ymm6, %ymm6
 233         vaddps    %ymm1, %ymm2, %ymm13
 234         vsubps    %ymm3, %ymm15, %ymm15
 235         vaddps    %ymm13, %ymm12, %ymm1
 236         vaddps    %ymm15, %ymm0, %ymm2
 237         vblendvps %ymm7, %ymm3, %ymm6, %ymm0
 238         vaddps    %ymm2, %ymm1, %ymm4
 239         vpsubd    %ymm5, %ymm0, %ymm1
 240         vpsrad    $23, %ymm1, %ymm6
 241         vpand     iOffExpoMask+__svml_sasinh_data_internal(%rip), %ymm1, %ymm2
 242         vmovups   sPoly+224+__svml_sasinh_data_internal(%rip), %ymm1
 243         vpslld    $23, %ymm6, %ymm10
 244         vpaddd    %ymm5, %ymm2, %ymm13
 245         vcvtdq2ps %ymm6, %ymm0
 246         vpsubd    %ymm10, %ymm8, %ymm12
 247 
 248 /* polynomial evaluation */
 249         vsubps    %ymm8, %ymm13, %ymm8
 250 
 251 /* Add 31 to the exponent in the "large" case to get log(2 * input) */
 252         vaddps    sThirtyOne+__svml_sasinh_data_internal(%rip), %ymm0, %ymm3
 253         vandps    %ymm7, %ymm4, %ymm11
 254         vmulps    %ymm12, %ymm11, %ymm14
 255         vblendvps %ymm7, %ymm0, %ymm3, %ymm0
 256         vaddps    %ymm8, %ymm14, %ymm2
 257         vfmadd213ps sPoly+192+__svml_sasinh_data_internal(%rip), %ymm2, %ymm1
 258         vfmadd213ps sPoly+160+__svml_sasinh_data_internal(%rip), %ymm2, %ymm1
 259         vfmadd213ps sPoly+128+__svml_sasinh_data_internal(%rip), %ymm2, %ymm1
 260         vfmadd213ps sPoly+96+__svml_sasinh_data_internal(%rip), %ymm2, %ymm1
 261         vfmadd213ps sPoly+64+__svml_sasinh_data_internal(%rip), %ymm2, %ymm1
 262         vfmadd213ps sPoly+32+__svml_sasinh_data_internal(%rip), %ymm2, %ymm1
 263         vfmadd213ps sPoly+__svml_sasinh_data_internal(%rip), %ymm2, %ymm1
 264         vmulps    %ymm1, %ymm2, %ymm4
 265         vfmadd213ps %ymm2, %ymm2, %ymm4
 266 
 267 /* final reconstruction */
 268         vfmadd132ps sLn2+__svml_sasinh_data_internal(%rip), %ymm4, %ymm0
 269 
 270 /* Finally, reincorporate the original sign. */
 271         vandps    sSign+__svml_sasinh_data_internal(%rip), %ymm9, %ymm7
 272         vxorps    %ymm0, %ymm7, %ymm0
 273         testl     %edx, %edx
 274 
 275 /* Go to special inputs processing branch */
 276         jne       L(SPECIAL_VALUES_BRANCH)
 277                                 # LOE rbx r12 r13 r14 r15 edx ymm0 ymm9
 278 
 279 /* Restore registers
 280  * and exit the function
 281  */
 282 
 283 L(EXIT):
 284         movq      %rbp, %rsp
 285         popq      %rbp
 286         cfi_def_cfa(7, 8)
 287         cfi_restore(6)
 288         ret
 289         cfi_def_cfa(6, 16)
 290         cfi_offset(6, -16)
 291

 293  * special inputs
 294  */
 295

 297         vmovups   %ymm9, 32(%rsp)
 298         vmovups   %ymm0, 64(%rsp)
 299                                 # LOE rbx r12 r13 r14 r15 edx ymm0

 302                                 # LOE rbx r12 r13 r14 r15 eax edx

 304         vzeroupper
 305         movq      %r12, 16(%rsp)
 306         /*  DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -80; DW_OP_plus)  */

 308         movl      %eax, %r12d
 309         movq      %r13, 8(%rsp)
 310         /*  DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -88; DW_OP_plus)  */

 312         movl      %edx, %r13d
 313         movq      %r14, (%rsp)
 314         /*  DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -96; DW_OP_plus)  */
 315         .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0xa0, 0xff, 0xff, 0xff, 0x22
 316                                 # LOE rbx r15 r12d r13d

 319  * bits check

 323         btl       %r12d, %r13d
 324

 326         jc        L(SCALAR_MATH_CALL)

 343         movq      8(%rsp), %r13

 345         movq      (%rsp), %r14

 355         /*  DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -32; DW_OP_and; DW_OP_const4s: -96; DW_OP_plus)  */
