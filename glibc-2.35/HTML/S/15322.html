  19 /*
  20  * ALGORITHM DESCRIPTION:
  21  *
  22  *   Compute sinh(x) as (exp(x)-exp(-x))/2,
  23  *   where exp is calculated as
  24  *   exp(M*ln2 + ln2*(j/2^k) + r) = 2^M * 2^(j/2^k) * exp(r)
  25  *
  26  *   Special cases:
  27  *
  28  *   sinh(NaN) = quiet NaN, and raise invalid exception
  29  *   sinh(INF) = that INF
  30  *   sinh(x)   = x for subnormals
  31  *   sinh(x) overflows for big x and returns MAXLOG+log(2)
  32  *
  33  */
  34 
  35 /* Offsets for data table __svml_ssinh_data_internal
  36  */

  38 #define _sLn2hi                         64
  39 #define _sLn2lo                         128
  40 #define _sSign                          192
  41 #define _sShifter                       256
  42 #define _iDomainRange                   320
  43 #define _sPC1                           384
  44 #define _sPC2                           448
  45 #define _sPC3                           512
  46 #define _sPC4                           576

  53         .text
  54     .section .text.exex512,"ax",@progbits
  55 ENTRY(_ZGVeN16v_sinhf_skx)
  56         pushq     %rbp
  57         cfi_def_cfa_offset(16)
  58         movq      %rsp, %rbp
  59         cfi_def_cfa(6, 16)
  60         cfi_offset(6, -16)
  61         andq      $-64, %rsp
  62         subq      $192, %rsp
  63         vmovaps   %zmm0, %zmm5
  64 
  65 /*
  66  *  Implementation
  67  *  Abs argument
  68  */
  69         vandps    _sSign+__svml_ssinh_data_internal(%rip), %zmm5, %zmm4
  70 
  71 /*
  72  * Check for overflow\underflow
  73  * MORE faster than GE?
  74  */
  75         vpternlogd $255, %zmm6, %zmm6, %zmm6
  76         vmovups   _sShifter+__svml_ssinh_data_internal(%rip), %zmm7
  77 
  78 /*
  79  *  Load argument
  80  * dM = x/log(2) + RShifter
  81  */
  82         vmovups   _sInvLn2+__svml_ssinh_data_internal(%rip), %zmm11
  83         vmovups   _sLn2hi+__svml_ssinh_data_internal(%rip), %zmm8
  84         vmovups   _sLn2lo+__svml_ssinh_data_internal(%rip), %zmm10
  85         vmovups   _iHalf+__svml_ssinh_data_internal(%rip), %zmm12
  86         vmovups   _sPC5+__svml_ssinh_data_internal(%rip), %zmm0
  87         vmovups   _sPC6+__svml_ssinh_data_internal(%rip), %zmm3
  88 
  89 /* x^2 */
  90         vmovups   _sPC2+__svml_ssinh_data_internal(%rip), %zmm2
  91         vxorps    %zmm5, %zmm4, %zmm1
  92         vfmadd213ps {rn-sae}, %zmm7, %zmm1, %zmm11
  93         vpcmpd    $2, _iDomainRange+__svml_ssinh_data_internal(%rip), %zmm1, %k1
  94 
  95 /*
  96  *  G1,G2 2^N,2^(-N)
  97  * iM now is an EXP(2^N)
  98  */
  99         vpslld    $23, %zmm11, %zmm13
 100 
 101 /*
 102  *  R
 103  * sN = sM - RShifter
 104  */
 105         vsubps    {rn-sae}, %zmm7, %zmm11, %zmm9
 106         vpaddd    %zmm13, %zmm12, %zmm14
 107         vpsubd    %zmm13, %zmm12, %zmm15
 108 
 109 /* sG1 = 2^(N-1)+2^(-N-1) */
 110         vaddps    {rn-sae}, %zmm15, %zmm14, %zmm7
 111         vpandnd   %zmm1, %zmm1, %zmm6{%k1}
 112 
 113 /* sR = sX - sN*Log2_hi */
 114         vfnmadd231ps {rn-sae}, %zmm8, %zmm9, %zmm1
 115         vptestmd  %zmm6, %zmm6, %k0
 116 
 117 /* sG2 = 2^(N-1)-2^(-N-1) */
 118         vsubps    {rn-sae}, %zmm15, %zmm14, %zmm8
 119 
 120 /* sR = (sX - sN*Log2_hi) - sN*Log2_lo */
 121         vfnmadd231ps {rn-sae}, %zmm10, %zmm9, %zmm1
 122 
 123 /*
 124  * sinh(r) = r*((a1=1)+r^2*(a3+r^2*(a5+{v1 r^2*a7})))) = r + r*(r^2*(a3+r^2*(a5+r^2*a7))) ....
 125  * sSinh_r = (a3+r^2*a5)
 126  */
 127         vmovups   _sPC3+__svml_ssinh_data_internal(%rip), %zmm14
 128         kmovw     %k0, %edx
 129 
 130 /* sR2 = sR^2 */
 131         vmulps    {rn-sae}, %zmm1, %zmm1, %zmm6
 132         vfmadd231ps {rn-sae}, %zmm6, %zmm0, %zmm14
 133 
 134 /* sSinh_r = r^2*(a3+r^2*a5) */
 135         vmulps    {rn-sae}, %zmm6, %zmm14, %zmm0
 136 
 137 /* sSinh_r = r + r*(r^2*(a3+r^2*a5)) */
 138         vfmadd213ps {rn-sae}, %zmm1, %zmm1, %zmm0
 139 
 140 /*
 141  * sinh(X) = sG2 + sG1*sinh(dR) + sG2*sR2*(a2+sR2*(a4+a6*sR2)
 142  * sOut = (a4 +a6*sR2)
 143  */
 144         vmovups   _sPC4+__svml_ssinh_data_internal(%rip), %zmm1
 145         vfmadd231ps {rn-sae}, %zmm6, %zmm3, %zmm1
 146 
 147 /* sOut = a2+sR2*(a4+a6*sR2) */
 148         vfmadd213ps {rn-sae}, %zmm2, %zmm6, %zmm1
 149 
 150 /* sOut = sR2*(a2+sR2*(a4+a6*sR2) */
 151         vmulps    {rn-sae}, %zmm6, %zmm1, %zmm2
 152 
 153 /* sOut = sG2*sR2*(a2+sR2*(a4+a6*sR2) */
 154         vmulps    {rn-sae}, %zmm8, %zmm2, %zmm3
 155 
 156 /* sOut = sG1*sinh(dR)+sG2*sR2*(a2+sR2*(a4+a6*sR2) */
 157         vfmadd213ps {rn-sae}, %zmm3, %zmm0, %zmm7
 158 
 159 /* sOut = sG2 + sG1*sinh(dR) + sG2*sR2*(a2+sR2*(a4+a6*sR2) */
 160         vaddps    {rn-sae}, %zmm8, %zmm7, %zmm9
 161 
 162 /*  Ret H  */
 163         vorps     %zmm9, %zmm4, %zmm0
 164         testl     %edx, %edx
 165 
 166 /* Go to special inputs processing branch */
 167         jne       L(SPECIAL_VALUES_BRANCH)
 168                                 # LOE rbx r12 r13 r14 r15 edx zmm0 zmm5
 169 
 170 /* Restore registers
 171  * and exit the function
 172  */
 173 
 174 L(EXIT):
 175         movq      %rbp, %rsp
 176         popq      %rbp
 177         cfi_def_cfa(7, 8)
 178         cfi_restore(6)
 179         ret
 180         cfi_def_cfa(6, 16)
 181         cfi_offset(6, -16)
 182 
 183 /* Branch to process
 184  * special inputs
 185  */
 186 
 187 L(SPECIAL_VALUES_BRANCH):
 188         vmovups   %zmm5, 64(%rsp)
 189         vmovups   %zmm0, 128(%rsp)
 190                                 # LOE rbx r12 r13 r14 r15 edx zmm0
 191 
 192         xorl      %eax, %eax
 193                                 # LOE rbx r12 r13 r14 r15 eax edx
 194 
 195         vzeroupper
 196         movq      %r12, 16(%rsp)
 197         /*  DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -176; DW_OP_plus)  */
 198         .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x50, 0xff, 0xff, 0xff, 0x22
 199         movl      %eax, %r12d
 200         movq      %r13, 8(%rsp)
 201         /*  DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -184; DW_OP_plus)  */
 202         .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x48, 0xff, 0xff, 0xff, 0x22
 203         movl      %edx, %r13d
 204         movq      %r14, (%rsp)
 205         /*  DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -192; DW_OP_plus)  */
 206         .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x40, 0xff, 0xff, 0xff, 0x22
 207                                 # LOE rbx r15 r12d r13d
 208 
 209 /* Range mask
 210  * bits check
 211  */
 212 
 213 L(RANGEMASK_CHECK):
 214         btl       %r12d, %r13d
 215 
 216 /* Call scalar math function */
 217         jc        L(SCALAR_MATH_CALL)
 218                                 # LOE rbx r15 r12d r13d
 219 
 220 /* Special inputs
 221  * processing loop
 222  */
 223 
 224 L(SPECIAL_VALUES_LOOP):
 225         incl      %r12d
 226         cmpl      $16, %r12d
 227 
 228 /* Check bits in range mask */
 229         jl        L(RANGEMASK_CHECK)
 230                                 # LOE rbx r15 r12d r13d
 231 
 232         movq      16(%rsp), %r12
 233         cfi_restore(12)
 234         movq      8(%rsp), %r13
 235         cfi_restore(13)
 236         movq      (%rsp), %r14
 237         cfi_restore(14)
 238         vmovups   128(%rsp), %zmm0
 239 
 240 /* Go to exit */
 241         jmp       L(EXIT)
 242         /*  DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -176; DW_OP_plus)  */
 243         .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x50, 0xff, 0xff, 0xff, 0x22
 244         /*  DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -184; DW_OP_plus)  */
 245         .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x48, 0xff, 0xff, 0xff, 0x22
 246         /*  DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -192; DW_OP_plus)  */
 247         .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x40, 0xff, 0xff, 0xff, 0x22
 248                                 # LOE rbx r12 r13 r14 r15 zmm0
 249 
 250 /* Scalar math fucntion call
 251  * to process special input
 252  */
 253 
 254 L(SCALAR_MATH_CALL):
 255         movl      %r12d, %r14d
 256         movss     64(%rsp,%r14,4), %xmm0
 257         call      sinhf@PLT
 258                                 # LOE rbx r14 r15 r12d r13d xmm0
 259 
 260         movss     %xmm0, 128(%rsp,%r14,4)
 261 
 262 /* Process special inputs in loop */
 263         jmp       L(SPECIAL_VALUES_LOOP)
 264                                 # LOE rbx r15 r12d r13d
