  21 /* The user-visible size of struct termios has changed.  Catch ioctl calls
  22    using the new-style struct termios, and translate them to old-style.  */
  23 static inline bool
  24 __ioctl_arch (int *r, int fd, unsigned long request, void *arg)
  25 {
  26   switch (request)
  27     {
  28     case TCGETS:
  29       *r = __tcgetattr (fd, (struct termios *) arg);
  30       break;
  31 
  32     case TCSETS:
  33       *r = __tcsetattr (fd, TCSANOW, (struct termios *) arg);
  34       break;
  35 
  36     case TCSETSW:
  37       *r = __tcsetattr (fd, TCSADRAIN, (struct termios *) arg);
  38       break;
  39 
  40     case TCSETSF:
  41       *r = __tcsetattr (fd, TCSAFLUSH, (struct termios *) arg);
  42       break;
  43 
  44     default:
  45       return false;
  46     }
  47   return true;
  48 }
