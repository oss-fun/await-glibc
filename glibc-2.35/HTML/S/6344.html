  88 /* Packs domain name SRC in presentation form into DST.  Returns the
  89    size of the compressed name or -1.  DSTSIZ is the size of the array
  90    pointed to by COMPDN.  */
  91 int
  92 ___dn_comp (const char *src, unsigned char *dst, int dstsiz,
  93             unsigned char **dnptrs, unsigned char **lastdnptr)
  94 {
  95   return __ns_name_compress (src, dst, (size_t) dstsiz,
  96                              (const unsigned char **) dnptrs,
  97                              (const unsigned char **) lastdnptr);
  98 }
