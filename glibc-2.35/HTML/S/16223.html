  32 static void *
  33 tf (void *arg)
  34 {
  35   /* Lock the read-write lock.  */
  36   TEST_COMPARE (pthread_rwlock_wrlock (&r), 0);
  37 
  38   pthread_t mt = *(pthread_t *) arg;
  39 
  40   xpthread_barrier_wait (&b);
  41 
  42   /* This call will never return.  */
  43   xpthread_join (mt);
  44 
  45   return NULL;
  46 }
