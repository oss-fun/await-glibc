  30 static int
  31 do_sendto (const struct sockaddr_in *addr, int payload)
  32 {
  33   int s = xsocket (AF_INET, SOCK_DGRAM | SOCK_CLOEXEC, 0);
  34   xconnect (s, (const struct sockaddr *) addr, sizeof (*addr));
  35 
  36   xsendto (s, &payload, sizeof (payload), 0, (const struct sockaddr *) addr,
  37        sizeof (*addr));
  38 
  39   xclose (s);
  40 
  41   return 0;
  42 }

  44 static void
  45 do_recvmsg_ancillary (bool use_multi_call, struct mmsghdr *mmhdr,
  46               void *msgbuf, size_t msgbuflen, int exp_payload)
  47 {
  48   int payload;
  49   struct iovec iov =
  50     {
  51       .iov_base = &payload,
  52       .iov_len = sizeof (payload)
  53     };
  54   mmhdr->msg_hdr.msg_name = NULL;
  55   mmhdr->msg_hdr.msg_iov = &iov;
  56   mmhdr->msg_hdr.msg_iovlen = 1;
  57   mmhdr->msg_hdr.msg_control = msgbuf;
  58   mmhdr->msg_hdr.msg_controllen = msgbuflen;
  59 
  60   int r;
  61   if (use_multi_call)
  62     {
  63       r = recvmmsg (srv, mmhdr, 1, 0, NULL);
  64       if (r >= 0)
  65     r = mmhdr->msg_len;
  66     }
  67   else
  68     r = recvmsg (srv, &mmhdr->msg_hdr, 0);
  69   TEST_COMPARE (r, sizeof (int));
  70   TEST_COMPARE (payload, exp_payload);
  71 }

  73 /* Check if recvmsg create the additional 64 bit timestamp if only 32 bit
  74    is enabled for 64 bit recvmsg symbol.  */
  75 static void
  76 do_test_large_buffer (bool mc)
  77 {
  78   struct mmsghdr mmhdr = { 0 };
  79   /* It should be large enought for either timeval/timespec and the
  80      64 time type as well.  */
  81 
  82   union
  83   {
  84     struct cmsghdr cmsghdr;
  85     char msgbuf[512];
  86   } control;
  87 
  88   /* Enable 32 bit timeval precision and check if no 64 bit timeval stamp
  89      is created.  */
  90   {
  91     int r = setsockopt (srv, SOL_SOCKET, SO_TIMESTAMP_OLD, &(int){1},
  92             sizeof (int));
  93     TEST_VERIFY_EXIT (r != -1);
  94 
  95     do_sendto (&srv_addr, 42);
  96     do_recvmsg_ancillary (mc, &mmhdr, &control, sizeof control, 42);
  97 
  98     bool found_timestamp = false;
  99     for (struct cmsghdr *cmsg = CMSG_FIRSTHDR (&mmhdr.msg_hdr);
 100      cmsg != NULL;
 101      cmsg = CMSG_NXTHDR (&mmhdr.msg_hdr, cmsg))
 102     {
 103       if (cmsg->cmsg_level != SOL_SOCKET)
 104     continue;
 105 
 106       if (sizeof (time_t) > 4 && cmsg->cmsg_type == SO_TIMESTAMP_NEW)
 107     found_timestamp = true;
 108       else
 109     TEST_VERIFY (cmsg->cmsg_type != SO_TIMESTAMP_NEW);
 110     }
 111 
 112     TEST_COMPARE (found_timestamp, sizeof (time_t) > 4);
 113   }
 114 
 115   /* Same as before, but for timespec.  */
 116   {
 117     int r = setsockopt (srv, SOL_SOCKET, SO_TIMESTAMPNS_OLD, &(int){1},
 118             sizeof (int));
 119     TEST_VERIFY_EXIT (r != -1);
 120 
 121     do_sendto (&srv_addr, 42);
 122     do_recvmsg_ancillary (mc, &mmhdr, &control, sizeof control, 42);
 123 
 124     bool found_timestamp = false;
 125     for (struct cmsghdr *cmsg = CMSG_FIRSTHDR (&mmhdr.msg_hdr);
 126      cmsg != NULL;
 127      cmsg = CMSG_NXTHDR (&mmhdr.msg_hdr, cmsg))
 128     {

 131

 133     found_timestamp = true;
 134       else
 135     TEST_VERIFY (cmsg->cmsg_type != SO_TIMESTAMPNS_NEW);
 136     }

 142 /* Check if recvmsg does not create the additional 64 bit timestamp if
 143    only 32 bit timestamp is enabled if the ancillary buffer is not large
 144    enought.  Also checks if MSG_CTRUNC is set iff for 64 bit recvmsg
 145    symbol.  */
 146 static void
 147 do_test_small_buffer (bool mc)
 148 {
 149   struct mmsghdr mmhdr = { 0 };
 150 
 151   /* Enable 32 bit timeval precision and check if no 64 bit timeval stamp
 152      is created.  */
 153   {
 154     int r = setsockopt (srv, SOL_SOCKET, SO_TIMESTAMP_OLD, &(int){1},
 155             sizeof (int));
 156     TEST_VERIFY_EXIT (r != -1);
 157 
 158     union
 159     {
 160       struct cmsghdr cmsghdr;
 161       char msgbuf[CMSG_SPACE (sizeof (struct timeval))];
 162     } control;
 163 
 164     do_sendto (&srv_addr, 42);
 165     do_recvmsg_ancillary (mc, &mmhdr, &control, sizeof control, 42);
 166 
 167     bool found_timestamp = false;
 168     for (struct cmsghdr *cmsg = CMSG_FIRSTHDR (&mmhdr.msg_hdr);
 169      cmsg != NULL;
 170      cmsg = CMSG_NXTHDR (&mmhdr.msg_hdr, cmsg))
 171     {
 172       if (cmsg->cmsg_level != SOL_SOCKET)
 173     continue;
 174 
 175       if (sizeof (time_t) > 4 && cmsg->cmsg_type == SO_TIMESTAMP_NEW)
 176     found_timestamp = true;
 177       else
 178     TEST_VERIFY (cmsg->cmsg_type != SO_TIMESTAMP_NEW);
 179     }
 180

 235 static int
 236 do_test (void)
 237 {
 238   /* This test only make sense for ABIs that support 32 bit time_t socket
 239      timestampss.  */
 240   if (sizeof (time_t) > 4 && __TIMESIZE == 64)
 241     return 0;
 242 
 243   srv = xsocket (AF_INET, SOCK_DGRAM, 0);
 244   srv_addr = (struct sockaddr_in) {
 245     .sin_family = AF_INET,
 246     .sin_addr = {.s_addr = htonl (INADDR_LOOPBACK) },
 247   };
 248   xbind (srv, (struct sockaddr *) &srv_addr, sizeof (srv_addr));
 249   {
 250     socklen_t sa_len = sizeof (srv_addr);
 251     xgetsockname (srv, (struct sockaddr *) &srv_addr, &sa_len);
 252     TEST_VERIFY (sa_len == sizeof (srv_addr));
 253   }
 254 
 255   /* Check recvmsg;  */
 256   do_test_large_buffer (false);
 257   do_test_small_buffer (false);
 258   /* Check recvmmsg.  */
 259   do_test_large_buffer (true);
 260   do_test_small_buffer (true);
 261 
 262   return 0;
 263 }
