  42 SIMPLE_STRCSPN (const CHAR *s, const CHAR *rej)
  43 {
  44   const CHAR *r, *str = s;
  45   CHAR c;
  46 
  47   while ((c = *s++) != '\0')
  48     for (r = rej; *r != '\0'; ++r)
  49       if (*r == c)
  50     return s - str - 1;
  51   return s - str - 1;
