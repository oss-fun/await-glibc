  84 #define clrbit(a,i)     ((a)[(i)/NBBY] &= ~(1<<((i)%NBBY)))

  93 # define roundup(x, y)  (__builtin_constant_p (y) && powerof2 (y)             \

  95                          : ((((x) + ((y) - 1)) / (y)) * (y)))
