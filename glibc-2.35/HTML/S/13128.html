  54     mov LEN(%esp), %edx
  55     test    %edx, %edx

  59

  74     je  L(unaligned_no_match_1)
  75 /* Check which byte is a match.  */
  76     bsf %ecx, %ecx
  77 
  78 # ifndef USE_AS_RAWMEMCHR
  79     sub %ecx, %edx
  80     jbe L(return_null_1)
  81 # endif

  83     ret
