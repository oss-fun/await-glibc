  91 /* Close all open sockets.  If FREE_ADDR is true, deallocate any
  92    separately allocated name server addresses.  */
  93 void
  94 __res_iclose (res_state statp, bool free_addr)
  95 {
  96   if (statp->_vcsock >= 0)
  97     {
  98       __close_nocancel_nostatus (statp->_vcsock);
  99       statp->_vcsock = -1;
 100       statp->_flags &= ~(RES_F_VC | RES_F_CONN);
 101     }
 102   for (int ns = 0; ns < statp->nscount; ns++)
 103     if (statp->_u._ext.nsaddrs[ns] != NULL)
 104       {
 105         if (statp->_u._ext.nssocks[ns] != -1)
 106           {
 107             __close_nocancel_nostatus (statp->_u._ext.nssocks[ns]);
 108             statp->_u._ext.nssocks[ns] = -1;
 109           }
 110         if (free_addr)
 111           {
 112             free (statp->_u._ext.nsaddrs[ns]);
 113             statp->_u._ext.nsaddrs[ns] = NULL;
 114           }
 115       }
 116   if (free_addr)
 117     __resolv_conf_detach (statp);
 118 }

 128 /* This is called when a thread is exiting to free resources held in _res.  */
 129 void
 130 __res_thread_freeres (void)
 131 {
 132   __resolv_context_freeres ();
 133 
 134   if (_res.nscount == 0)
 135     /* Never called res_ninit.  */
 136     return;
 137 
 138   __res_iclose (&_res, true);           /* Close any VC sockets.  */
 139 
 140   /* Make sure we do a full re-initialization the next time.  */
 141   _res.options = 0;
 142 }
 143 /* Also must be called when the main thread exits.  */
 144 text_set_element (__libc_subfreeres, __res_thread_freeres);
