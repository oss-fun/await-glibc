  27 static int
  28 do_test (void)
  29 {
  30   char buf[3];
  31   const wchar_t wc[1] = L"a";
  32   iconv_t cd;
  33   char *inptr;
  34   size_t inlen;
  35   char *outptr;
  36   size_t outlen;
  37   size_t n;
  38   int e;
  39   int result = 0;
  40 
  41   mtrace ();
  42 
  43   cd = iconv_open ("UCS4", "WCHAR_T");
  44   if (cd == (iconv_t) -1)
  45     {
  46       printf ("cannot convert from wchar_t to UCS4: %m\n");
  47       exit (1);
  48     }
  49 
  50   inptr = (char *) wc;
  51   inlen = sizeof (wchar_t);
  52   outptr = buf;
  53   outlen = 3;
  54 
  55   n = iconv (cd, &inptr, &inlen, &outptr, &outlen);
  56   e = errno;
  57 
  58   if (n != (size_t) -1)
  59     {
  60       printf ("incorrect iconv() return value: %zd, expected -1\n", n);
  61       result = 1;
  62     }
  63 
  64   if (e != E2BIG)
  65     {
  66       printf ("incorrect error value: %s, expected %s\n",
  67           strerror (e), strerror (E2BIG));
  68       result = 1;
  69     }
  70 
  71   if (inptr != (char *) wc)
  72     {
  73       puts ("inptr changed");
  74       result = 1;
  75     }
  76 
  77   if (inlen != sizeof (wchar_t))
  78     {
  79       puts ("inlen changed");
  80       result = 1;
  81     }
  82 
  83   if (outptr != buf)
  84     {
  85       puts ("outptr changed");
  86       result = 1;
  87     }
  88 
  89   if (outlen != 3)
  90     {
  91       puts ("outlen changed");
  92       result = 1;
  93     }
  94 
  95   iconv_close (cd);
  96 
  97   return result;
  98 }
