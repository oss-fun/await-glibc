  18 #ifndef _SYS_USER_H

  21 /* The whole purpose of this file is for gdb/strace and gdb/strace

  25 #include <asm/reg.h>
  26 #include <stddef.h>
  27 
  28 struct user
  29 {
  30   unsigned long int regs[EF_SIZE / 8 + 32]; /* integer and fp regs */
  31   size_t u_tsize;               /* text size (pages) */
  32   size_t u_dsize;               /* data size (pages) */
  33   size_t u_ssize;               /* stack size (pages) */
  34   unsigned long int start_code;         /* text starting address */
  35   unsigned long int start_data;         /* data starting address */
  36   unsigned long int start_stack;        /* stack starting address */
  37   long int signal;              /* signal causing core dump */
  38   struct regs *u_ar0;               /* help gdb find registers */
  39   unsigned long int magic;          /* identifies a core file */
  40   char u_comm[32];              /* user command name */
  41 };
  42 
  43 #define PAGE_SHIFT      13
  44 #define PAGE_SIZE       (1UL << PAGE_SHIFT)
  45 #define PAGE_MASK       (~(PAGE_SIZE-1))
  46 #define NBPG            PAGE_SIZE
  47 #define UPAGES          1
  48 #define HOST_TEXT_START_ADDR    (u.start_code)
  49 #define HOST_DATA_START_ADDR    (u.start_data)
  50 #define HOST_STACK_END_ADDR (u.start_stack + u.u_ssize * NBPG)
  51 
  52 #endif  /* sys/user.h */
