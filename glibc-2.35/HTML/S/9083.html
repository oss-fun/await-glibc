  17 /* asinh(x)
  18  * Method :
  19  *  Based on
  20  *      asinh(x) = sign(x) * log [ |x| + sqrt(x*x+1) ]
  21  *  we have
  22  *  asinh(x) := x  if  1+x*x=1,
  23  *       := sign(x)*(log(x)+ln2)) for large |x|, else
  24  *       := sign(x)*log(2|x|+1/(|x|+sqrt(x*x+1))) if|x|>2, else
  25  *       := sign(x)*log1p(|x| + x^2/(1 + sqrt(1+x^2)))
  26  */
  27

  39 long double __asinhl(long double x)
  40 {
  41     long double t,w;
  42     int64_t hx,ix;
  43     double xhi;
  44 
  45     xhi = ldbl_high (x);
  46     EXTRACT_WORDS64 (hx, xhi);
  47     ix = hx&0x7fffffffffffffffLL;
  48     if(ix>=0x7ff0000000000000LL) return x+x;    /* x is inf or NaN */
  49     if(ix< 0x3c70000000000000LL) {  /* |x|<2**-56 */
  50         math_check_force_underflow (x);
  51         if(huge+x>one) return x;    /* return x inexact except 0 */
  52     }
  53     if(ix>0x4370000000000000LL) {   /* |x| > 2**56 */
  54         w = __ieee754_logl(fabsl(x))+ln2;
  55     } else if (ix>0x4000000000000000LL) {   /* 2**56 >= |x| > 2.0 */
  56         t = fabs(x);
  57         w = __ieee754_logl(2.0*t+one/(sqrtl(x*x+one)+t));
  58     } else {        /* 2.0 >= |x| >= 2**-56 */
  59         t = x*x;
  60         w =__log1pl(fabsl(x)+t/(one+sqrtl(one+t)));
  61     }
  62     if(hx>0) return w; else return -w;
  63 }
