  30 int
  31 __fileno (FILE *fp)
  32 {
  33   CHECK_FILE (fp, EOF);
  34 
  35   if (!(fp->_flags & _IO_IS_FILEBUF) || _IO_fileno (fp) < 0)
  36     {
  37       __set_errno (EBADF);
  38       return -1;
  39     }
  40 
  41   return _IO_fileno (fp);
  42 }
