  19 /*
  20  * ALGORITHM DESCRIPTION:
  21  *   Typical exp10() implementation, except that:
  22  *    - tables are small (16 elements), allowing for fast gathers
  23  *    - all arguments processed in the main path
  24  *        - final VSCALEF assists branch-free design (correct overflow/underflow and special case responses)
  25  *        - a VAND is used to ensure the reduced argument |R|<2, even for large inputs
  26  *        - RZ mode used to avoid oveflow to +/-Inf for x*log2(e); helps with special case handling
  27  *        - SAE used to avoid spurious flag settings
  28  *
  29  */
  30

  38 #define L2L                             448

  48     .section .text.exex512,"ax",@progbits
  49 ENTRY(_ZGVeN16v_exp10f_skx)
  50         pushq     %rbp
  51         cfi_def_cfa_offset(16)
  52         movq      %rsp, %rbp
  53         cfi_def_cfa(6, 16)
  54         cfi_offset(6, -16)
  55         andq      $-64, %rsp
  56         subq      $192, %rsp
  57         vmovups   L2E+__svml_sexp10_data_internal_avx512(%rip), %zmm2
  58         vmovups   Shifter+__svml_sexp10_data_internal_avx512(%rip), %zmm1
  59         vmovups   L2H+__svml_sexp10_data_internal_avx512(%rip), %zmm5
  60         vmovups   L2L+__svml_sexp10_data_internal_avx512(%rip), %zmm4
  61 
  62 /* ensure |R|<2 even for special cases */
  63         vmovups   EMask+__svml_sexp10_data_internal_avx512(%rip), %zmm6
  64         vmovups   poly_coeff2+__svml_sexp10_data_internal_avx512(%rip), %zmm9
  65 
  66 /* 2^(52-4)*1.5 + x * log2(e) */
  67         vfmadd213ps {rz-sae}, %zmm1, %zmm0, %zmm2
  68         vmovups   poly_coeff1+__svml_sexp10_data_internal_avx512(%rip), %zmm10
  69         vmovups   __svml_sexp10_data_internal_avx512(%rip), %zmm8
  70         vmovups   Exp_tbl_H+__svml_sexp10_data_internal_avx512(%rip), %zmm15
  71         vmovups   Threshold+__svml_sexp10_data_internal_avx512(%rip), %zmm13
  72         vpsrld    $5, %zmm2, %zmm3
  73 
  74 /* Z0 ~ x*log2(e), rounded down to 6 fractional bits */
  75         vsubps    {rn-sae}, %zmm1, %zmm2, %zmm1
  76         vpermt2ps Exp_tbl_L+64+__svml_sexp10_data_internal_avx512(%rip), %zmm2, %zmm8
  77         vpermt2ps Exp_tbl_H+64+__svml_sexp10_data_internal_avx512(%rip), %zmm3, %zmm15
  78         vandps    AbsMask+__svml_sexp10_data_internal_avx512(%rip), %zmm0, %zmm12
  79 
  80 /* R = x - Z0*log(2) */
  81         vfnmadd213ps {rn-sae}, %zmm0, %zmm1, %zmm5
  82         vcmpps    $29, {sae}, %zmm13, %zmm12, %k0
  83         vfnmadd231ps {rn-sae}, %zmm1, %zmm4, %zmm5
  84         kmovw     %k0, %edx
  85         vrangeps  $2, {sae}, %zmm6, %zmm5, %zmm11
  86         vfmadd231ps {rn-sae}, %zmm11, %zmm9, %zmm10
  87         vmulps    {rn-sae}, %zmm11, %zmm10, %zmm14
  88 
  89 /* x!=0? */
  90         vpxord    %zmm7, %zmm7, %zmm7
  91         vcmpps    $4, {sae}, %zmm7, %zmm0, %k1
  92 
  93 /* Th*Tl */
  94         vmulps    {rn-sae}, %zmm8, %zmm15, %zmm15{%k1}
  95         vfmadd213ps {rn-sae}, %zmm15, %zmm14, %zmm15
  96         vscalefps {rn-sae}, %zmm1, %zmm15, %zmm1
  97         testl     %edx, %edx
  98 
  99 /* Go to special inputs processing branch */
 100         jne       L(SPECIAL_VALUES_BRANCH)
 101                                 # LOE rbx r12 r13 r14 r15 edx zmm0 zmm1
 102 
 103 /* Restore registers
 104  * and exit the function
 105  */
 106 
 107 L(EXIT):
 108         vmovaps   %zmm1, %zmm0
 109         movq      %rbp, %rsp
 110         popq      %rbp
 111         cfi_def_cfa(7, 8)
 112         cfi_restore(6)
 113         ret
 114         cfi_def_cfa(6, 16)
 115         cfi_offset(6, -16)
 116 
 117 /* Branch to process
 118  * special inputs
 119  */
 120 
 121 L(SPECIAL_VALUES_BRANCH):
 122         vmovups   %zmm0, 64(%rsp)
 123         vmovups   %zmm1, 128(%rsp)
 124                                 # LOE rbx r12 r13 r14 r15 edx zmm1
 125 
 126         xorl      %eax, %eax
 127                                 # LOE rbx r12 r13 r14 r15 eax edx
 128 
 129         vzeroupper
 130         movq      %r12, 16(%rsp)

 132         .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x50, 0xff, 0xff, 0xff, 0x22
 133         movl      %eax, %r12d
 134         movq      %r13, 8(%rsp)
 135         /*  DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -184; DW_OP_plus)  */
 136         .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x48, 0xff, 0xff, 0xff, 0x22
 137         movl      %edx, %r13d
 138         movq      %r14, (%rsp)
 139         /*  DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -192; DW_OP_plus)  */
 140         .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x40, 0xff, 0xff, 0xff, 0x22
 141                                 # LOE rbx r15 r12d r13d
 142 
 143 /* Range mask
 144  * bits check
 145  */
 146 
 147 L(RANGEMASK_CHECK):
 148         btl       %r12d, %r13d
 149 
 150 /* Call scalar math function */
 151         jc        L(SCALAR_MATH_CALL)
 152                                 # LOE rbx r15 r12d r13d
 153 
 154 /* Special inputs
 155  * processing loop
 156  */
 157 
 158 L(SPECIAL_VALUES_LOOP):
 159         incl      %r12d
 160         cmpl      $16, %r12d
 161 
 162 /* Check bits in range mask */
 163         jl        L(RANGEMASK_CHECK)
 164                                 # LOE rbx r15 r12d r13d
 165 
 166         movq      16(%rsp), %r12
 167         cfi_restore(12)
 168         movq      8(%rsp), %r13
 169         cfi_restore(13)
 170         movq      (%rsp), %r14
 171         cfi_restore(14)
 172         vmovups   128(%rsp), %zmm1
 173 
 174 /* Go to exit */
 175         jmp       L(EXIT)
 176         /*  DW_CFA_expression: r12 (r12) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -176; DW_OP_plus)  */
 177         .cfi_escape 0x10, 0x0c, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x50, 0xff, 0xff, 0xff, 0x22
 178         /*  DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -184; DW_OP_plus)  */
 179         .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x48, 0xff, 0xff, 0xff, 0x22
 180         /*  DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -192; DW_OP_plus)  */
 181         .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x40, 0xff, 0xff, 0xff, 0x22
 182                                 # LOE rbx r12 r13 r14 r15 zmm1
 183 
 184 /* Scalar math fucntion call
 185  * to process special input
 186  */
 187 
 188 L(SCALAR_MATH_CALL):
 189         movl      %r12d, %r14d
 190         movss     64(%rsp,%r14,4), %xmm0
 191         call      exp10f@PLT
 192                                 # LOE rbx r14 r15 r12d r13d xmm0
 193

 195 
 196 /* Process special inputs in loop */
 197         jmp       L(SPECIAL_VALUES_LOOP)
 198                                 # LOE rbx r15 r12d r13d
