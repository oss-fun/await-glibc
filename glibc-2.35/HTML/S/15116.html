  19 /*
  20  * ALGORITHM DESCRIPTION:
  21  *
  22  *   Compute sinh(x) as (exp(x)-exp(-x))/2,
  23  *   where exp is calculated as
  24  *   exp(M*ln2 + ln2*(j/2^k) + r) = 2^M * 2^(j/2^k) * exp(r)
  25  *
  26  *   Special cases:
  27  *
  28  *   sinh(NaN) = quiet NaN, and raise invalid exception
  29  *   sinh(INF) = that INF
  30  *   sinh(x)   = x for subnormals
  31  *   sinh(x) overflows for big x and returns MAXLOG+log(2)
  32  *
  33  */
  34 
  35 /* Offsets for data table __svml_dsinh_data_internal

  41 #define _dbT                            64
  42 #define _dbShifter                      2112
  43 #define _iDomainRange                   2128
  44 #define _dPC2                           2144
  45 #define _dPC3                           2160
  46 #define _dPC4                           2176
  47 #define _dPC5                           2192

  52         .text
  53     .section .text.sse4,"ax",@progbits
  54 ENTRY(_ZGVbN2v_sinh_sse4)
  55         subq      $72, %rsp
  56         cfi_def_cfa_offset(80)
  57         movaps    %xmm0, %xmm2
  58 
  59 /*  Abs argument  */
  60         movups    _dSign+__svml_dsinh_data_internal(%rip), %xmm0
  61         lea       _dbT+8+__svml_dsinh_data_internal(%rip), %rsi
  62         andps     %xmm2, %xmm0
  63         movaps    %xmm0, %xmm1
  64 
  65 /*
  66  *  Load argument
  67  * dM = x*2^K/log(2) + RShifter
  68  */
  69         movups    _dbInvLn2+__svml_dsinh_data_internal(%rip), %xmm10
  70         pxor      %xmm2, %xmm1
  71         mulpd     %xmm1, %xmm10
  72         movups    _dbShifter+__svml_dsinh_data_internal(%rip), %xmm5
  73         addpd     %xmm5, %xmm10
  74 
  75 /*
  76  *  R
  77  * dN = dM - RShifter
  78  */
  79         movaps    %xmm10, %xmm7
  80         subpd     %xmm5, %xmm7
  81 
  82 /* dR = dX - dN*Log2_hi/2^K */
  83         movups    _dbLn2hi+__svml_dsinh_data_internal(%rip), %xmm6
  84         mulpd     %xmm7, %xmm6
  85 
  86 /* dR = (dX - dN*Log2_hi/2^K) - dN*Log2_lo/2^K */
  87         movups    _dbLn2lo+__svml_dsinh_data_internal(%rip), %xmm8
  88         mulpd     %xmm7, %xmm8
  89 
  90 /*
  91  * Check for overflow\underflow
  92  *
  93  */
  94         pshufd    $221, %xmm1, %xmm4
  95         subpd     %xmm6, %xmm1
  96         subpd     %xmm8, %xmm1
  97 
  98 /* VLOAD_CONST( D, dPC[0],         TAB._dPC1 ); */
  99         movq      _iDomainRange+__svml_dsinh_data_internal(%rip), %xmm3
 100         pcmpgtd   %xmm3, %xmm4
 101 
 102 /* dR2 = dR^2 */
 103         movaps    %xmm1, %xmm3
 104         mulpd     %xmm1, %xmm3
 105         movmskps  %xmm4, %edx
 106 
 107 /*
 108  * sinh(r) = r*((a1=1)+r^2*(a3+r^2*a5)) = r + r*(r^2*(a3+r^2*a5)) ....
 109  * dSinh_r = (a3+r^2*a5)
 110  */
 111         movups    _dPC5+__svml_dsinh_data_internal(%rip), %xmm12
 112 
 113 /*
 114  * poly(r) = (dG2+dG1)+dG3*sinh(dR)+dG1*sinh(dR)+(dG1+dG2)*dR2*(a2 +a4*dR2)
 115  * dOut = (a2 +a4*dR2)
 116  */
 117         movups    _dPC4+__svml_dsinh_data_internal(%rip), %xmm13
 118         mulpd     %xmm3, %xmm12
 119         mulpd     %xmm3, %xmm13
 120         addpd     _dPC3+__svml_dsinh_data_internal(%rip), %xmm12
 121         addpd     _dPC2+__svml_dsinh_data_internal(%rip), %xmm13
 122 
 123 /* dSinh_r = r^2*(a3+r^2*a5) */
 124         mulpd     %xmm3, %xmm12
 125 
 126 /* dOut = dR2*(a2 +a4*dR2) */
 127         mulpd     %xmm13, %xmm3
 128 
 129 /* dSinh_r = r + r*(r^2*(a3+r^2*a5)) */
 130         mulpd     %xmm1, %xmm12
 131 
 132 /*
 133  *  Index and lookup
 134  * j
 135  */
 136         movups    _lIndexMask+__svml_dsinh_data_internal(%rip), %xmm9
 137         andps     %xmm10, %xmm9
 138         movd      %xmm9, %eax
 139 
 140 /* split j and N */
 141         pxor      %xmm9, %xmm10
 142 
 143 /*
 144  *  G1,G2,G3: dTdif,dTn * 2^N,2^(-N)
 145  * lM now is an EXP(2^N)
 146  */
 147         psllq     $45, %xmm10
 148 
 149 /*  */
 150         movaps    %xmm10, %xmm4
 151         pextrw    $4, %xmm9, %ecx
 152         addpd     %xmm12, %xmm1
 153         shll      $4, %eax
 154         shll      $4, %ecx
 155         movq      (%rax,%rsi), %xmm11
 156         movhpd    (%rcx,%rsi), %xmm11
 157         paddq     %xmm11, %xmm4
 158 
 159 /*  */
 160         psubq     %xmm10, %xmm11
 161 
 162 /* dG3 = dTn*2^N + dTn*2^-N */
 163         movdqa    %xmm4, %xmm14
 164         addpd     %xmm11, %xmm14
 165 
 166 /* dG2 = dTn*2^N - dTn*2^-N */
 167         subpd     %xmm11, %xmm4
 168         movq      -8(%rax,%rsi), %xmm15
 169         movhpd    -8(%rcx,%rsi), %xmm15
 170         paddq     %xmm10, %xmm15
 171 
 172 /* dG2 += dG1 */
 173         addpd     %xmm15, %xmm4
 174 
 175 /* dG1 += dG3 */
 176         addpd     %xmm14, %xmm15
 177 
 178 /* dOut = dG2*dR2*(a2 +a4*dR2) */
 179         mulpd     %xmm4, %xmm3
 180 
 181 /* dOut = dG1*sinh(dR)+dG2*dR2*(a2 +a4*dR2) */
 182         mulpd     %xmm15, %xmm1
 183         addpd     %xmm1, %xmm3
 184 
 185 /* dOut = dG2 + dG1*sinh(dR)+dG2*dR2*(a2 +a4*dR2) */
 186         addpd     %xmm3, %xmm4
 187 
 188 /*  Ret H  */
 189         orps      %xmm4, %xmm0
 190         andl      $3, %edx
 191 
 192 /* Go to special inputs processing branch */
 193         jne       L(SPECIAL_VALUES_BRANCH)
 194                                 # LOE rbx rbp r12 r13 r14 r15 edx xmm0 xmm2
 195 
 196 /* Restore registers
 197  * and exit the function
 198  */
 199 
 200 L(EXIT):
 201         addq      $72, %rsp
 202         cfi_def_cfa_offset(8)
 203         ret
 204         cfi_def_cfa_offset(80)
 205 
 206 /* Branch to process
 207  * special inputs
 208  */
 209 
 210 L(SPECIAL_VALUES_BRANCH):
 211         movups    %xmm2, 32(%rsp)
 212         movups    %xmm0, 48(%rsp)
 213                                 # LOE rbx rbp r12 r13 r14 r15 edx xmm0
 214 
 215         xorl      %eax, %eax
 216         movq      %r12, 16(%rsp)
 217         cfi_offset(12, -64)
 218         movl      %eax, %r12d
 219         movq      %r13, 8(%rsp)
 220         cfi_offset(13, -72)
 221         movl      %edx, %r13d
 222         movq      %r14, (%rsp)
 223         cfi_offset(14, -80)
 224                                 # LOE rbx rbp r15 r12d r13d
 225 
 226 /* Range mask
 227  * bits check
 228  */
 229 
 230 L(RANGEMASK_CHECK):
 231         btl       %r12d, %r13d
 232 
 233 /* Call scalar math function */
 234         jc        L(SCALAR_MATH_CALL)
 235                                 # LOE rbx rbp r15 r12d r13d
 236 
 237 /* Special inputs
 238  * processing loop
 239  */
 240 
 241 L(SPECIAL_VALUES_LOOP):
 242         incl      %r12d
 243         cmpl      $2, %r12d
 244 
 245 /* Check bits in range mask */
 246         jl        L(RANGEMASK_CHECK)
 247                                 # LOE rbx rbp r15 r12d r13d
 248 
 249         movq      16(%rsp), %r12
 250         cfi_restore(12)
 251         movq      8(%rsp), %r13
 252         cfi_restore(13)
 253         movq      (%rsp), %r14
 254         cfi_restore(14)
 255         movups    48(%rsp), %xmm0
 256 
 257 /* Go to exit */
 258         jmp       L(EXIT)
 259         cfi_offset(12, -64)
 260         cfi_offset(13, -72)
 261         cfi_offset(14, -80)
 262                                 # LOE rbx rbp r12 r13 r14 r15 xmm0
 263 
 264 /* Scalar math fucntion call
 265  * to process special input
 266  */
 267 
 268 L(SCALAR_MATH_CALL):
 269         movl      %r12d, %r14d
 270         movsd     32(%rsp,%r14,8), %xmm0
 271         call      sinh@PLT
 272                                 # LOE rbx rbp r14 r15 r12d r13d xmm0
 273 
 274         movsd     %xmm0, 48(%rsp,%r14,8)
 275 
 276 /* Process special inputs in loop */
 277         jmp       L(SPECIAL_VALUES_LOOP)
 278                                 # LOE rbx rbp r15 r12d r13d
 279 END(_ZGVbN2v_sinh_sse4)
