  73 static void
  74 init_slotinfo (void)
  75 {
  76   /* Create the slotinfo list.  Note that the type of static_slotinfo
  77      has effectively a zero-length array, so we cannot use the size of
  78      static_slotinfo to determine the array length.  */
  79   static_slotinfo.len = array_length (_dl_static_dtv);
  80   /* static_slotinfo.next = NULL; -- Already zero.  */
  81 
  82   /* The slotinfo list.  Will be extended by the code doing dynamic
  83      linking.  */
  84   GL(dl_tls_max_dtv_idx) = 1;
  85   GL(dl_tls_dtv_slotinfo_list) = &static_slotinfo;
  86 }

  88 static void
  89 init_static_tls (size_t memsz, size_t align)
  90 {
  91   /* That is the size of the TLS memory for this object.  */
  92   GL(dl_tls_static_size) = roundup (memsz + GLRO(dl_tls_static_surplus),
  93                     TCB_ALIGNMENT);
  94 #if TLS_TCB_AT_TP
  95   GL(dl_tls_static_size) += TLS_TCB_SIZE;
  96 #endif
  97   GL(dl_tls_static_used) = memsz;
  98   /* The alignment requirement for the static TLS block.  */
  99   GL(dl_tls_static_align) = align;
 100   /* Number of elements in the static TLS block.  */
 101   GL(dl_tls_static_nelem) = GL(dl_tls_max_dtv_idx);
 102 }

 104 void
 105 __libc_setup_tls (void)
 106 {
 107   void *tlsblock;
 108   size_t memsz = 0;
 109   size_t filesz = 0;
 110   void *initimage = NULL;
 111   size_t align = 0;
 112   size_t max_align = TCB_ALIGNMENT;
 113   size_t tcb_offset;
 114   const ElfW(Phdr) *phdr;
 115 
 116   struct link_map *main_map = GL(dl_ns)[LM_ID_BASE]._ns_loaded;
 117 
 118   __tls_pre_init_tp ();
 119 
 120   /* Look through the TLS segment if there is any.  */
 121   if (_dl_phdr != NULL)
 122     for (phdr = _dl_phdr; phdr < &_dl_phdr[_dl_phnum]; ++phdr)
 123       if (phdr->p_type == PT_TLS)
 124     {
 125       /* Remember the values we need.  */
 126       memsz = phdr->p_memsz;
 127       filesz = phdr->p_filesz;
 128       initimage = (void *) phdr->p_vaddr + main_map->l_addr;
 129       align = phdr->p_align;
 130       if (phdr->p_align > max_align)
 131         max_align = phdr->p_align;
 132       break;
 133     }
 134 
 135   /* Calculate the size of the static TLS surplus, with 0 auditors.  */
 136   _dl_tls_static_surplus_init (0);
 137 
 138   /* We have to set up the TCB block which also (possibly) contains
 139      'errno'.  Therefore we avoid 'malloc' which might touch 'errno'.
 140      Instead we use 'sbrk' which would only uses 'errno' if it fails.
 141      In this case we are right away out of memory and the user gets
 142      what she/he deserves.  */
 143 #if TLS_TCB_AT_TP
 144   /* Align the TCB offset to the maximum alignment, as
 145      _dl_allocate_tls_storage (in elf/dl-tls.c) does using __libc_memalign
 146      and dl_tls_static_align.  */
 147   tcb_offset = roundup (memsz + GLRO(dl_tls_static_surplus), max_align);
 148   tlsblock = __sbrk (tcb_offset + TLS_INIT_TCB_SIZE + max_align);
 149 #elif TLS_DTV_AT_TP
 150   tcb_offset = roundup (TLS_INIT_TCB_SIZE, align ?: 1);
 151   tlsblock = __sbrk (tcb_offset + memsz + max_align
 152              + TLS_PRE_TCB_SIZE + GLRO(dl_tls_static_surplus));
 153   tlsblock += TLS_PRE_TCB_SIZE;
 154 #else
 155   /* In case a model with a different layout for the TCB and DTV
 156      is defined add another #elif here and in the following #ifs.  */
 157 # error "Either TLS_TCB_AT_TP or TLS_DTV_AT_TP must be defined"
 158 #endif
 159 
 160   /* Align the TLS block.  */
 161   tlsblock = (void *) (((uintptr_t) tlsblock + max_align - 1)
 162                & ~(max_align - 1));
 163 
 164   /* Initialize the dtv.  [0] is the length, [1] the generation counter.  */
 165   _dl_static_dtv[0].counter = (sizeof (_dl_static_dtv) / sizeof (_dl_static_dtv[0])) - 2;
 166   // _dl_static_dtv[1].counter = 0;     would be needed if not already done
 167 
 168   /* Initialize the TLS block.  */
 169 #if TLS_TCB_AT_TP
 170   _dl_static_dtv[2].pointer.val = ((char *) tlsblock + tcb_offset
 171                    - roundup (memsz, align ?: 1));
 172   main_map->l_tls_offset = roundup (memsz, align ?: 1);
 173 #elif TLS_DTV_AT_TP
 174   _dl_static_dtv[2].pointer.val = (char *) tlsblock + tcb_offset;
 175   main_map->l_tls_offset = tcb_offset;
 176 #else
 177 # error "Either TLS_TCB_AT_TP or TLS_DTV_AT_TP must be defined"
 178 #endif
 179   _dl_static_dtv[2].pointer.to_free = NULL;
 180   /* sbrk gives us zero'd memory, so we don't need to clear the remainder.  */
 181   memcpy (_dl_static_dtv[2].pointer.val, initimage, filesz);
 182 
 183   /* Install the pointer to the dtv.  */
 184 
 185   /* Initialize the thread pointer.  */
 186 #if TLS_TCB_AT_TP
 187   INSTALL_DTV ((char *) tlsblock + tcb_offset, _dl_static_dtv);
 188 
 189   const char *lossage = TLS_INIT_TP ((char *) tlsblock + tcb_offset);
 190 #elif TLS_DTV_AT_TP
 191   INSTALL_DTV (tlsblock, _dl_static_dtv);
 192   const char *lossage = TLS_INIT_TP (tlsblock);
 193 #else
 194 # error "Either TLS_TCB_AT_TP or TLS_DTV_AT_TP must be defined"
 195 #endif
 196   if (__builtin_expect (lossage != NULL, 0))
 197     _startup_fatal (lossage);
 198   __tls_init_tp ();
 199 
 200   /* Update the executable's link map with enough information to make
 201      the TLS routines happy.  */
 202   main_map->l_tls_align = align;
 203   main_map->l_tls_blocksize = memsz;
 204   main_map->l_tls_initimage = initimage;
 205   main_map->l_tls_initimage_size = filesz;
 206   main_map->l_tls_modid = 1;
 207 
 208   init_slotinfo ();
 209   /* static_slotinfo.slotinfo[1].gen = 0; -- Already zero.  */
 210   static_slotinfo.slotinfo[1].map = main_map;
 211 
 212   memsz = roundup (memsz, align ?: 1);
 213 
 214 #if TLS_DTV_AT_TP
 215   memsz += tcb_offset;
 216 #endif
 217 
 218   init_static_tls (memsz, MAX (TCB_ALIGNMENT, max_align));
 219 }
