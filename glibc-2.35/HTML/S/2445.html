  28 void
  29 __explicit_bzero_chk (void *dst, size_t len, size_t dstlen)
  30 {
  31   /* Inline __memset_chk to avoid a PLT reference to __memset_chk.  */
  32   if (__glibc_unlikely (dstlen < len))
  33     __chk_fail ();
  34   memset (dst, '\0', len);
  35   /* Compiler barrier.  */
  36   asm volatile ("" ::: "memory");
  37 }
