1218 {

1222   size_t getconf_dirlen = strlen (getconf_dir);

1226   memcpy (mempcpy (mempcpy (name, getconf_dir, getconf_dirlen),
1227            "/POSIX_V6_", sizeof ("/POSIX_V6_") - 1),
1228       spec, speclen + 1);
1229
