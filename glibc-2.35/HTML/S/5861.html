  23 char *
  24 support_format_address_family (int family)
  25 {
  26   switch (family)
  27     {
  28     case AF_INET:
  29       return xstrdup ("INET");
  30     case AF_INET6:
  31       return xstrdup ("INET6");
  32     case AF_LOCAL:
  33       return xstrdup ("LOCAL");
  34     case AF_UNSPEC:
  35       return xstrdup ("UNSPEC");
  36     default:
  37       return xasprintf ("<unknown address family %d>", family);
  38     }
  39 }
