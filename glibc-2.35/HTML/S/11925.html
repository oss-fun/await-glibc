  18 #include <sysdep.h>
  19 #define _ERRNO_H    1
  20 #include <bits/errno.h>
  21 #include <bits/wordsize.h>
  22 #include <tcb-offsets.h>
  23 
  24 /* Clone the calling process, but without copying the whole address space.
  25    The calling process is suspended until the new process exits or is
  26    replaced by a call to `execve'.  Return -1 for errors, 0 to the new process,
  27    and the process ID of the new process to the old process.  */
  28 
  29 ENTRY (__libc_vfork)
  30     /* Do vfork system call.  */
  31     svc SYS_ify (vfork)
  32 
  33     /* Check for error.  */
  34     lghi    %r4,-4095
  35     clgr    %r2,%r4
  36     jgnl    SYSCALL_ERROR_LABEL
  37 
  38     /* Normal return.  */
  39     br  %r14
  40 PSEUDO_END (__libc_vfork)
  41 
  42 #if IS_IN (libc)
  43 weak_alias (__libc_vfork, vfork)
  44 strong_alias (__libc_vfork, __vfork)
  45 libc_hidden_def (__vfork)
  46 #endif
