  36 #define __arch_compare_and_exchange_val_32_acq(mem, newval, oldval)            \
  37   ({                                                                           \
  38       __typeof (*(mem)) __tmp;                                                 \
  39       __typeof (mem)  __memp = (mem);                                          \
  40       int test;                                                                \
  41       __asm __volatile (                                                       \
  42                 "   addc    r0, r0, r0;"                                       \
  43                 "1: lwx     %0, %3, r0;"                                       \
  44                 "   addic   %1, r0, 0;"                                        \
  45                 "   bnei    %1, 1b;"                                           \
  46                 "   cmp     %1, %0, %4;"                                       \
  47                 "   bnei    %1, 2f;"                                           \
  48                 "   swx     %5, %3, r0;"                                       \
  49                 "   addic   %1, r0, 0;"                                        \
  50                 "   bnei    %1, 1b;"                                           \
  51                 "2:"                                                           \
  52                     : "=&r" (__tmp),                                           \
  53                     "=&r" (test),                                              \
  54                     "=m" (*__memp)                                             \
  55                     : "r" (__memp),                                            \
  56                     "r" (oldval),                                              \
  57                     "r" (newval)                                               \
  58                     : "cc", "memory");                                         \
  59       __tmp;                                                                   \
  60   })
