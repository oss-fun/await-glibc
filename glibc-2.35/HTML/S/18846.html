  25 td_err_e
  26 td_thr_get_info (const td_thrhandle_t *th, td_thrinfo_t *infop)
  27 {
  28   td_err_e err;
  29   void *copy;
  30   psaddr_t tls, schedpolicy, schedprio, cancelhandling, tid, report_events;
  31 
  32   LOG ("td_thr_get_info");
  33 
  34   if (th->th_unique == 0)
  35     {
  36       /* Special case for the main thread before initialization.  */
  37       copy = NULL;
  38       tls = 0;
  39       cancelhandling = 0;
  40       schedpolicy = SCHED_OTHER;
  41       schedprio = 0;
  42       tid = 0;
  43 
  44       /* Ignore errors to obtain the __nptl_initial_report_events
  45      value because GDB no longer uses the events interface, and
  46      other libthread_db consumers hopefully can handle different
  47      libpthread/lds.o load orders.  */
  48       report_events = 0;
  49       (void) DB_GET_VALUE (report_events, th->th_ta_p,
  50                __nptl_initial_report_events, 0);
  51       err = TD_OK;
  52     }
  53   else
  54     {
  55       /* Copy the whole descriptor in once so we can access the several
  56      fields locally.  Excess copying in one go is much better than
  57      multiple ps_pdread calls.  */
  58       err = DB_GET_STRUCT (copy, th->th_ta_p, th->th_unique, pthread);
  59       if (err != TD_OK)
  60     return err;
  61 
  62       err = DB_GET_FIELD_ADDRESS (tls, th->th_ta_p, th->th_unique,
  63                   pthread, specific, 0);
  64       if (err != TD_OK)
  65     return err;
  66 
  67       err = DB_GET_FIELD_LOCAL (schedpolicy, th->th_ta_p, copy, pthread,
  68                 schedpolicy, 0);
  69       if (err != TD_OK)
  70     return err;
  71       err = DB_GET_FIELD_LOCAL (schedprio, th->th_ta_p, copy, pthread,
  72                 schedparam_sched_priority, 0);
  73       if (err != TD_OK)
  74     return err;
  75       err = DB_GET_FIELD_LOCAL (tid, th->th_ta_p, copy, pthread, tid, 0);
  76       if (err != TD_OK)
  77     return err;
  78       err = DB_GET_FIELD_LOCAL (cancelhandling, th->th_ta_p, copy, pthread,
  79                 cancelhandling, 0);
  80       if (err != TD_OK)
  81     return err;
  82       err = DB_GET_FIELD_LOCAL (report_events, th->th_ta_p, copy, pthread,
  83                 report_events, 0);
  84     }
  85   if (err != TD_OK)
  86     return err;
  87 
  88   /* Fill in information.  Clear first to provide reproducable
  89      results for the fields we do not fill in.  */
  90   memset (infop, '\0', sizeof (td_thrinfo_t));
  91 
  92   infop->ti_tid = (thread_t) th->th_unique;
  93   infop->ti_tls = (char *) tls;
  94   infop->ti_pri = ((uintptr_t) schedpolicy == SCHED_OTHER
  95            ? 0 : (uintptr_t) schedprio);
  96   infop->ti_type = TD_THR_USER;
  97 
  98   if ((((int) (uintptr_t) cancelhandling) & EXITING_BITMASK) == 0)
  99     /* XXX For now there is no way to get more information.  */
 100     infop->ti_state = TD_THR_ACTIVE;
 101   else if ((((int) (uintptr_t) cancelhandling) & TERMINATED_BITMASK) == 0)
 102     infop->ti_state = TD_THR_ZOMBIE;
 103   else
 104     infop->ti_state = TD_THR_UNKNOWN;
 105 
 106   /* Initialization which are the same in both cases.  */
 107   infop->ti_ta_p = th->th_ta_p;
 108   infop->ti_lid = tid == 0 ? ps_getpid (th->th_ta_p->ph) : (uintptr_t) tid;
 109   infop->ti_traceme = report_events != 0;
 110 
 111   if (copy != NULL)
 112     err = DB_GET_FIELD_LOCAL (infop->ti_startfunc, th->th_ta_p, copy, pthread,
 113                   start_routine, 0);
 114   if (copy != NULL && err == TD_OK)
 115     {
 116       uint32_t idx;
 117       for (idx = 0; idx < TD_EVENTSIZE; ++idx)
 118     {
 119       psaddr_t word;
 120       err = DB_GET_FIELD_LOCAL (word, th->th_ta_p, copy, pthread,
 121                     eventbuf_eventmask_event_bits, idx);
 122       if (err != TD_OK)
 123         break;
 124       infop->ti_events.event_bits[idx] = (uintptr_t) word;
 125     }
 126       if (err == TD_NOAPLIC)
 127     memset (&infop->ti_events.event_bits[idx], 0,
 128         (TD_EVENTSIZE - idx) * sizeof infop->ti_events.event_bits[0]);
 129     }
 130 
 131   return err;
 132 }
