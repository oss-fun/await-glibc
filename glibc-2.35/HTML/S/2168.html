  24 /* Change the current root directory to FD.  */
  25 int
  26 fchroot (int fd)
  27 {
  28   error_t err;
  29   file_t dir;
  30 
  31   err = HURD_DPORT_USE (fd,
  32             ({
  33               dir = __file_name_lookup_under (port, ".", 0, 0);
  34               dir == MACH_PORT_NULL ? errno : 0;
  35             }));
  36 
  37   if (! err)
  38     {
  39       file_t root;
  40 
  41       /* Prevent going through DIR's ..  */
  42       err = __file_reparent (dir, MACH_PORT_NULL, &root);
  43       __mach_port_deallocate (__mach_task_self (), dir);
  44       if (! err)
  45     _hurd_port_set (&_hurd_ports[INIT_PORT_CRDIR], root);
  46     }
  47 
  48   return err ? __hurd_fail (err) : 0;
  49 }
