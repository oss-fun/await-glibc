  19 /*
  20  * ALGORITHM DESCRIPTION:
  21  *
  22  *    Get short reciprocal approximation Rcp ~ 1/mantissa(x)
  23  *    R = Rcp*x - 1.0
  24  *    log2(x) = k - log2(Rcp) + poly_approximation(R)
  25  *       log2(Rcp) is tabulated
  26  *
  27  *
  28  */
  29 
  30 /* Offsets for data table __svml_slog2_data_internal
  31  */
  32 #define MinNorm                         0
  33 #define MaxNorm                         32
  34 #define iBrkValue                       64
  35 #define iOffExpoMask                    96
  36 #define One                             128
  37 #define sPoly                           160

  39 #include <sysdep.h>

  41         .text
  42     .section .text.avx2,"ax",@progbits
  43 ENTRY(_ZGVdN8v_log2f_avx2)
  44         pushq     %rbp
  45         cfi_def_cfa_offset(16)
  46         movq      %rsp, %rbp
  47         cfi_def_cfa(6, 16)
  48         cfi_offset(6, -16)
  49         andq      $-32, %rsp
  50         subq      $96, %rsp
  51 
  52 /* reduction: compute r,n */
  53         vmovups   iBrkValue+__svml_slog2_data_internal(%rip), %ymm4
  54         vmovups   sPoly+64+__svml_slog2_data_internal(%rip), %ymm9
  55         vmovups   sPoly+128+__svml_slog2_data_internal(%rip), %ymm10
  56         vmovups   sPoly+192+__svml_slog2_data_internal(%rip), %ymm12
  57         vpsubd    %ymm4, %ymm0, %ymm1
  58         vcmplt_oqps MinNorm+__svml_slog2_data_internal(%rip), %ymm0, %ymm5
  59         vcmpnle_uqps MaxNorm+__svml_slog2_data_internal(%rip), %ymm0, %ymm6
  60         vpand     iOffExpoMask+__svml_slog2_data_internal(%rip), %ymm1, %ymm3
  61         vpsrad    $23, %ymm1, %ymm2
  62         vmovups   sPoly+__svml_slog2_data_internal(%rip), %ymm1
  63         vpaddd    %ymm4, %ymm3, %ymm8
  64         vcvtdq2ps %ymm2, %ymm14
  65         vsubps    One+__svml_slog2_data_internal(%rip), %ymm8, %ymm13
  66         vfmadd213ps sPoly+32+__svml_slog2_data_internal(%rip), %ymm13, %ymm1
  67         vfmadd213ps sPoly+96+__svml_slog2_data_internal(%rip), %ymm13, %ymm9
  68         vmulps    %ymm13, %ymm13, %ymm11
  69         vfmadd213ps sPoly+160+__svml_slog2_data_internal(%rip), %ymm13, %ymm10
  70         vfmadd213ps sPoly+224+__svml_slog2_data_internal(%rip), %ymm13, %ymm12
  71         vfmadd213ps %ymm9, %ymm11, %ymm1
  72         vfmadd213ps %ymm10, %ymm11, %ymm1
  73         vfmadd213ps %ymm12, %ymm11, %ymm1
  74         vfmadd213ps sPoly+256+__svml_slog2_data_internal(%rip), %ymm13, %ymm1
  75         vorps     %ymm6, %ymm5, %ymm7
  76 
  77 /* combine and get argument value range mask */
  78         vmovmskps %ymm7, %edx
  79         vfmadd213ps %ymm14, %ymm13, %ymm1
  80         testl     %edx, %edx
  81 
  82 /* Go to special inputs processing branch */
  83         jne       L(SPECIAL_VALUES_BRANCH)
  84                                 # LOE rbx r12 r13 r14 r15 edx ymm0 ymm1
  85 
  86 /* Restore registers
  87  * and exit the function
  88  */
  89 
  90 L(EXIT):
  91         vmovaps   %ymm1, %ymm0
  92         movq      %rbp, %rsp
  93         popq      %rbp
  94         cfi_def_cfa(7, 8)
  95         cfi_restore(6)
  96         ret
  97         cfi_def_cfa(6, 16)
  98         cfi_offset(6, -16)

 119         .cfi_escape 0x10, 0x0d, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0xa8, 0xff, 0xff, 0xff, 0x22

 123         .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xe0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0xa0, 0xff, 0xff, 0xff, 0x22
 124                                 # LOE rbx r15 r12d r13d

 129

 132 
 133 /* Call scalar math function */

 138  * processing loop

 143         cmpl      $8, %r12d

 147                                 # LOE rbx r15 r12d r13d

 165                                 # LOE rbx r12 r13 r14 r15 ymm1
