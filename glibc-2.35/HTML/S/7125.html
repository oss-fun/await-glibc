  22 int
  23 check_supports (int supports, int active, const char *supports_name,
  24         const char *name)
  25 {
  26   printf ("Checking %s:\n", name);
  27   printf ("  %s: %d\n", name, active);
  28   printf ("  __builtin_cpu_supports (%s): %d\n",
  29       supports_name, supports);
  30 
  31   if ((supports != 0) != (active != 0))
  32     {
  33       printf (" *** failure ***\n");
  34       return 1;
  35     }
  36 
  37   return 0;
  38 }
