  27 float __ieee754_acoshf(float x)
  28 {
  29     float t;
  30     int32_t hx;
  31     GET_FLOAT_WORD(hx,x);
  32     if(hx<0x3f800000) {     /* x < 1 */
  33         return (x-x)/(x-x);
  34     } else if(hx >=0x4d800000) {    /* x > 2**28 */
  35         if(hx >=0x7f800000) {   /* x is inf of NaN */
  36         return x+x;
  37         } else
  38         return __ieee754_logf(x)+ln2;   /* acosh(huge)=log(2x) */
  39     } else if (hx==0x3f800000) {
  40         return 0.0;         /* acosh(1) = 0 */
  41     } else if (hx > 0x40000000) {   /* 2**28 > x > 2 */
  42         t=x*x;
  43         return __ieee754_logf((float)2.0*x-one/(x+sqrtf(t-one)));
  44     } else {            /* 1<x<2 */
  45         t = x-one;
  46         return __log1pf(t+sqrtf((float)2.0*t+t*t));
  47     }
  48 }
