  27 struct a64l_test

  33 static const struct a64l_test tests[] =

  37     { "/", 1 },
  38     { "FT", 2001 },

  48 int
  49 do_test (int argc, char ** argv)
  50 {
  51   const struct a64l_test *at;
  52   long int l;
  53   const char *s;
  54   int status = 0;
  55 
  56   for (at = tests; at->base64 != NULL; ++at)
  57     {
  58       printf ("a64l (\"%s\")", at->base64);
  59       l = a64l (at->base64);
  60       if (l == at->value)
  61     puts ("\tOK");
  62       else
  63     {
  64       printf ("\tBAD\n  returns %ld, expected %ld\n", l, at->value);
  65       status = 1;
  66     }
  67       printf ("l64a (%ld)", at->value);
  68       s = l64a (at->value);
  69       if (strcmp (s, at->base64) == 0)
  70     puts ("\tOK");
  71       else
  72     {
  73       printf ("\tBAD\n  returns \"%s\", expected \"%s\"\n", s, at->base64);
  74       status = 1;
  75     }
  76     }
  77 
  78   return status ? EXIT_FAILURE : EXIT_SUCCESS;
  79 }
