  24 static int
  25 do_test (void)
  26 {
  27   int fd = xsocket (AF_INET, SOCK_STREAM, IPPROTO_TCP);
  28 
  29   struct linger value = { -1, -1 };
  30   socklen_t optlen = sizeof (value);
  31   TEST_COMPARE (getsockopt (fd, SOL_SOCKET, SO_LINGER, &value, &optlen), 0);
  32   TEST_COMPARE (optlen, sizeof (value));
  33   TEST_COMPARE (value.l_onoff, 0);
  34   TEST_COMPARE (value.l_linger, 0);
  35 
  36   value.l_onoff = 1;
  37   value.l_linger = 30;
  38   TEST_COMPARE (setsockopt (fd, SOL_SOCKET, SO_LINGER, &value, sizeof (value)),
  39                 0);
  40 
  41   value.l_onoff = -1;
  42   value.l_linger = -1;
  43   TEST_COMPARE (getsockopt (fd, SOL_SOCKET, SO_LINGER, &value, &optlen), 0);
  44   TEST_COMPARE (optlen, sizeof (value));
  45   TEST_COMPARE (value.l_onoff, 1);
  46   TEST_COMPARE (value.l_linger, 30);
  47 
  48   xclose (fd);
  49   return 0;
  50 }
