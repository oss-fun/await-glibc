 265 /* Open FILE_NAME and return a Hurd I/O for it in *PORT, or return an
 266    error.  If STAT is non-zero, stat the file into that stat buffer.  */
 267 static error_t
 268 open_file (const char *file_name, int flags,
 269        mach_port_t *port, struct stat64 *stat)
 270 {
 271   enum retry_type doretry;
 272   char retryname[1024];     /* XXX string_t LOSES! */
 273   file_t startdir;
 274   error_t err;
 275 
 276   error_t use_init_port (int which, error_t (*operate) (file_t))
 277     {
 278       return (which < _dl_hurd_data->portarraysize
 279           ? ((*operate) (_dl_hurd_data->portarray[which]))
 280           : EGRATUITOUS);
 281     }
 282   file_t get_dtable_port (int fd)
 283     {
 284       if ((unsigned int) fd < _dl_hurd_data->dtablesize
 285       && _dl_hurd_data->dtable[fd] != MACH_PORT_NULL)
 286     {
 287       __mach_port_mod_refs (__mach_task_self (), _dl_hurd_data->dtable[fd],
 288                 MACH_PORT_RIGHT_SEND, +1);
 289       return _dl_hurd_data->dtable[fd];
 290     }
 291       errno = EBADF;
 292       return MACH_PORT_NULL;
 293     }
 294 
 295   assert (!(flags & ~(O_READ | O_EXEC | O_CLOEXEC)));
 296   flags &= ~O_CLOEXEC;
 297 
 298   startdir = _dl_hurd_data->portarray[file_name[0] == '/'
 299                       ? INIT_PORT_CRDIR : INIT_PORT_CWDIR];
 300 
 301   while (file_name[0] == '/')
 302     file_name++;
 303 
 304   err = __dir_lookup (startdir, (char *)file_name, flags, 0,
 305               &doretry, retryname, port);
 306 
 307   if (!err)
 308     err = __hurd_file_name_lookup_retry (use_init_port, get_dtable_port,
 309                      __dir_lookup, doretry, retryname,
 310                      O_RDONLY, 0, port);
 311   if (!err && stat)
 312     {
 313       err = __io_stat (*port, stat);
 314       if (err)
 315     __mach_port_deallocate (__mach_task_self (), *port);
 316     }
 317 
 318   return err;
 319 }

 451 check_no_hidden(__mmap);
 452 void *weak_function
 453 __mmap (void *addr, size_t len, int prot, int flags, int fd, off_t offset)
 454 {
 455   error_t err;
 456   vm_prot_t vmprot;
 457   vm_address_t mapaddr;
 458   mach_port_t memobj_rd, memobj_wr;
 459 
 460   vmprot = VM_PROT_NONE;
 461   if (prot & PROT_READ)
 462     vmprot |= VM_PROT_READ;
 463   if (prot & PROT_WRITE)
 464     vmprot |= VM_PROT_WRITE;
 465   if (prot & PROT_EXEC)
 466     vmprot |= VM_PROT_EXECUTE;
 467 
 468   if (flags & MAP_ANON)
 469     memobj_rd = MACH_PORT_NULL;
 470   else
 471     {
 472       assert (!(flags & MAP_SHARED));
 473       err = __io_map ((mach_port_t) fd, &memobj_rd, &memobj_wr);
 474       if (err)
 475     return __hurd_fail (err), MAP_FAILED;
 476       if (memobj_wr != MACH_PORT_NULL)
 477     __mach_port_deallocate (__mach_task_self (), memobj_wr);
 478     }
 479 
 480   mapaddr = (vm_address_t) addr;
 481   err = __vm_map (__mach_task_self (),
 482           &mapaddr, (vm_size_t) len, 0,
 483           !(flags & MAP_FIXED),
 484           memobj_rd,
 485           (vm_offset_t) offset,
 486           flags & (MAP_COPY|MAP_PRIVATE),
 487           vmprot, VM_PROT_ALL,
 488           (flags & MAP_SHARED) ? VM_INHERIT_SHARE : VM_INHERIT_COPY);
 489   if (err == KERN_NO_SPACE && (flags & MAP_FIXED))
 490     {
 491       /* XXX this is not atomic as it is in unix! */
 492       /* The region is already allocated; deallocate it first.  */
 493       err = __vm_deallocate (__mach_task_self (), mapaddr, len);
 494       if (! err)
 495     err = __vm_map (__mach_task_self (),
 496             &mapaddr, (vm_size_t) len,
 497             0,
 498             !(flags & MAP_FIXED),
 499             memobj_rd, (vm_offset_t) offset,
 500             flags & (MAP_COPY|MAP_PRIVATE),
 501             vmprot, VM_PROT_ALL,
 502             (flags & MAP_SHARED)
 503             ? VM_INHERIT_SHARE : VM_INHERIT_COPY);
 504     }
 505 
 506   if ((flags & MAP_ANON) == 0)
 507     __mach_port_deallocate (__mach_task_self (), memobj_rd);
 508 
 509   if (err)
 510     return __hurd_fail (err), MAP_FAILED;
 511   return (void *) mapaddr;
 512 }

 563 int
 564 __rtld_execve (const char *file_name, char *const argv[],
 565                char *const envp[])
 566 {
 567   file_t file;
 568   error_t err;
 569   char *args, *env;
 570   size_t argslen, envlen;
 571   mach_port_t *ports = _dl_hurd_data->portarray;
 572   unsigned int portarraysize = _dl_hurd_data->portarraysize;
 573   file_t *dtable = _dl_hurd_data->dtable;
 574   unsigned int dtablesize = _dl_hurd_data->dtablesize;
 575   int *intarray = _dl_hurd_data->intarray;
 576   unsigned int i, j;
 577   mach_port_t *please_dealloc, *pdp;
 578   mach_port_t *portnames = NULL;
 579   mach_msg_type_number_t nportnames = 0;
 580   mach_port_type_t *porttypes = NULL;
 581   mach_msg_type_number_t nporttypes = 0;
 582   int flags;
 583 
 584   err = open_file (file_name, O_EXEC, &file, NULL);
 585   if (err)
 586     goto out;
 587 
 588   if (argv == NULL)
 589     args = NULL, argslen = 0;
 590   else if (err = __argz_create (argv, &args, &argslen))
 591     goto outfile;
 592   if (envp == NULL)
 593     env = NULL, envlen = 0;
 594   else if (err = __argz_create (envp, &env, &envlen))
 595     goto outargs;
 596 
 597   please_dealloc = __alloca ((portarraysize + dtablesize)
 598                  * sizeof (mach_port_t));
 599   pdp = please_dealloc;
 600 
 601   /* Get all ports that we may not know about and we should thus destroy.  */
 602   err = __mach_port_names (__mach_task_self (),
 603                &portnames, &nportnames,
 604                &porttypes, &nporttypes);
 605   if (err)
 606     goto outenv;
 607   if (nportnames != nporttypes)
 608     {
 609       err = EGRATUITOUS;
 610       goto outenv;
 611     }
 612 
 613   for (i = 0; i < portarraysize; ++i)
 614     if (ports[i] != MACH_PORT_NULL)
 615       {
 616     *pdp++ = ports[i];
 617     for (j = 0; j < nportnames; j++)
 618       if (portnames[j] == ports[i])
 619         portnames[j] = MACH_PORT_NULL;
 620       }
 621   for (i = 0; i < dtablesize; ++i)
 622     if (dtable[i] != MACH_PORT_NULL)
 623       {
 624     *pdp++ = dtable[i];
 625     for (j = 0; j < nportnames; j++)
 626       if (portnames[j] == dtable[i])
 627         portnames[j] = MACH_PORT_NULL;
 628       }
 629 
 630   /* Pack ports to be destroyed together.  */
 631   for (i = 0, j = 0; i < nportnames; i++)
 632     {
 633       if (portnames[i] == MACH_PORT_NULL)
 634     continue;
 635       if (j != i)
 636     portnames[j] = portnames[i];
 637       j++;
 638     }
 639   nportnames = j;
 640 
 641   flags = 0;
 642 #ifdef EXEC_SIGTRAP
 643   if (__sigismember (&intarray[INIT_TRACEMASK], SIGKILL))
 644     flags |= EXEC_SIGTRAP;
 645 #endif
 646 
 647   err = __file_exec_paths (file, __mach_task_self (), flags,
 648                file_name, file_name[0] == '/' ? file_name : "",
 649                args, argslen,
 650                env, envlen,
 651                dtable, MACH_MSG_TYPE_COPY_SEND, dtablesize,
 652                ports, MACH_MSG_TYPE_COPY_SEND, portarraysize,
 653                intarray, INIT_INT_MAX,
 654                please_dealloc, pdp - please_dealloc,
 655                portnames, nportnames);
 656 
 657   /* Oh well.  Might as well be tidy.  */
 658 outenv:
 659   free (env);
 660 outargs:
 661   free (args);
 662 outfile:
 663   __mach_port_deallocate (__mach_task_self (), file);
 664 out:
 665   return err;
 666 }
