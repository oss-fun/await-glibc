  20 long double
  21 __frexpl (long double value, int *expptr)
  22 {
  23   long double mantissa, exponent;
  24   int iexponent;
  25   unsigned long fpsr;
  26 
  27   __asm ("ftst%.x %1\n"
  28      "fmove%.l %/fpsr, %0"
  29      : "=dm" (fpsr) : "f" (value));
  30   if (fpsr & (7 << 24))
  31     {
  32       /* Not finite or zero.  */
  33       *expptr = 0;
  34       return value;
  35     }
  36   __asm ("fgetexp%.x %1, %0" : "=f" (exponent) : "f" (value));
  37   iexponent = (int) exponent + 1;
  38   *expptr = iexponent;
  39   /* Unnormalized numbers must be handled specially, otherwise fscale
  40      results in overflow.  */
  41   if (iexponent <= -16384)
  42     {
  43       value *= 0x1p16383L;
  44       iexponent += 16383;
  45     }
  46   else if (iexponent >= 16384)
  47     {
  48       value *= 0x1p-16383L;
  49       iexponent -= 16383;
  50     }
  51 
  52   __asm ("fscale%.l %2, %0"
  53      : "=f" (mantissa)
  54      : "0" (value), "dmi" (-iexponent));
  55   return mantissa;
  56 }
