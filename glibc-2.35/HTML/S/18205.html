  26 /* XDR 64bit integers */
  27 bool_t
  28 xdr_int64_t (XDR *xdrs, int64_t *ip)
  29 {
  30   int32_t t1, t2;
  31 
  32   switch (xdrs->x_op)
  33     {
  34     case XDR_ENCODE:
  35       t1 = (int32_t) ((*ip) >> 32);
  36       t2 = (int32_t) (*ip);
  37       return (XDR_PUTINT32(xdrs, &t1) && XDR_PUTINT32(xdrs, &t2));
  38     case XDR_DECODE:
  39       if (!XDR_GETINT32(xdrs, &t1) || !XDR_GETINT32(xdrs, &t2))
  40     return FALSE;
  41       *ip = ((int64_t) t1) << 32;
  42       *ip |= (uint32_t) t2; /* Avoid sign extension.  */
  43       return TRUE;
  44     case XDR_FREE:
  45       return TRUE;
  46     default:
  47       return FALSE;
  48     }
  49 }
