  38 static inline Elf32_Addr *
  39 or1k_get_got (void)
  40 {
  41   Elf32_Addr *got;
  42 
  43   asm ("l.jal    0x8\n"
  44        " l.movhi %0, gotpchi(_GLOBAL_OFFSET_TABLE_-4)\n"
  45        "l.ori    %0, %0, gotpclo(_GLOBAL_OFFSET_TABLE_+0)\n"
  46        "l.add    %0, %0, r9\n"
  47        : "=r" (got) : : "r9");
  48 
  49   return got;
  50 }

  62 /* Return the run-time load address of the shared object.  */
  63 static inline Elf32_Addr
  64 elf_machine_load_address (void)
  65 {
  66   /* Compute the difference between the runtime address of _DYNAMIC as seen
  67      by a GOTOFF reference, and the link-time address found in the special
  68      unrelocated first GOT entry.  */
  69   Elf32_Addr dyn;
  70   Elf32_Addr *got = or1k_get_got ();
  71 
  72   asm ("l.movhi %0, gotoffhi(_DYNAMIC);"
  73        "l.ori   %0, %0, gotofflo(_DYNAMIC);"
  74        "l.add   %0, %0, %1;"
  75        : "=&r"(dyn) : "r"(got));
  76 
  77   return dyn - *got;
  78 }

 107 /* Set up the loaded object described by L so its unrelocated PLT
 108    entries will jump to the on-demand fixup code in dl-runtime.c.  */
 109 static inline int __attribute__ ((unused, always_inline))
 110 elf_machine_runtime_setup (struct link_map *l, struct r_scope_elem *scope[],
 111                int lazy, int profile)
 112 {
 113   ElfW(Addr) *pltgot;
 114   extern void _dl_runtime_resolve (ElfW(Word));
 115   extern void _dl_runtime_profile (ElfW(Word));
 116 
 117   if (l->l_info[DT_JMPREL] && lazy)
 118     {
 119       pltgot = (ElfW(Addr) *) D_PTR (l, l_info[DT_PLTGOT]);
 120 
 121       /* Fill in initial entrys of the plt */
 122 
 123       /* Register the link_map address in the plt at pltgot[1].
 124      This will also be used in the resolver for accessing the
 125      link_map structure.  */
 126       pltgot[1] = (ElfW(Addr)) l;
 127 
 128       /* This function will get called to fix up the GOT entry and
 129      then jump to the resolved address.  */
 130       pltgot[2] = (ElfW(Addr)) &_dl_runtime_resolve;
 131 
 132     }
 133 
 134   return lazy;
 135 }

 176    MAP is the object containing the reloc.  */

 178 static inline void
 179 __attribute ((always_inline))
 180 elf_machine_rela (struct link_map *map, struct r_scope_elem *scope[],
 181           const ElfW(Rela) *reloc, const ElfW(Sym) *sym,
 182           const struct r_found_version *version,
 183           void *const reloc_addr_arg, int skip_ifunc)
 184 {
 185   Elf32_Addr *const reloc_addr = reloc_addr_arg;
 186   const unsigned int r_type = ELF32_R_TYPE (reloc->r_info);
 187 
 188   if (__glibc_unlikely (r_type == R_OR1K_NONE))
 189     return;
 190   else
 191     {
 192 # ifndef RESOLVE_CONFLICT_FIND_MAP
 193       const Elf32_Sym *const refsym = sym;
 194 # endif
 195       struct link_map *sym_map = RESOLVE_MAP (map, scope, &sym, version,
 196                           r_type);
 197       Elf32_Addr value = SYMBOL_ADDRESS (sym_map, sym, true);
 198 
 199       if (sym != NULL
 200       && __glibc_unlikely (ELFW(ST_TYPE) (sym->st_info) == STT_GNU_IFUNC)
 201       && __glibc_likely (sym->st_shndx != SHN_UNDEF)
 202       && __glibc_likely (!skip_ifunc))
 203     value = elf_ifunc_invoke (value);
 204 
 205       switch (r_type)
 206     {
 207 # ifndef RESOLVE_CONFLICT_FIND_MAP
 208       case R_OR1K_COPY:
 209         if (sym == NULL)
 210           /* This can happen in trace mode if an object could not be
 211          found.  */
 212           break;
 213         if (__glibc_unlikely (sym->st_size > refsym->st_size)
 214         || (__glibc_unlikely (sym->st_size < refsym->st_size)
 215           && GLRO(dl_verbose)))
 216         {
 217           const char *strtab;
 218 
 219           strtab = (const char *) D_PTR (map, l_info[DT_STRTAB]);
 220           _dl_error_printf ("\
 221 %s: Symbol `%s' has different size in shared object, consider re-linking\n",
 222           rtld_progname ?: "<program name unknown>",
 223           strtab + refsym->st_name);
 224         }
 225         memcpy (reloc_addr_arg, (void *) value,
 226         MIN (sym->st_size, refsym->st_size));
 227         break;
 228 # endif /* !RESOLVE_CONFLICT_FIND_MAP */
 229       case R_OR1K_32:
 230         /* Support relocations on mis-aligned offsets.  */
 231         value += reloc->r_addend;
 232         memcpy (reloc_addr_arg, &value, 4);
 233         break;
 234       case R_OR1K_GLOB_DAT:
 235       case R_OR1K_JMP_SLOT:
 236         *reloc_addr = value + reloc->r_addend;
 237         break;
 238       case R_OR1K_TLS_DTPMOD:
 239 # ifdef RTLD_BOOTSTRAP
 240         /* During startup the dynamic linker is always the module
 241            with index 1.  */
 242         *reloc_addr = 1;
 243 # else
 244         if (sym_map != NULL)
 245           *reloc_addr = sym_map->l_tls_modid;
 246 # endif
 247         break;
 248       case R_OR1K_TLS_DTPOFF:
 249 # ifndef RTLD_BOOTSTRAP
 250         *reloc_addr = (sym == NULL ? 0 : sym->st_value) + reloc->r_addend;
 251 # endif
 252         break;
 253 
 254       case R_OR1K_TLS_TPOFF:
 255 # ifdef RTLD_BOOTSTRAP
 256         *reloc_addr = sym->st_value + reloc->r_addend +
 257           map->l_tls_offset - TLS_TCB_SIZE;
 258 # else
 259         if (sym_map != NULL)
 260           {
 261         CHECK_STATIC_TLS (map, sym_map);
 262         *reloc_addr = sym->st_value + reloc->r_addend +
 263           sym_map->l_tls_offset - TLS_TCB_SIZE;
 264           }
 265 # endif
 266         break;
 267       default:
 268         _dl_reloc_bad_type (map, r_type, 0);
 269         break;
 270     }
 271     }
 272 }

 283 static inline void
 284 __attribute__ ((always_inline))
 285 elf_machine_lazy_rel (struct link_map *map, struct r_scope_elem *scope[],
 286               ElfW(Addr) l_addr, const ElfW(Rela) *reloc,
 287               int skip_ifunc)
 288 {
 289   Elf32_Addr *const reloc_addr = (void *) (l_addr + reloc->r_offset);
 290   const unsigned int r_type = ELF32_R_TYPE (reloc->r_info);
 291 
 292   if (__glibc_likely (r_type == R_OR1K_JMP_SLOT))
 293       *reloc_addr += l_addr;
 294   else if (__glibc_unlikely (r_type == R_OR1K_NONE))
 295     return;
 296   else
 297     _dl_reloc_bad_type (map, r_type, 1);
 298 }
