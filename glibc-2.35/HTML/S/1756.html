  89 static int
  90 do_test (void)
  91 {
  92   struct passwd *pw;
  93   struct group *gr;
  94   struct hostent *he;
  95   char buf1[PATH_MAX];
  96   char buf2[PATH_MAX];
  97 
  98   sprintf (buf1, "/subdir%s", support_slibdir_prefix);
  99   xmkdirp (buf1, 0777);
 100 
 101   /* Copy this DSO into the chroot so it *could* be loaded.  */
 102   sprintf (buf1, "%s/libnss_files.so.2", support_slibdir_prefix);
 103   sprintf (buf2, "/subdir%s/libnss_files.so.2", support_slibdir_prefix);
 104   support_copy_file (buf1, buf2);
 105 
 106   /* Check we're using the "outer" nsswitch.conf.  */
 107 
 108   /* This uses the test1 DSO.  */
 109   pw = getpwnam ("test1");
 110   TEST_VERIFY (pw != NULL);
 111   if (pw)
 112     TEST_COMPARE (pw->pw_uid, 1234);
 113 
 114   /* This just loads the test2 DSO.  */
 115   gr = getgrgid (5);
 116   TEST_VERIFY (gr != NULL);
 117 
 118 
 119   /* Change the root dir.  */
 120 
 121   TEST_VERIFY (chroot ("/subdir") == 0);
 122   chdir ("/");
 123 
 124   /* Check we're NOT using the "inner" nsswitch.conf.  */
 125 
 126   /* Both DSOs are loaded, which is used?  */
 127   pw = getpwnam ("test2");
 128   TEST_VERIFY (pw != NULL);
 129   if (pw)
 130     TEST_VERIFY (pw->pw_uid != 2468);
 131 
 132   /* We should still be using the old configuration.  */
 133   pw = getpwnam ("test1");
 134   TEST_VERIFY (pw != NULL);
 135   if (pw)
 136     TEST_COMPARE (pw->pw_uid, 1234);
 137   gr = getgrgid (5);
 138   TEST_VERIFY (gr != NULL);
 139   gr = getgrnam ("name4");
 140   TEST_VERIFY (gr == NULL);
 141 
 142   /* hosts in the outer nsswitch is files; the inner one is test1.
 143      Verify that we're still using the outer nsswitch *and* that we
 144      can load the files DSO. */
 145   he = gethostbyname ("test2");
 146   TEST_VERIFY (he != NULL);
 147 
 148   return 0;
 149 }
