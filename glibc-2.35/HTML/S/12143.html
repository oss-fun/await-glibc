  32 #ifndef _NETINET_TCP_H
  33 
  34 #define _NETINET_TCP_H  1
  35 #include <sys/cdefs.h>
  36 
  37 
  38 __BEGIN_DECLS
  39 
  40 typedef unsigned int    tcp_seq;
  41 /*
  42  * TCP header.
  43  * Per RFC 793, September, 1981.
  44  */
  45 struct tcphdr {
  46     unsigned short  th_sport;   /* source port */
  47     unsigned short  th_dport;   /* destination port */
  48     tcp_seq     th_seq;     /* sequence number */
  49     tcp_seq     th_ack;     /* acknowledgement number */
  50 #if __BYTE_ORDER == __LITTLE_ENDIAN
  51     unsigned char   th_x2:4,    /* (unused) */
  52             th_off:4;   /* data offset */
  53 #endif
  54 #if __BYTE_ORDER == __BIG_ENDIAN
  55     unsigned char   th_off:4,   /* data offset */
  56             th_x2:4;    /* (unused) */
  57 #endif
  58     unsigned char   th_flags;
  59 #define TH_FIN  0x01
  60 #define TH_SYN  0x02
  61 #define TH_RST  0x04
  62 #define TH_PUSH 0x08
  63 #define TH_ACK  0x10
  64 #define TH_URG  0x20
  65     unsigned short  th_win;     /* window */
  66     unsigned short  th_sum;     /* checksum */
  67     unsigned short  th_urp;     /* urgent pointer */
  68 };
  69 
  70 #define TCPOPT_EOL      0
  71 #define TCPOPT_NOP      1
  72 #define TCPOPT_MAXSEG       2
  73 #define TCPOLEN_MAXSEG      4
  74 #define TCPOPT_WINDOW       3
  75 #define TCPOLEN_WINDOW      3
  76 #define TCPOPT_SACK_PERMITTED   4       /* Experimental */
  77 #define TCPOLEN_SACK_PERMITTED  2
  78 #define TCPOPT_SACK     5       /* Experimental */
  79 #define TCPOPT_TIMESTAMP    8
  80 #define TCPOLEN_TIMESTAMP   10
  81 #define TCPOLEN_TSTAMP_APPA (TCPOLEN_TIMESTAMP+2) /* appendix A */
  82 
  83 #define TCPOPT_TSTAMP_HDR   \
  84     (TCPOPT_NOP<<24|TCPOPT_NOP<<16|TCPOPT_TIMESTAMP<<8|TCPOLEN_TIMESTAMP)
  85 
  86 /*
  87  * Default maximum segment size for TCP.
  88  * With an IP MSS of 576, this is 536,
  89  * but 512 is probably more convenient.
  90  * This should be defined as MIN(512, IP_MSS - sizeof (struct tcpiphdr)).
  91  */
  92 #define TCP_MSS 512
  93 
  94 #define TCP_MAXWIN  65535   /* largest value for (unscaled) window */
  95 
  96 #define TCP_MAX_WINSHIFT    14  /* maximum window shift */
  97 
  98 /*
  99  * User-settable options (used with setsockopt).
 100  */
 101 #define TCP_NODELAY 0x01    /* don't delay send to coalesce packets */
 102 #define TCP_MAXSEG  0x02    /* set maximum segment size */
 103 
 104 #define SOL_TCP     6   /* TCP level */
 105 
 106 __END_DECLS
 107 
 108 #endif /* netinet/tcp.h */
