  23 void *
  24 lsearch (const void *key, void *base, size_t *nmemb, size_t size,
  25      __compar_fn_t compar)
  26 {
  27   void *result;
  28 
  29   /* Try to find it.  */
  30   result = lfind (key, base, nmemb, size, compar);
  31   if (result == NULL)
  32     {
  33       /* Not available.  Insert at the end.  */
  34       result = memcpy (base + (*nmemb) * size, key, size);
  35       ++(*nmemb);
  36     }
  37 
  38   return result;
  39 }

  42 void *
  43 lfind (const void *key, const void *base, size_t *nmemb, size_t size,
  44        __compar_fn_t compar)
  45 {
  46   const void *result = base;
  47   size_t cnt = 0;
  48 
  49   while (cnt < *nmemb && (*compar) (key, result) != 0)
  50     {
  51       result += size;
  52       ++cnt;
  53     }
  54 
  55   return cnt < *nmemb ? (void *) result : NULL;
  56 }
