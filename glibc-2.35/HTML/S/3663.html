  24 M_DECL_FUNC (__casin) (CFLOAT x)
  25 {
  26   CFLOAT res;
  27 
  28   if (isnan (__real__ x) || isnan (__imag__ x))
  29     {
  30       if (__real__ x == 0)
  31     {
  32       res = x;
  33     }
  34       else if (isinf (__real__ x) || isinf (__imag__ x))
  35     {
  36       __real__ res = M_NAN;
  37       __imag__ res = M_COPYSIGN (M_HUGE_VAL, __imag__ x);
  38     }
  39       else
  40     {
  41       __real__ res = M_NAN;
  42       __imag__ res = M_NAN;
  43     }
  44     }
  45   else
  46     {
  47       CFLOAT y;
  48 
  49       __real__ y = -__imag__ x;
  50       __imag__ y = __real__ x;
  51 
  52       y = M_SUF (__casinh) (y);
  53 
  54       __real__ res = __imag__ y;
  55       __imag__ res = -__real__ y;
  56     }
  57 
  58   return res;
  59 }
