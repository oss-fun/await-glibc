  25 /* Tests supposed to not match.  */
  26 struct
  27 {
  28   const char *pattern;
  29   const char *string;
  30   int flags, nmatch;
  31 } tests[] = {
  32   { "^<\\([^~]*\\)\\([^~]\\)[^~]*~\\1\\(.\\).*|=.*\\3.*\\2",
  33     "<,.8~2,~so-|=-~.0,123456789<><", REG_NOSUB, 0 },
  34   /* In ERE, all carets must be treated as anchors.  */
  35   { "a^b", "a^b", REG_EXTENDED, 0 }
  36 };
  37 
  38 int
  39 main (void)
  40 {
  41   regex_t re;
  42   regmatch_t rm[4];
  43   size_t i;
  44   int n, ret = 0;
  45 
  46   mtrace ();
  47 
  48   for (i = 0; i < sizeof (tests) / sizeof (tests[0]); ++i)
  49     {
  50       n = regcomp (&re, tests[i].pattern, tests[i].flags);
  51       if (n != 0)
  52     {
  53       char buf[500];
  54       regerror (n, &re, buf, sizeof (buf));
  55       printf ("regcomp %zd failed: %s\n", i, buf);
  56       ret = 1;
  57       continue;
  58     }
  59 
  60       if (! regexec (&re, tests[i].string, tests[i].nmatch,
  61              tests[i].nmatch ? rm : NULL, 0))
  62     {
  63       printf ("regexec %zd incorrectly matched\n", i);
  64       ret = 1;
  65     }
  66 
  67       regfree (&re);
  68     }
  69 
  70   return ret;
  71 }
