  25 static int
  26 do_test (void)
  27 {
  28   TEST_VERIFY_EXIT (setenv ("TZ", "PST8PDT,M3.2.0,M11.1.0", 1) == 0);
  29   tzset ();
  30   if (sizeof (time_t) > 4)
  31     {
  32       time_t ouch = (time_t) 2187810000LL;
  33       char buf[500];
  34       struct tm *tm = localtime (&ouch);
  35       TEST_VERIFY_EXIT (tm != NULL);
  36       TEST_VERIFY_EXIT (strftime (buf, sizeof buf, "%Y-%m-%d %H:%M:%S %Z", tm)
  37             > 0);
  38       puts (buf);
  39       TEST_VERIFY (strcmp (buf, "2039-04-30 14:00:00 PDT") == 0);
  40 
  41       /* Same as before but for localtime_r.  */
  42       struct tm tmd;
  43       tm = localtime_r (&ouch, &tmd);
  44       TEST_VERIFY_EXIT (tm == &tmd);
  45 
  46       TEST_VERIFY_EXIT (strftime (buf, sizeof buf, "%Y-%m-%d %H:%M:%S %Z", tm)
  47             > 0);
  48       puts (buf);
  49       TEST_VERIFY (strcmp (buf, "2039-04-30 14:00:00 PDT") == 0);
  50     }
  51   else
  52     FAIL_UNSUPPORTED ("32-bit time_t");
  53   return 0;
  54 }
