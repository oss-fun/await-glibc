  19 /* vfork() is just a special case of clone().  */
  20 
  21 #include <sys/asm.h>
  22 #include <sysdep.h>
  23 #define __ASSEMBLY__
  24 #include <asm/signal.h>
  25 
  26 #define CLONE_VM      0x00000100 /* Set if VM shared between processes.  */
  27 #define CLONE_VFORK   0x00004000 /* Set if the parent wants the child to
  28                     wake it up on mm_release.  */
  29 
  30     .text
  31 LEAF (__libc_vfork)
  32 
  33     li  a0, (CLONE_VFORK | CLONE_VM | SIGCHLD)
  34     mv  a1, sp
  35 
  36     li  a7, __NR_clone
  37     scall
  38 
  39     bltz    a0, 1f
  40     ret
  41 
  42 1:  j       __syscall_error
  43 END (__libc_vfork)
  44 
  45 weak_alias (__libc_vfork, vfork)
  46 strong_alias (__libc_vfork, __vfork)
  47 libc_hidden_def (__vfork)
