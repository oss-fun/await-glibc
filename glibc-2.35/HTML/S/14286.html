  25 /* Start THREAD.  Get the kernel thread scheduled and running.  */
  26 int
  27 __pthread_thread_start (struct __pthread *thread)
  28 {
  29   static int do_start;
  30   error_t err;
  31 
  32   if (!do_start)
  33     {
  34       /* The main thread is already running: do nothing.  */
  35       assert (__pthread_total == 1);
  36       assert ((
  37         {
  38           mach_port_t ktid = __mach_thread_self ();
  39           int ok = thread->kernel_thread == ktid;
  40           __mach_port_deallocate (__mach_task_self (),
  41                       thread->kernel_thread);
  42           ok;
  43         }));
  44       do_start = 1;
  45     }
  46   else
  47     {
  48       err = __thread_resume (thread->kernel_thread);
  49       assert_perror (err);
  50     }
  51 
  52   return 0;
  53 }
