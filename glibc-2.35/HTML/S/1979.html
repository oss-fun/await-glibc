  31 static char *
  32 stupid_strcasestr (const char *s1, const char *s2)
  33 {
  34   ssize_t s1len = strlen (s1);
  35   ssize_t s2len = strlen (s2);
  36 
  37   if (s2len > s1len)
  38     return NULL;
  39 
  40   for (ssize_t i = 0; i <= s1len - s2len; ++i)
  41     {
  42       size_t j;
  43       for (j = 0; j < s2len; ++j)
  44     if (tolower (s1[i + j]) != tolower (s2[j]))
  45       break;
  46       if (j == s2len)
  47     return (char *) s1 + i;
  48     }
  49 
  50   return NULL;
  51 }
