  19 /*
  20  * ALGORITHM DESCRIPTION:
  21  *   After computing exp(x) in high-low parts, an accurate computation is performed to obtain exp(x)-1
  22  *   Typical exp() implementation, except that:
  23  *    - tables are small (16 elements), allowing for fast gathers
  24  *    - all arguments processed in the main path
  25  *        - final VSCALEF assists branch-free design (correct overflow/underflow and special case responses)
  26  *        - a VAND is used to ensure the reduced argument |R|<2, even for large inputs
  27  *        - RZ mode used to avoid oveflow to +/-Inf for x*log2(e); helps with special case handling
  28  *
  29  *
  30  */
  31

  56 ENTRY(_ZGVeN8v_expm1_skx)
  57         pushq     %rbp
  58         cfi_def_cfa_offset(16)
  59         movq      %rsp, %rbp
  60         cfi_def_cfa(6, 16)
  61         cfi_offset(6, -16)
  62         andq      $-64, %rsp
  63         subq      $192, %rsp
  64         vmovups   L2E+__svml_dexpm1_data_internal_avx512(%rip), %zmm6
  65         vmovups   Shifter+__svml_dexpm1_data_internal_avx512(%rip), %zmm4
  66         vmovups   L2H+__svml_dexpm1_data_internal_avx512(%rip), %zmm11
  67         vmovups   L2L+__svml_dexpm1_data_internal_avx512(%rip), %zmm5
  68         vmovups   Threshold+__svml_dexpm1_data_internal_avx512(%rip), %zmm3
  69         vmovups   poly_coeff5+__svml_dexpm1_data_internal_avx512(%rip), %zmm13
  70         vmovups   poly_coeff4+__svml_dexpm1_data_internal_avx512(%rip), %zmm15
  71 
  72 /* polynomial */
  73         vmovups   poly_coeff7+__svml_dexpm1_data_internal_avx512(%rip), %zmm12
  74 
  75 /* set Z0=max(Z0, -128.0) */
  76         vmovups   ZThres+__svml_dexpm1_data_internal_avx512(%rip), %zmm8
  77         vmovups   poly_coeff3+__svml_dexpm1_data_internal_avx512(%rip), %zmm14
  78         vmovups   __svml_dexpm1_data_internal_avx512(%rip), %zmm9
  79         vmovaps   %zmm0, %zmm2
  80 
  81 /* 2^(52-4)*1.5 + x * log2(e) */
  82         vfmadd213pd {rn-sae}, %zmm4, %zmm2, %zmm6
  83         vmovups   Exp_tbl_L+__svml_dexpm1_data_internal_avx512(%rip), %zmm0
  84         vcmppd    $21, {sae}, %zmm3, %zmm2, %k0
  85 
  86 /* Z0 ~ x*log2(e), rounded to 4 fractional bits */
  87         vsubpd    {rn-sae}, %zmm4, %zmm6, %zmm7
  88         vpermt2pd Exp_tbl_H+64+__svml_dexpm1_data_internal_avx512(%rip), %zmm6, %zmm9
  89         vpermt2pd Exp_tbl_L+64+__svml_dexpm1_data_internal_avx512(%rip), %zmm6, %zmm0
  90         vandpd    SgnMask+__svml_dexpm1_data_internal_avx512(%rip), %zmm2, %zmm1
  91 
  92 /* R = x - Z0*log(2) */
  93         vfnmadd213pd {rn-sae}, %zmm2, %zmm7, %zmm11
  94         vmaxpd    {sae}, %zmm8, %zmm7, %zmm10
  95         vfnmadd231pd {rn-sae}, %zmm7, %zmm5, %zmm11
  96         kmovw     %k0, %edx
  97 
  98 /* ensure |R|<2 even for special cases */
  99         vandpd    EMask+__svml_dexpm1_data_internal_avx512(%rip), %zmm11, %zmm3
 100         vmovups   poly_coeff6+__svml_dexpm1_data_internal_avx512(%rip), %zmm11
 101 
 102 /* scale Th */
 103         vscalefpd {rn-sae}, %zmm10, %zmm9, %zmm4
 104         vfmadd231pd {rn-sae}, %zmm3, %zmm13, %zmm15
 105         vfmadd231pd {rn-sae}, %zmm3, %zmm12, %zmm11
 106         vmovups   poly_coeff2+__svml_dexpm1_data_internal_avx512(%rip), %zmm12
 107         vmulpd    {rn-sae}, %zmm3, %zmm3, %zmm13
 108         vfmadd231pd {rn-sae}, %zmm3, %zmm14, %zmm12
 109         vfmadd213pd {rn-sae}, %zmm15, %zmm13, %zmm11
 110         vfmadd213pd {rn-sae}, %zmm12, %zmm13, %zmm11
 111 
 112 /* Tlr + R+ R*Poly */
 113         vfmadd213pd {rn-sae}, %zmm0, %zmm13, %zmm11
 114 
 115 /* Th - 1 */
 116         vmovups   One+__svml_dexpm1_data_internal_avx512(%rip), %zmm0
 117         vaddpd    {rn-sae}, %zmm3, %zmm11, %zmm14
 118         vsubpd    {rn-sae}, %zmm0, %zmm4, %zmm15
 119 
 120 /* (Th-1)+Th*(Tlr + R+ R*Poly) */
 121         vfmadd213pd {rn-sae}, %zmm15, %zmm14, %zmm4
 122         vorpd     %zmm1, %zmm4, %zmm0
 123         testl     %edx, %edx
 124 
 125 /* Go to special inputs processing branch */
 126         jne       L(SPECIAL_VALUES_BRANCH)
 127                                 # LOE rbx r12 r13 r14 r15 edx zmm0 zmm2
 128 
 129 /* Restore registers
 130  * and exit the function
 131  */
 132 
 133 L(EXIT):
 134         movq      %rbp, %rsp
 135         popq      %rbp
 136         cfi_def_cfa(7, 8)
 137         cfi_restore(6)
 138         ret
 139         cfi_def_cfa(6, 16)
 140         cfi_offset(6, -16)
 141 
 142 /* Branch to process
 143  * special inputs
 144  */
 145 
 146 L(SPECIAL_VALUES_BRANCH):
 147         vmovups   %zmm2, 64(%rsp)
 148         vmovups   %zmm0, 128(%rsp)
 149                                 # LOE rbx r12 r13 r14 r15 edx zmm0

 158         movl      %eax, %r12d
 159         movq      %r13, 8(%rsp)
 160         /*  DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -184; DW_OP_plus)  */

 162         movl      %edx, %r13d
 163         movq      %r14, (%rsp)
 164         /*  DW_CFA_expression: r14 (r14) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -192; DW_OP_plus)  */
 165         .cfi_escape 0x10, 0x0e, 0x0e, 0x38, 0x1c, 0x0d, 0xc0, 0xff, 0xff, 0xff, 0x1a, 0x0d, 0x40, 0xff, 0xff, 0xff, 0x22
 166                                 # LOE rbx r15 r12d r13d
 167 
 168 /* Range mask
 169  * bits check
 170  */
 171 
 172 L(RANGEMASK_CHECK):
 173         btl       %r12d, %r13d
 174 
 175 /* Call scalar math function */
 176         jc        L(SCALAR_MATH_CALL)
 177                                 # LOE rbx r15 r12d r13d

 179 /* Special inputs
 180  * processing loop
 181  */

 196         cfi_restore(14)

 203         /*  DW_CFA_expression: r13 (r13) (DW_OP_lit8; DW_OP_minus; DW_OP_const4s: -64; DW_OP_and; DW_OP_const4s: -184; DW_OP_plus)  */
