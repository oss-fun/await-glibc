  37 do_test (void)

  41   unsigned char *sp;
  42   size_t size;

  46   struct sigaction sa;
  47   sa.sa_handler = handler;
  48   sa.sa_flags   = SA_RESTART | SA_ONSTACK;
  49   sigfillset (&sa.sa_mask);
  50   if (sigaction (SIGUSR1, &sa, 0))
  51     FAIL_RET ("sigaction (SIGUSR1, handler): %m\n");
  52 
  53   raise (SIGUSR1);
