  61   DWunion nn, dd;
  62   DWunion rr;

  72   n0 = nn.s.low;
  73   n1 = nn.s.high;

  76   if (d1 == 0)
  77     {
  78       if (d0 > n1)
  79     {
  80       /* 0q = nn / 0D */
  81 
  82       udiv_qrnnd (q0, n0, n1, n0, d0);
  83       q1 = 0;
  84 
  85       /* Remainder in n0.  */
  86     }
  87       else

  89       /* qq = NN / 0d */

 111     {

 113     {
 114       /* 0q = nn / 0D */
 115 
 116       count_leading_zeros (bm, d0);
 117 
 118       if (bm != 0)
 119         {
 120           /* Normalize, i.e. make the most significant bit of the
 121          denominator set.  */
 122 
 123           d0 = d0 << bm;
 124           n1 = (n1 << bm) | (n0 >> (W_TYPE_SIZE - bm));
 125           n0 = n0 << bm;
 126         }
 127 
 128       udiv_qrnnd (q0, n0, n1, n0, d0);
 129       q1 = 0;

 131       /* Remainder in n0 >> bm.  */
 132     }
 133       else
 134     {
 135       /* qq = NN / 0d */
 136 
 137       if (d0 == 0)

 148          This special case is necessary, not an optimization.

 164 
 165           udiv_qrnnd (q1, n1, n2, n1, d0);
 166         }
 167 
 168       /* n1 != d0...  */
 169

 171 
 172       /* Remainder in n0 >> bm.  */
 173     }
 174 
 175       if (rp != 0)
 176     {
 177       rr.s.low = n0 >> bm;

 180     }
 181     }
 182 #endif /* UDIV_NEEDS_NORMALIZATION */

 184   else
 185     {
 186       if (d1 > n1)
 187     {
 188       /* 00 = nn / DD */
 189 
 190       q0 = 0;
 191       q1 = 0;
 192 
 193       /* Remainder in n1n0.  */
 194       if (rp != 0)
 195         {
 196           rr.s.low = n0;
 197           rr.s.high = n1;
 198           *rp = rr.ll;
 199         }
 200     }

 203       /* 0q = NN / dd */
 204 
 205       count_leading_zeros (bm, d1);

 232         }

 234         {

 237

 239 
 240           d1 = (d1 << bm) | (d0 >> b);
 241           d0 = d0 << bm;
 242           n2 = n1 >> b;

 244           n0 = n0 << bm;

 247           umul_ppmm (m1, m0, q0, d0);

 250         {
 251           q0--;
 252           sub_ddmmss (m1, m0, m1, m0, d1, d0);
 253         }
 254 
 255           q1 = 0;

 261           rr.s.low = (n1 << b) | (n0 >> bm);
 262           rr.s.high = n1 >> bm;
 263           *rp = rr.ll;
 264         }
 265         }
 266     }
 267     }
