  25 /* Put the local address of FD into *ADDR and its length in *LEN.  */
  26 int
  27 __getsockname (int fd, __SOCKADDR_ARG addrarg, socklen_t *len)
  28 {
  29   error_t err;
  30   struct sockaddr *addr = addrarg.__sockaddr__;
  31   char *buf = (char *) addr;
  32   mach_msg_type_number_t buflen = *len;
  33   int type;
  34   addr_port_t aport;
  35 
  36   if (err = HURD_DPORT_USE (fd, __socket_name (port, &aport)))
  37     return __hurd_dfail (fd, err);
  38 
  39   err = __socket_whatis_address (aport, &type, &buf, &buflen);
  40   __mach_port_deallocate (__mach_task_self (), aport);
  41 
  42   if (err)
  43     return __hurd_dfail (fd, err);
  44 
  45   if (*len > buflen)
  46     *len = buflen;
  47 
  48   if (buf != (char *) addr)
  49     {
  50       memcpy (addr, buf, *len);
  51       __vm_deallocate (__mach_task_self (), (vm_address_t) buf, buflen);
  52     }
  53 
  54   addr->sa_family = type;
  55 
  56   return 0;
  57 }
