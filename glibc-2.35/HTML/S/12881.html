  10 /* acos = atan (sqrt(1 - x^2) / x) */
  11 ENTRY(__ieee754_acosf)
  12     flds    4(%esp)         /* x */
  13     fld %st
  14     fmul    %st(0)          /* x^2 */
  15     fld1
  16     fsubp               /* 1 - x^2 */
  17     fsqrt               /* sqrt (1 - x^2) */
  18     fabs
  19     fxch    %st(1)
  20     fpatan
  21     ret
  22 END (__ieee754_acosf)
