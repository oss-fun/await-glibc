  23 int
  24 pkey_set (int key, unsigned int rights)
  25 {
  26   if (key < 0 || key > PKEY_MAX || rights > 3)
  27     {
  28       __set_errno (EINVAL);
  29       return -1;
  30     }
  31 
  32   /* Translate to AMR bit values.  */
  33   unsigned long int bits;
  34   if (rights & PKEY_DISABLE_ACCESS)
  35     /* The PKEY_DISABLE_WRITE bit does not matter.  */
  36     bits = PKEY_AMR_READ | PKEY_AMR_WRITE;
  37   else if (rights == PKEY_DISABLE_WRITE)
  38     bits = PKEY_AMR_WRITE;
  39   else
  40     bits = 0;
  41 
  42   unsigned int index = pkey_index (key);
  43   unsigned long int mask = 3UL << index;
  44   unsigned long int amr = pkey_read ();
  45   amr = (amr & ~mask) | (bits << index);
  46   pkey_write (amr);
  47   return 0;
  48 }
